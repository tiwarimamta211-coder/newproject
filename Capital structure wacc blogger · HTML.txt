<script>
  var d = new Date();
  document.getElementById('reportDate').innerText = d.toLocaleDateString('en-IN', { day:'2-digit', month:'long', year:'numeric' });

  function fmt(v) {
    return new Intl.NumberFormat('en-IN', { style: 'currency', currency: 'INR', maximumFractionDigits: 0 }).format(v);
  }

  function sync() {
    var b = document.getElementById('bizName').value.trim();
    var r = document.getElementById('bizRef').value.trim();
    document.getElementById('dr_bizName').innerText = b || 'Enterprise Name';
    document.getElementById('dr_bizRef').innerText  = r ? 'Ref: ' + r : 'Ref: FY 2026 Audit';
  }

  function calc() {
    var eqAmt   = parseFloat(document.getElementById('equityAmt').value)  || 0;
    var ke      = parseFloat(document.getElementById('costEquity').value) || 0;
    var debtAmt = parseFloat(document.getElementById('debtAmt').value)    || 0;
    var kdPre   = parseFloat(document.getElementById('costDebt').value)   || 0;
    var tax     = parseFloat(document.getElementById('taxRate').value)    || 0;

    var totalCap = eqAmt + debtAmt;
    if (totalCap === 0) return;

    var wEq   = eqAmt   / totalCap;
    var wDebt = debtAmt / totalCap;
    var kdAfter = kdPre * (1 - (tax / 100));
    var wacc    = (wEq * ke) + (wDebt * kdAfter);
    var deRatio = eqAmt > 0 ? (debtAmt / eqAmt) : debtAmt;

    // Header
    document.getElementById('resTotalCap').innerText = fmt(totalCap);
    document.getElementById('resWACC').innerText     = wacc.toFixed(2) + '%';
    document.getElementById('resDERatio').innerText  = deRatio.toFixed(2);

    // Leverage status
    var el = document.getElementById('resStatus');
    if (deRatio > 2) {
      el.innerText   = '⚠ High Leverage';
      el.className   = 'status-badge status-high';
    } else if (deRatio < 0.5) {
      el.innerText   = '↓ Under Leveraged';
      el.className   = 'status-badge status-under';
    } else {
      el.innerText   = '✔ Optimal Mix';
      el.className   = 'status-badge status-optimal';
    }

    // Table
    document.getElementById('tabEqAmt').innerText     = fmt(eqAmt);
    document.getElementById('tabDebtAmt').innerText   = fmt(debtAmt);
    document.getElementById('tabTotalAmt').innerText  = fmt(totalCap);
    document.getElementById('tabEqWeight').innerText  = (wEq   * 100).toFixed(1) + '%';
    document.getElementById('tabDebtWeight').innerText= (wDebt * 100).toFixed(1) + '%';
    document.getElementById('tabEqCost').innerText    = ke.toFixed(2) + '%';
    document.getElementById('tabDebtCost').innerText  = kdAfter.toFixed(2) + '%';
  }

  calc();
</script>
