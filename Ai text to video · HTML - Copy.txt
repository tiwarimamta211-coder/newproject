<script>
// ── Voice loader ──────────────────────────────────────────────
let voices = [];
const voiceSelect = document.getElementById('voiceSelect');

function loadVoices() {
  voices = speechSynthesis.getVoices();
  voiceSelect.innerHTML = voices.map((v, i) =>
    `<option value="${i}">${v.name} (${v.lang})</option>`
  ).join('');
}
speechSynthesis.onvoiceschanged = loadVoices;
loadVoices();

// ── Status helper ─────────────────────────────────────────────
function setStatus(msg, active = false) {
  const s = document.getElementById('status');
  s.textContent = msg;
  s.className = 'status' + (active ? ' active' : '');
  s.insertAdjacentHTML('afterbegin', '<div class="status-dot"></div>');
}

// ── Narration ─────────────────────────────────────────────────
function speakText() {
  const text = document.getElementById('script').value.trim();
  if (!text) return setStatus('⚠ No script to narrate');
  speechSynthesis.cancel();
  const utt = new SpeechSynthesisUtterance(text);
  utt.voice = voices[voiceSelect.value] || null;
  utt.onend = () => setStatus('Narration complete');
  speechSynthesis.speak(utt);
  setStatus('Narrating…', true);
}
function stopNarration() { speechSynthesis.cancel(); setStatus('Stopped'); }

// ── Scene generator ───────────────────────────────────────────
let scenes = [], currentScene = 0, subtitleTimer = null;

function generateScenes() {
  const text = document.getElementById('script').value.trim();
  if (!text) return setStatus('⚠ Enter a script first');

  scenes = text.split(/[.!?]+/).map(s => s.trim()).filter(Boolean);
  const timeline = document.getElementById('timeline');
  timeline.innerHTML = scenes.map((s, i) => `
    <div class="scene ${i===0?'active':''}" onclick="jumpScene(${i})" id="sc${i}">
      <div class="scene-num">${String(i+1).padStart(2,'0')}</div>
      ${s.slice(0,40)}${s.length>40?'…':''}
    </div>
  `).join('');

  const badge = document.getElementById('sceneBadge');
  badge.style.display = 'block';
  badge.textContent = `${scenes.length} SCENES`;

  clearInterval(subtitleTimer);
  currentScene = 0;
  showScene(0);
  subtitleTimer = setInterval(() => {
    currentScene = (currentScene + 1) % scenes.length;
    showScene(currentScene);
  }, 3000);

  setStatus(`${scenes.length} scenes generated`, true);
}

function showScene(i) {
  document.getElementById('subtitle').textContent = scenes[i] || '';
  document.querySelectorAll('.scene').forEach((el, j) => {
    el.classList.toggle('active', j === i);
  });
}

function jumpScene(i) {
  clearInterval(subtitleTimer);
  currentScene = i;
  showScene(i);
  subtitleTimer = setInterval(() => {
    currentScene = (currentScene + 1) % scenes.length;
    showScene(currentScene);
  }, 3000);
}

// ── Background upload ─────────────────────────────────────────
document.getElementById('bgUpload').addEventListener('change', e => {
  const file = e.target.files[0];
  if (!file) return;
  const reader = new FileReader();
  reader.onload = ev => {
    document.getElementById('bgImage').src = ev.target.result;
    document.querySelector('.file-label').textContent = '✓ ' + file.name;
  };
  reader.readAsDataURL(file);
});

// ── SEO generator ─────────────────────────────────────────────
function generateSEO() {
  const title = document.getElementById('title').value.trim();
  const script = document.getElementById('script').value.trim();
  if (!script) return setStatus('⚠ Enter a script first');

  const words = script.toLowerCase().replace(/[^a-z0-9 ]/g, '').split(/\s+/);
  const freq = {};
  words.forEach(w => { if (w.length > 4) freq[w] = (freq[w] || 0) + 1; });
  const keywords = Object.entries(freq).sort((a,b)=>b[1]-a[1]).slice(0,10).map(([w])=>w);
  const desc = script.length > 250 ? script.slice(0,250) + '…' : script;

  const box = document.getElementById('seoOutput');
  box.className = 'seo-box visible';
  box.innerHTML = `
    <div class="seo-row">
      <div class="seo-key">Title</div>
      <div class="seo-val">${title || '[Add a title above]'}</div>
    </div>
    <div class="seo-row">
      <div class="seo-key">Description</div>
      <div class="seo-val">${desc}</div>
    </div>
    <div class="seo-row">
      <div class="seo-key">Tags</div>
      <div class="seo-tags">${keywords.map(k=>`<span class="tag">#${k}</span>`).join('')}</div>
    </div>
    <div class="seo-row">
      <div class="seo-key">Scenes</div>
      <div class="seo-val">${scenes.length || 'Generate scenes first'} total scenes</div>
    </div>
  `;
  document.getElementById('noSeo').style.display = 'none';
  setStatus('SEO metadata generated', true);
}

// ── Thumbnail generator ───────────────────────────────────────
function generateThumbnail() {
  const title = document.getElementById('title').value.trim() || 'Your Video Title';
  const canvas = document.getElementById('thumbCanvas');
  const ctx = canvas.getContext('2d');
  const img = document.getElementById('bgImage');
  const wrap = document.getElementById('thumbWrap');

  function draw() {
    ctx.drawImage(img, 0, 0, 1280, 720);
    // Dark overlay
    const grad = ctx.createLinearGradient(0,0,0,720);
    grad.addColorStop(0, 'rgba(0,0,0,0.3)');
    grad.addColorStop(1, 'rgba(0,0,0,0.82)');
    ctx.fillStyle = grad;
    ctx.fillRect(0,0,1280,720);
    // Accent line
    ctx.fillStyle = '#00e5ff';
    ctx.fillRect(80, 580, 220, 5);
    // Title text
    ctx.fillStyle = '#ffffff';
    ctx.font = 'bold 72px Arial';
    ctx.fillText(title.slice(0,28), 80, 560);
    // Sub text
    ctx.fillStyle = '#00e5ff';
    ctx.font = '28px monospace';
    ctx.fillText('AI GENERATED VIDEO', 80, 630);
    wrap.className = 'thumb-canvas visible';
    setStatus('Thumbnail generated', true);
  }

  if (img.complete && img.naturalWidth > 0) {
    draw();
  } else {
    img.onload = draw;
    setStatus('Loading image…');
  }
}

function downloadThumbnail() {
  const link = document.createElement('a');
  link.download = 'thumbnail.png';
  link.href = document.getElementById('thumbCanvas').toDataURL('image/png');
  link.click();
}

// ── Script download ───────────────────────────────────────────
function downloadScript() {
  const text = document.getElementById('script').value.trim();
  if (!text) return setStatus('⚠ Nothing to download');
  const blob = new Blob([text], { type: 'text/plain' });
  const link = document.createElement('a');
  link.href = URL.createObjectURL(blob);
  link.download = 'video-script.txt';
  link.click();
  setStatus('Script downloaded');
}
