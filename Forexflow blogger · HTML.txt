<script>
  // ── CONFIG ──────────────────────────────────────────────────
  var API_KEY = '626d833959838031c5874403'; // ExchangeRate-API free key
  // ────────────────────────────────────────────────────────────

  var currencies = [
    'USD','EUR','GBP','JPY','AUD','CAD','CHF','CNY','INR',
    'BRL','ZAR','SGD','HKD','MXN','NOK','SEK','DKK','NZD',
    'KRW','TRY','SAR','AED','MYR','THB','IDR','PKR','BDT'
  ];

  // Fallback mock rates (USD base — updated March 2026 approximate)
  var mockRates = {
    USD:1, EUR:0.921, GBP:0.787, JPY:149.8, AUD:1.534, CAD:1.358,
    CHF:0.898, CNY:7.243, INR:83.12, BRL:4.972, ZAR:18.63, SGD:1.341,
    HKD:7.821, MXN:17.15, NOK:10.52, SEK:10.44, DKK:6.876, NZD:1.627,
    KRW:1325, TRY:30.54, SAR:3.751, AED:3.673, MYR:4.712, THB:35.42,
    IDR:15680, PKR:278.5, BDT:109.7
  };

  var liveRates  = null;
  var ratesBase  = 'USD';
  var lastFetch  = null;

  var popularPairs = [
    {from:'USD',to:'INR'}, {from:'EUR',to:'INR'}, {from:'GBP',to:'INR'},
    {from:'USD',to:'EUR'}, {from:'USD',to:'GBP'}, {from:'AED',to:'INR'}
  ];

  // Populate selects
  function initSelects() {
    var f = document.getElementById('fromCurrency');
    var t = document.getElementById('toCurrency');
    currencies.forEach(function(c) {
      f.add(new Option(c + ' – ' + getCurrencyName(c), c));
      t.add(new Option(c + ' – ' + getCurrencyName(c), c));
    });
    f.value = 'USD';
    t.value = 'INR';
  }

  function getCurrencyName(code) {
    var names = {
      USD:'US Dollar', EUR:'Euro', GBP:'Brit. Pound', JPY:'Japanese Yen',
      AUD:'Aus. Dollar', CAD:'Can. Dollar', CHF:'Swiss Franc', CNY:'Chinese Yuan',
      INR:'Indian Rupee', BRL:'Brazilian Real', ZAR:'S. African Rand',
      SGD:'Singapore $', HKD:'Hong Kong $', MXN:'Mexican Peso',
      AED:'UAE Dirham', SAR:'Saudi Riyal', MYR:'Malaysian Ringgit',
      THB:'Thai Baht', KRW:'South Korean Won', TRY:'Turkish Lira',
      NOK:'Norwegian Kr', SEK:'Swedish Kr', DKK:'Danish Kr',
      NZD:'NZ Dollar', IDR:'Indonesian Rp', PKR:'Pakistani Rs', BDT:'Bangladeshi Tk'
    };
    return names[code] || code;
  }

  function getRate() {
    var from = document.getElementById('fromCurrency').value;
    var btn  = document.getElementById('convertBtn');
    btn.disabled = true;
    btn.innerHTML = '<span class="spin">🔄</span> Syncing...';

    fetch('https://v6.exchangerate-api.com/v6/' + API_KEY + '/latest/' + from)
      .then(function(r){ return r.json(); })
      .then(function(data) {
        if (data.result === 'success') {
          liveRates  = data.conversion_rates;
          ratesBase  = from;
          lastFetch  = new Date();
          document.getElementById('statusBar').innerHTML = 'Live rates fetched • <span>✅ API Active</span>';
          convert(true);
        } else {
          useMockRates(from);
        }
      })
      .catch(function() { useMockRates(from); })
      .finally(function() {
        btn.disabled = false;
        btn.innerHTML = '🔄 Update Conversion';
      });
  }

  function useMockRates(from) {
    liveRates = {};
    var fromRate = mockRates[from] || 1;
    Object.keys(mockRates).forEach(function(k) {
      liveRates[k] = mockRates[k] / fromRate;
    });
    ratesBase = from;
    lastFetch = new Date();
    document.getElementById('statusBar').innerHTML = 'Demo rates (approx) • <span>Add API Key for live</span>';
    convert(false);
  }

  function convert(isLive) {
    var amount = parseFloat(document.getElementById('amount').value) || 0;
    var from   = document.getElementById('fromCurrency').value;
    var to     = document.getElementById('toCurrency').value;

    if (!liveRates || from !== ratesBase) { getRate(); return; }

    var rate  = liveRates[to] || (mockRates[to] / (mockRates[from] || 1));
    var total = amount * rate;

    // Format large numbers nicely
    var formatted = total >= 1000
      ? total.toLocaleString('en-IN', {minimumFractionDigits:2, maximumFractionDigits:2})
      : total.toFixed(4);

    document.getElementById('convertedAmount').innerText = formatted + ' ' + to;
    document.getElementById('rateInfo').innerText =
      '1 ' + from + ' = ' + rate.toFixed(4) + ' ' + to;
    document.getElementById('rateTime').innerText =
      isLive ? '⚡ Live rate • ' + lastFetch.toLocaleTimeString() : '📊 Indicative rate';

    renderPairs();
  }

  function convertNow() {
    if (liveRates && document.getElementById('fromCurrency').value === ratesBase) {
      convert(!!lastFetch);
    } else {
      getRate();
    }
  }

  function swapCurrencies() {
    var f = document.getElementById('fromCurrency');
    var t = document.getElementById('toCurrency');
    var tmp = f.value; f.value = t.value; t.value = tmp;
    getRate();
  }

  function renderPairs() {
    var grid = document.getElementById('pairsGrid');
    grid.innerHTML = '';
    popularPairs.forEach(function(p) {
      var fromR = mockRates[p.from] || 1;
      var toR   = mockRates[p.to]   || 1;
      // Use live rate if available
      var rate;
      if (liveRates && ratesBase === p.from && liveRates[p.to]) {
        rate = liveRates[p.to];
      } else {
        rate = toR / fromR;
      }
      var chip = document.createElement('div');
      chip.className = 'pair-chip';
      chip.innerHTML =
        '<div class="pair-name">' + p.from + ' → ' + p.to + '</div>' +
        '<div class="pair-rate">' + rate.toFixed(2) + '</div>';
      chip.onclick = (function(pf, pt) {
        return function() {
          document.getElementById('fromCurrency').value = pf;
          document.getElementById('toCurrency').value   = pt;
          getRate();
        };
      })(p.from, p.to);
      grid.appendChild(chip);
    });
  }

  // INIT
  initSelects();
  renderPairs();
  getRate();
</script>
