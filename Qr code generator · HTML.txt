// ======= BUILD QR CONTENT =======
function buildContent() {
  switch(currentType) {
    case 'url': return document.getElementById('url-input').value.trim() || 'https://example.com';
    case 'text': return document.getElementById('text-input').value.trim() || 'Hello World';
    case 'wifi': {
      const s = document.getElementById('wifi-ssid').value.trim();
      const p = document.getElementById('wifi-pass').value;
      const sec = document.getElementById('wifi-sec').value;
      return `WIFI:T:${sec};S:${s};P:${p};;`;
    }
    case 'vcard': {
      const fn = document.getElementById('vc-fname').value.trim();
      const ln = document.getElementById('vc-lname').value.trim();
      const ph = document.getElementById('vc-phone').value.trim();
      const em = document.getElementById('vc-email').value.trim();
      const web = document.getElementById('vc-web').value.trim();
      const org = document.getElementById('vc-org').value.trim();
      return `BEGIN:VCARD\nVERSION:3.0\nN:${ln};${fn}\nFN:${fn} ${ln}\nTEL:${ph}\nEMAIL:${em}\nURL:${web}\nORG:${org}\nEND:VCARD`;
    }
    case 'email': {
      const to = document.getElementById('email-to').value.trim();
      const sub = encodeURIComponent(document.getElementById('email-sub').value.trim());
      const body = encodeURIComponent(document.getElementById('email-body').value.trim());
      return `mailto:${to}?subject=${sub}&body=${body}`;
    }
    case 'sms': {
      const ph = document.getElementById('sms-phone').value.trim();
      const msg = encodeURIComponent(document.getElementById('sms-msg').value.trim());
      return `sms:${ph}?body=${msg}`;
    }
    case 'upi': {
      const id = document.getElementById('upi-id').value.trim();
      const name = encodeURIComponent(document.getElementById('upi-name').value.trim());
      const amt = document.getElementById('upi-amt').value;
      const note = encodeURIComponent(document.getElementById('upi-note').value.trim());
      let url = `upi://pay?pa=${id}&pn=${name}`;
      if(amt) url += `&am=${amt}&cu=INR`;
      if(note) url += `&tn=${note}`;
      return url;
    }
    case 'geo': {
      const lat = document.getElementById('geo-lat').value.trim();
      const lng = document.getElementById('geo-lng').value.trim();
      const label = document.getElementById('geo-label').value.trim();
      return label ? `geo:${lat},${lng}?q=${lat},${lng}(${encodeURIComponent(label)})` : `geo:${lat},${lng}`;
    }
    default: return 'https://example.com';
  }
}

// ======= GENERATE QR =======
function generateQR() {
  const content = buildContent();
  lastContent = content;
  const size = parseInt(document.getElementById('qr-size').value);
  const fg = fgColor.value;
  const bg = bgColor.value;

  const wrap = document.getElementById('qr-output');
  wrap.innerHTML = '';

  new QRCode(wrap, {
    text: content,
    width: size,
    height: size,
    colorDark: fg,
    colorLight: bg,
    correctLevel: QRErrorCorrectLevel_map[eclLevel]
  });

  // If dot style, apply post-processing (visual)
  if (qrStyle === 'dots') {
    setTimeout(() => {
      const canvas = wrap.querySelector('canvas');
      if (canvas) applyDotStyle(canvas, fg, bg);
    }, 100);
  }

  document.getElementById('qr-placeholder').style.display = 'none';
  const canvasWrap = document.getElementById('qr-canvas-wrap');
  canvasWrap.classList.add('visible');
  canvasWrap.querySelector('#qr-output').classList.add('qr-pop');
  setTimeout(() => canvasWrap.querySelector('#qr-output').classList.remove('qr-pop'), 400);

  document.getElementById('qr-size-tag').textContent = `${size} × ${size} px`;
  document.getElementById('download-row').style.display = 'grid';

  // Save to history
  saveHistory(content, currentType);

  showToast('✅ QR Code generated!');
}

function applyDotStyle(canvas, fg, bg) {
  const ctx = canvas.getContext('2d');
  const w = canvas.width, h = canvas.height;
  const imgData = ctx.getImageData(0, 0, w, h);
  const data = imgData.data;
  const out = ctx.createImageData(w, h);
  const od = out.data;
  // copy bg
  for (let i = 0; i < od.length; i++) od[i] = data[i];
  ctx.clearRect(0, 0, w, h);
  const bgR = parseInt(bg.slice(1,3),16), bgG = parseInt(bg.slice(3,5),16), bgB = parseInt(bg.slice(5,7),16);
  const fgR = parseInt(fg.slice(1,3),16), fgG = parseInt(fg.slice(3,5),16), fgB = parseInt(fg.slice(5,7),16);
  ctx.fillStyle = bg;
  ctx.fillRect(0,0,w,h);
  const step = Math.floor(w/25) || 1;
  for (let y = 0; y < h; y += step) {
    for (let x = 0; x < w; x += step) {
      const idx = (y * w + x) * 4;
      const isDark = data[idx] < 128;
      if (isDark) {
        ctx.fillStyle = fg;
        ctx.beginPath();
        ctx.arc(x + step/2, y + step/2, step/2 * 0.82, 0, Math.PI*2);
        ctx.fill();
      }
    }
  }
}

// ======= HISTORY =======
function saveHistory(content, type) {
  const canvas = document.querySelector('#qr-output canvas');
  const thumb = canvas ? canvas.toDataURL('image/png') : '';
  const entry = { content, type, thumb, ts: Date.now() };
  history = history.filter(h => h.content !== content);
  history.unshift(entry);
  if (history.length > 10) history = history.slice(0, 10);
  localStorage.setItem('qrforge_history', JSON.stringify(history));
  renderHistory();
}

function renderHistory() {
  const list = document.getElementById('history-list');
  if (!history.length) {
    list.innerHTML = '<div class="history-empty">No QR codes yet — generate your first one!</div>';
    return;
  }
  list.innerHTML = history.map((h, i) => `
    <div class="history-item" data-i="${i}">
      <div class="history-thumb">
        ${h.thumb ? `<img src="${h.thumb}" style="width:100%;height:100%;display:block;" alt="">` : '▦'}
      </div>
      <div class="history-info">
        <div class="history-type">${h.type.toUpperCase()}</div>
        <div class="history-val">${h.content.length > 50 ? h.content.slice(0,50)+'…' : h.content}</div>
      </div>
      <button class="history-del" data-del="${i}" title="Remove">✕</button>
    </div>
  `).join('');

  list.querySelectorAll('.history-del').forEach(btn => {
    btn.addEventListener('click', e => {
      e.stopPropagation();
      history.splice(parseInt(btn.dataset.del), 1);
      localStorage.setItem('qrforge_history', JSON.stringify(history));
      renderHistory();
    });
  });
}

document.getElementById('clear-history').addEventListener('click', () => {
