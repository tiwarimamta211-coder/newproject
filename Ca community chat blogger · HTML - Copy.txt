<script>
  var bannedWords = ['spam', 'scam', 'abuse', 'attack', 'fake'];

  var roleColors = {
    'CA':               { name: '#1d4ed8', badge: 'badge-ca',   avatar: '#eff6ff', acolor: '#1d4ed8', label: 'CA' },
    'GST Practitioner': { name: '#15803d', badge: 'badge-gst',  avatar: '#f0fdf4', acolor: '#15803d', label: 'GST Prac.' },
    'Accountant':       { name: '#7e22ce', badge: 'badge-user', avatar: '#fdf4ff', acolor: '#7e22ce', label: 'Accountant' },
    'Student':          { name: '#b45309', badge: 'badge-user', avatar: '#fffbeb', acolor: '#b45309', label: 'Student' },
    'Member':           { name: '#334155', badge: 'badge-user', avatar: '#f8fafc', acolor: '#64748b', label: 'Member' }
  };

  function fillMsg(text) {
    document.getElementById('msgInput').value = text;
    document.getElementById('msgInput').focus();
  }

  function handleKey(e) { if (e.key === 'Enter') sendMessage(); }

  function sendMessage() {
    var input   = document.getElementById('msgInput');
    var nameEl  = document.getElementById('yourName');
    var roleEl  = document.getElementById('yourRole');
    var chatWin = document.getElementById('chat-window');

    var rawText  = input.value.trim();
    var userName = nameEl.value.trim() || 'You';
    var userRole = roleEl.value;
    if (!rawText) return;

    // Moderation
    var isModerated = false;
    var filtered = rawText;
    bannedWords.forEach(function(w) {
      var re = new RegExp(w, 'gi');
      if (re.test(filtered)) { filtered = filtered.replace(re, '****'); isModerated = true; }
    });

    var time = new Date().toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' });
    var rc   = roleColors[userRole] || roleColors['Member'];
    var initials = userName.split(' ').map(function(w){ return w[0]; }).join('').substring(0,2).toUpperCase();

    // Build message
    var msgDiv = document.createElement('div');
    msgDiv.className = 'message self';
    msgDiv.innerHTML =
      '<div class="avatar" style="background:' + rc.avatar + ';color:' + rc.acolor + '">' + initials + '</div>' +
      '<div class="msg-body">' +
        '<div class="msg-meta">' +
          '<span class="msg-name" style="color:' + rc.name + '">' + escHtml(userName) + '</span>' +
          '<span class="msg-role-badge ' + rc.badge + '">' + rc.label + '</span>' +
          '<span class="msg-time">' + time + '</span>' +
        '</div>' +
        '<div class="msg-text sent-bubble' + (isModerated ? ' moderated' : '') + '">' + escHtml(filtered) + '</div>' +
      '</div>';
    chatWin.appendChild(msgDiv);

    if (isModerated) {
      setTimeout(function() {
        var sysDiv = document.createElement('div');
        sysDiv.className = 'system-msg warn';
        sysDiv.innerText = '⚠ Your message was filtered. Please follow community guidelines.';
        chatWin.appendChild(sysDiv);
        chatWin.scrollTop = chatWin.scrollHeight;
      }, 600);
    }

    input.value = '';
    chatWin.scrollTop = chatWin.scrollHeight;
  }

  function escHtml(t) {
    return t.replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;');
  }
</script>
