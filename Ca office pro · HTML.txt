// ─── ADD VOUCHER ─────────────────────────────────────────────────────────────
function addVoucher(){
  const vType    = document.getElementById("vType").value;
  const vDate    = document.getElementById("vDate").value;
  const vParty   = document.getElementById("vParty").value.trim();
  const vGSTIN   = document.getElementById("vGSTIN").value.trim();
  const vInvoice = document.getElementById("vInvoice").value.trim();
  const vLedger  = document.getElementById("vLedger").value.trim();
  const vAmount  = parseFloat(document.getElementById("vAmount").value)||0;
  const vGSTRate = parseFloat(document.getElementById("vGSTRate").value)||0;
  const vTxType  = document.getElementById("vTxType").value;
  const vHSN     = document.getElementById("vHSN").value.trim();
  const vNarration=document.getElementById("vNarration").value.trim();

  if(!vDate){ toast("❌ Date जरूरी है"); return; }
  if(!vParty){ toast("❌ Party का नाम जरूरी है"); return; }
  if(vAmount<=0){ toast("❌ Amount 0 से ज्यादा होना चाहिए"); return; }

  const gst = vAmount * vGSTRate/100;
  const isInter = vTxType==="inter";
  const v = {
    id:uid(), type:vType, date:vDate, party:vParty, gstin:vGSTIN,
    invoice:vInvoice||"—", ledger:vLedger, amount:vAmount,
    gstRate:vGSTRate, txType:vTxType, hsn:vHSN,
    cgst: isInter?0:gst/2, sgst:isInter?0:gst/2, igst:isInter?gst:0,
    totalGST:gst, total:vAmount+gst, narration:vNarration,
    createdAt:new Date().toISOString()
  };
  vouchers.unshift(v);
  save();
  renderVouchers(vouchers);
  updateStats();
  clearForm();
  toast("✅ Entry save हुई! — "+vParty+" — "+fmt(v.total));
}

// ─── GST SUMMARY ─────────────────────────────────────────────────────────────
function renderGSTSummary(){
  const month = document.getElementById("gstMonth").value;
  const filtered = month ? vouchers.filter(v=>v.date.startsWith(month)) : vouchers;

  const outward = filtered.filter(v=>v.type==="Sales"||v.type==="Debit Note");
  const inward  = filtered.filter(v=>v.type==="Purchase"||v.type==="Credit Note");

  const outTax = outward.reduce((s,v)=>s+v.totalGST,0);
  const inTax  = inward.reduce((s,v)=>s+v.totalGST,0);
  const net    = Math.max(0, outTax-inTax);

  const outCGST= outward.reduce((s,v)=>s+v.cgst,0);
  const outSGST= outward.reduce((s,v)=>s+v.sgst,0);
  const outIGST= outward.reduce((s,v)=>s+v.igst,0);
  const inCGST = inward.reduce((s,v)=>s+v.cgst,0);
  const inSGST = inward.reduce((s,v)=>s+v.sgst,0);
  const inIGST = inward.reduce((s,v)=>s+v.igst,0);

  setVal("g31a",  fmt(outward.reduce((s,v)=>s+v.amount,0)));
  setVal("g4a",   fmt(inward.reduce((s,v)=>s+v.amount,0)));
  setVal("gNet",  fmt(net));
  setVal("gCGST", fmt(Math.max(0,outCGST-inCGST)));
  setVal("gSGST", fmt(Math.max(0,outSGST-inSGST)));
  setVal("gIGST", fmt(Math.max(0,outIGST-inIGST)));

  // HSN Summary
  const hsnMap = {};
  filtered.forEach(v=>{
    const k = v.hsn||"N/A";
    if(!hsnMap[k]) hsnMap[k]={hsn:k,desc:v.ledger,taxable:0,rates:new Set(),cgst:0,sgst:0,igst:0};
    hsnMap[k].taxable+=v.amount;
    hsnMap[k].rates.add(v.gstRate+"%");
    hsnMap[k].cgst+=v.cgst;
    hsnMap[k].sgst+=v.sgst;
    hsnMap[k].igst+=v.igst;
  });
  const hsnRows = Object.values(hsnMap);
  const hsnTb = document.getElementById("hsnTable");
  if(!hsnRows.length){
    hsnTb.innerHTML='<tr><td colspan="7" style="text-align:center;color:#6b7280;padding:20px;">Data नहीं है।</td></tr>';
  } else {
    hsnTb.innerHTML = hsnRows.map(h=>`
      <tr>
        <td style="font-weight:700;color:#60a5fa;">${h.hsn}</td>
        <td>${h.desc||"—"}</td>
        <td style="font-weight:700;">${fmt(h.taxable)}</td>
        <td>${[...h.rates].join(", ")}</td>
        <td style="color:#a78bfa;">${fmt(h.cgst)}</td>
        <td style="color:#a78bfa;">${fmt(h.sgst)}</td>
        <td style="color:#4dc3ff;">${fmt(h.igst)}</td>
      </tr>
    `).join("");
  }

  // B2B Table
  const b2bTb = document.getElementById("b2bTable");
  const salesList = filtered.filter(v=>(v.type==="Sales")&&v.gstin);
  if(!salesList.length){
    b2bTb.innerHTML='<tr><td colspan="7" style="text-align:center;color:#6b7280;padding:20px;">Sales entries नहीं हैं।</td></tr>';
  } else {
    b2bTb.innerHTML = salesList.map(v=>`
      <tr>
        <td style="color:#4a90d9;font-size:11px;">${v.gstin}</td>
        <td style="font-weight:700;">${v.party}</td>
        <td>${v.invoice}</td>
        <td>${v.date}</td>
        <td>${fmt(v.amount)}</td>
        <td style="color:#a78bfa;">${fmt(v.totalGST)}</td>
        <td style="font-weight:700;color:#00e5a0;">${fmt(v.total)}</td>
      </tr>
    `).join("");
  }
}

        <LEDGERNAME>${isPurchase?"Input IGST":"Output IGST"}</LEDGERNAME>
        <ISDEEMEDPOSITIVE>${isPurchase?"Yes":"No"}</ISDEEMEDPOSITIVE>
        <AMOUNT>${(s * v.igst).toFixed(2)}</AMOUNT>
      </ALLLEDGERENTRIES.LIST>`:""}
    </VOUCHER>`;
  }).join("\n");

  const xml = `<?xml version="1.0" encoding="UTF-8"?>
<ENVELOPE>

${voucherXML}

  const list  = month ? vouchers.filter(v=>v.date.startsWith(month)) : vouchers;
  if(!list.length){ toast("❌ कोई entry नहीं है"); return; }
  const rows  = [["Date","Type","Party","GSTIN","Invoice#","Ledger","Taxable","Rate","CGST","SGST","IGST","Total","HSN","Narration"]];
  list.forEach(v=>rows.push([v.date,v.type,`"${v.party}"`,v.gstin,v.invoice,`"${v.ledger}"`,v.amount,v.gstRate+"%",v.cgst.toFixed(2),v.sgst.toFixed(2),v.igst.toFixed(2),v.total.toFixed(2),v.hsn,`"${v.narration}"`]));
  const csv = rows.map(r=>r.join(",")).join("\n");
  const blob = new Blob(["\uFEFF"+csv],{type:"text/csv;charset=utf-8;"});
  const url = URL.createObjectURL(blob);
  const a = document.createElement("a"); a.href=url; a.download=`Vouchers_${month||"All"}.csv`; a.click();
  toast("✅ CSV download हुई!");
}
