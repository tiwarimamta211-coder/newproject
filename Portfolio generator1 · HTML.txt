          <input class="form-input" value="${e.date}" oninput="state.experiences[${i}].date=this.value; render();">
        </div>
        <div class="form-group">
          <label class="form-label">Job Title</label>
          <input class="form-input" value="${e.role}" oninput="state.experiences[${i}].role=this.value; render();">
        </div>
        <div class="form-group">
          <label class="form-label">Company</label>
          <input class="form-input" value="${e.company}" oninput="state.experiences[${i}].company=this.value; render();">
        </div>
        <div class="form-group">
          <label class="form-label">Description</label>
          <textarea class="form-textarea" rows="2" oninput="state.experiences[${i}].desc=this.value; render();">${e.desc}</textarea>
        </div>
        <button class="btn-sm btn-danger" style="margin-top:6px;" onclick="removeExp(${i})">✕ Remove</button>
      </div>
    </div>
  `).join('');
}

function render() {

  const location = val('iLocation') || '';
  const github   = val('iGithub')   || '#';
  const linkedin = val('iLinkedin') || '#';
  const twitter  = val('iTwitter')  || '';

  const clients  = val('iClients')  || '32';

  // NAV

      <div class="pf-stat"><div class="pf-stat-num">${clients}<em style="color:var(--pf-accent);">+</em></div><div class="pf-stat-label">Clients</div></div>
      <div class="pf-stat"><div class="pf-stat-num">${awards}</div><div class="pf-stat-label">Awards</div></div>
    </div>`;
  }

  // SKILLS
  if (state.showSkills && (state.skills.length || state.tags.length)) {
    html += `<div class="pf-section" style="padding:${state.spacing * 0.8}px 5%;">
      <div class="pf-section-header">
        <div class="pf-section-label">Expertise</div>
        <h2 class="pf-section-title">Skills & Tools</h2>
      </div>`;
    if (state.skills.length) {
      html += `<div class="pf-skills-grid">`;
      state.skills.forEach(s => {
        html += `<div class="pf-skill-card" style="border-radius:${state.borderRadius}px;">

        </div>`;
      });
      html += `</div>`;
    }
    if (state.tags.length) {
      html += `<div style="display:flex; flex-wrap:wrap; gap:8px; margin-top:20px;">`;
      state.tags.forEach(t => {
        html += `<span style="background:var(--pf-bg2); border:1px solid var(--pf-border); border-radius:20px; padding:5px 14px; font-size:0.78rem; font-weight:600; color:var(--pf-ink2);">${t}</span>`;
      });
      html += `</div>`;
    }
    html += `</div>`;
  }

  // PROJECTS
  if (state.showProjects && state.projects.length) {
    html += `<div class="pf-section" style="padding:${state.spacing * 0.8}px 5%; background:var(--pf-bg2); max-width:none;">
      <div style="max-width:1100px; margin:0 auto;">
      <div class="pf-section-header">
        <div class="pf-section-label">Portfolio</div>
        <h2 class="pf-section-title">Selected Work</h2>
      </div>
      <div class="pf-projects-grid">`;
    state.projects.forEach(p => {
      html += `<div class="pf-project-card" style="border-radius:${state.borderRadius + 4}px;">
        <div class="pf-project-thumb" style="font-size:2.5rem;">${p.emoji}</div>
        <div class="pf-project-body">
          <div class="pf-project-category" style="color:var(--pf-accent);">${p.category}</div>
          <div class="pf-project-name">${p.name}</div>
          <div class="pf-project-desc">${p.desc}</div>
          <div class="pf-project-tags">`;
      p.tags.forEach(t => {
        html += `<span class="pf-tag" style="border-radius:20px;">${t}</span>`;
      });
      html += `</div></div></div>`;
    });
    html += `</div></div></div>`;
  }

  // EXPERIENCE
  if (state.showExperience && state.experiences.length) {
    html += `<div class="pf-section" style="padding:${state.spacing * 0.8}px 5%;">
      <div class="pf-section-header">
        <div class="pf-section-label">Career</div>
        <h2 class="pf-section-title">Experience</h2>
      </div>
      <div class="pf-exp-list">`;
    state.experiences.forEach(e => {
      html += `<div class="pf-exp-item">
        <div class="pf-exp-date">${e.date}</div>
        <div>
          <div class="pf-exp-role">${e.role}</div>
          <div class="pf-exp-company" style="color:var(--pf-accent);">${e.company}</div>
          <div class="pf-exp-desc">${e.desc}</div>
        </div>
      </div>`;
    });
    html += `</div></div>`;
  }

  // CONTACT
  if (state.showContact) {
    html += `<div class="pf-section" style="padding:${state.spacing * 0.8}px 5%; background:var(--pf-bg2); max-width:none;">
      <div style="max-width:1100px; margin:0 auto;">
      <div class="pf-section-header">
        <div class="pf-section-label">Get in touch</div>
        <h2 class="pf-section-title">Let's Work Together</h2>
      </div>
      <div class="pf-contact-grid">
        <div class="pf-contact-info">
          ${email ? `<div class="pf-contact-item"><div class="pf-contact-icon">✉</div><span>${email}</span></div>` : ''}
          ${location ? `<div class="pf-contact-item"><div class="pf-contact-icon">📍</div><span>${location}</span></div>` : ''}
          ${github ? `<div class="pf-contact-item"><div class="pf-contact-icon">⟨/⟩</div><span>${github}</span></div>` : ''}
          ${twitter ? `<div class="pf-contact-item"><div class="pf-contact-icon">𝕏</div><span>${twitter}</span></div>` : ''}
          <div style="margin-top:16px;">
            <p style="font-size:0.85rem; color:var(--pf-ink2); line-height:1.8;">Open to freelance opportunities, collaborations, and full-time roles. Let's build something remarkable together.</p>
          </div>
        </div>
        <div class="pf-contact-form">
          <input class="pf-form-input" style="border-radius:${state.borderRadius}px;" placeholder="Your name" readonly>
          <input class="pf-form-input" style="border-radius:${state.borderRadius}px;" placeholder="Your email" readonly>
          <textarea class="pf-form-textarea" style="border-radius:${state.borderRadius}px;" placeholder="Your message…" readonly></textarea>
          <button class="pf-btn-primary" style="background:var(--pf-accent); border-radius:${state.borderRadius}px; align-self:flex-start;">Send Message →</button>
        </div>
      </div>
      </div></div>`;
  }

  // FOOTER
  if (state.showFooter) {
    html += `<footer class="pf-footer">
      <span>© ${new Date().getFullYear()} ${name}. Crafted with care.</span>
      <div class="pf-socials">
        ${github ? `<a class="pf-social" href="${github}" title="GitHub">⟨/⟩</a>` : ''}
        ${linkedin ? `<a class="pf-social" href="${linkedin}" title="LinkedIn">in</a>` : ''}
        ${twitter ? `<a class="pf-social" title="Twitter">𝕏</a>` : ''}
        ${email ? `<a class="pf-social" title="Email">✉</a>` : ''}
      </div>
    </footer>`;
  }

  html += `</div>`;

  get('portfolioRoot').innerHTML = html;
  updateCodePanel(html);
  updateStatus();
}

function updateCodePanel(html) {
  const codeEl = get('codeOutput');
  if (codeEl) {
    const pretty = html
      .replace(/></g, '>\n<')
      .replace(/\n\s*\n/g, '\n')
      .split('\n').map(l => '  ' + l).join('\n');
    codeEl.textContent = pretty.substring(0, 4000) + (pretty.length > 4000 ? '\n  … (truncated, use Export for full code)' : '');
  }
}
