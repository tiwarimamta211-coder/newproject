function switchTab(name) {
  document.querySelectorAll('.tab').forEach((t,i) => t.classList.toggle('active', ['profile','skills','projects','design'][i]===name));

// ── AVATAR ──
document.getElementById('avatarInput').onchange = function(e) {
  const file = e.target.files[0];
  if (!file) return;
  if (file.size > 2*1024*1024) { showToast('Image too large (max 2MB)', 'error'); return; }
  const reader = new FileReader();
  reader.onload = r => {
    state.avatar = r.target.result;
    const p = document.getElementById('avatarPreview');
    p.innerHTML = `<img src="${state.avatar}" alt="avatar">`;
    liveUpdate();
    showToast('Photo uploaded!');
  };
  reader.readAsDataURL(file);
};

// ── SKILLS ──
function renderSkills() {
  const c = document.getElementById('skillTags');
  c.innerHTML = state.skills.map((s,i) => `
    <div class="skill-tag">
      ${s}
      <span class="remove" onclick="removeSkill(${i})">×</span>
    </div>`).join('');
  document.getElementById('statSkills').textContent = state.skills.length;
}

function addSkill() {
  const inp = document.getElementById('skillInput');
  const val = inp.value.trim();
  if (!val) return;
  if (state.skills.includes(val)) { showToast('Skill already added', 'error'); return; }
  state.skills.push(val);
  inp.value = '';
  renderSkills();
  liveUpdate();
}

function quickAddSkill(btn) {
  const s = btn.textContent.trim();
  if (state.skills.includes(s)) { showToast(`"${s}" already added`, 'error'); return; }
  state.skills.push(s);
  renderSkills();
  liveUpdate();
  showToast(`Added "${s}"`);
}

function removeSkill(i) {
  state.skills.splice(i,1);
  renderSkills();
  liveUpdate();
}

// ── PROJECTS ──
function addProject() {
  projectCount++;
  const id = 'proj_'+Date.now();
  const obj = { id, title:'', desc:'', tech:'', url:'', github:'' };
  state.projects.push(obj);
  renderProjects();
}

function renderProjects() {
  const c = document.getElementById('projectsList');
  c.innerHTML = '';
  state.projects.forEach((p,i) => {
    const div = document.createElement('div');
    div.className = 'project-card';
    div.innerHTML = `
      <div class="project-card-header">
        <span class="project-card-title">Project #${i+1}</span>
        <button class="remove-btn" onclick="removeProject('${p.id}')">🗑</button>
      </div>
      <div class="form-group"><label>Title</label><input type="text" value="${p.title||''}" placeholder="My Awesome App" oninput="updateProject('${p.id}','title',this.value)"></div>
      <div class="form-group"><label>Description</label><textarea placeholder="What does this project do?" oninput="updateProject('${p.id}','desc',this.value)">${p.desc||''}</textarea></div>
      <div class="form-group"><label>Tech Stack</label><input type="text" value="${p.tech||''}" placeholder="React, Node.js, MongoDB" oninput="updateProject('${p.id}','tech',this.value)"></div>
      <div class="row2">

    c.appendChild(div);
  });
  document.getElementById('statProjects').textContent = state.projects.length;
}

function updateProject(id, key, val) {
  const p = state.projects.find(x => x.id===id);
  if (p) { p[key] = val; liveUpdate(); }
}

function removeProject(id) {
  state.projects = state.projects.filter(x => x.id!==id);
  renderProjects();
  liveUpdate();
}

function generatePreview() {
  collectState();
  const html = buildPortfolioHTML();
  const frame = document.getElementById('previewFrame');
  const doc = frame.contentDocument || frame.contentWindow.document;
  doc.open(); doc.write(html); doc.close();
}

// ── DEVICE ──
function setDevice(d, btn) {
  const wrapper = document.getElementById('deviceWrapper');
  wrapper.className = 'preview-frame-wrapper device-'+d;
  document.querySelectorAll('.device-btn').forEach(b => b.classList.remove('active'));
  btn.classList.add('active');
}

// ── MOBILE TOGGLE ──
let showingPreview = false;
function toggleMobileView() {
  showingPreview = !showingPreview;
  const ep = document.querySelector('.editor-panel');
  const pp = document.getElementById('previewPanel');
  const btn = document.getElementById('mobileToggle');
  ep.style.display = showingPreview ? 'none' : '';
  pp.style.display = showingPreview ? 'flex' : '';
  btn.textContent = showingPreview ? '✏️ Edit' : '👁 Preview';
  if (showingPreview) generatePreview();
}

// ── EXPORT MODAL ──
function openModal() { document.getElementById('exportModal').classList.add('open'); }
function closeModal() { document.getElementById('exportModal').classList.remove('open'); }

function selectExport(type, el) {
  selectedExport = type;
  document.querySelectorAll('.export-option').forEach(e => e.classList.remove('selected'));
  el.classList.add('selected');
}

function doExport() {
  collectState();
  const html = buildPortfolioHTML();
  if (selectedExport === 'html') {
    const blob = new Blob([html], { type: 'text/html' });
    const a = document.createElement('a');
    a.href = URL.createObjectURL(blob);
    a.download = `${(state.name||'portfolio').replace(/\s+/g,'-').toLowerCase()}-portfolio.html`;
    a.click();
    showToast('Portfolio downloaded!');
    closeModal();
  } else if (selectedExport === 'copy') {
    navigator.clipboard.writeText(html).then(() => { showToast('Code copied to clipboard!'); closeModal(); });
  } else if (selectedExport === 'github') {
    showToast('1. Create repo on GitHub\n2. Upload your HTML file as index.html\n3. Enable Pages in Settings', 'info');
    closeModal();
  } else if (selectedExport === 'netlify') {
    showToast('Download your file, then drag it to netlify.com/drop', 'info');
    closeModal();
  }
}

// ── TOAST ──
function showToast(msg, type='success') {
  const c = document.getElementById('toastContainer');
  const t = document.createElement('div');
  t.className = `toast${type==='error'?' error':type==='info'?' info':''}`;
  t.textContent = msg;
  c.appendChild(t);
  setTimeout(() => t.remove(), 3500);
}

// ── CLICK OUTSIDE MODAL ──
document.getElementById('exportModal').addEventListener('click', function(e) {
  if (e.target === this) closeModal();
});

document.getElementById('previewBtn').addEventListener('click', () => {
  if (window.innerWidth > 900) { generatePreview(); showToast('Preview refreshed!'); }
  else toggleMobileView();
});
