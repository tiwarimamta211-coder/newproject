  // Generate
  async function generate() {
    const name = document.getElementById('productName').value.trim();
    if (!name) { showToast('⚠ Please enter a product name'); return; }

    const category = document.getElementById('category').value;
    const audience = document.getElementById('audience').value.trim();
    const usp = document.getElementById('usp').value.trim();
    const brand = document.getElementById('brand').value.trim();
    const platform = document.getElementById('platform').value;
    const variantCount = parseInt(document.getElementById('variantCount').value);

    const lengthMap = {1: 'short (1 paragraph)', 2: 'medium (2-3 paragraphs)', 3: 'long and detailed (4+ paragraphs with all sections)'};

    if (document.getElementById('incBullets').checked) extras.push('include a bullet-point feature list');
    if (document.getElementById('incSEO').checked) extras.push('naturally weave in high-volume SEO keywords');
    if (document.getElementById('incHashtags').checked) extras.push('add 5-8 relevant hashtags at the end');
    if (document.getElementById('incHindi').checked) extras.push('mix in some Hinglish phrases naturally');

    const prompt = `You are an expert product copywriter. Generate ${variantCount} distinct product description variant(s) for the following product.

Product Name: ${name}
Category: ${category || 'General'}
Key Features: ${tags.length ? tags.join(', ') : 'Not specified'}
Target Audience: ${audience || 'General consumers'}
USP / Notes: ${usp || 'Not specified'}
Brand Voice: ${brand || 'Not specified'}
Tone: ${selectedTone}
Platform: ${platform}
Language: ${language}
Length: ${length}
Additional Requirements: ${extras.length ? extras.join(', ') : 'None'}

Format your response EXACTLY like this (use === VARIANT N === as separator):
=== VARIANT 1 ===
[description here]

=== VARIANT 2 ===
[description here]

(and so on for each variant)

Make each variant meaningfully different in angle, hook, and structure. Be creative, conversion-focused, and authentic. Do not add any preamble or explanation outside the variant blocks.`;

    btn.disabled = true;
    btn.classList.add('loading');

    document.getElementById('outputBody').innerHTML = `<div class="placeholder"><div class="placeholder-icon" style="animation: spin 1s linear infinite; opacity:0.7">✦</div><div>Claude is crafting your descriptions...</div></div>`;
    document.getElementById('variantTabs').style.display = 'none';
    document.getElementById('statsBar').style.display = 'none';
    document.getElementById('seoPanel').style.display = 'none';
    document.getElementById('regenerateBtn').style.display = 'none';

      const response = await fetch('https://api.anthropic.com/v1/messages', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          model: 'claude-sonnet-4-20250514',

          messages: [{ role: 'user', content: prompt }]

      while (true) {
        const { done, value } = await reader.read();

  function parseVariants(text, count) {
    const parts = text.split(/===\s*VARIANT\s*\d+\s*===/i).filter(p => p.trim());
    if (parts.length >= 2) return parts.map(p => p.trim());
    // fallback: split by double newline
    if (count === 1) return [text.trim()];
    const half = Math.floor(text.length / count);
    return Array.from({length: count}, (_, i) => text.slice(i * half, (i+1) * half).trim()).filter(Boolean);
  }

  function formatText(text) {
    return text
      .replace(/\*\*(.+?)\*\*/g, '<strong>$1</strong>')
      .replace(/^[-•]\s+(.+)/gm, '<li>$1</li>')
      .replace(/(<li>.*<\/li>\n?)+/g, m => `<ul style="margin:10px 0 10px 18px;display:flex;flex-direction:column;gap:4px;">${m}</ul>`)
      .replace(/\n{2,}/g, '</p><p style="margin-top:12px">')
      .replace(/\n/g, '<br>');
  }

  function renderVariants() {
    const tabs = document.getElementById('variantTabs');

    if (variants.length > 1) {

      tabs.innerHTML = variants.map((_, i) =>

    body.innerHTML = `<div class="description-text">${formatText(variants[activeVariant])}</div>`;
    document.getElementById('statsBar').style.display = 'flex';
  }

  function renderSEO(text) {
    const words = text.trim().split(/\s+/).length;
    const hasEmoji = /\p{Emoji}/u.test(text);
    const hasBullets = /[-•*]|\d\./m.test(text);
    const hasCTA = /buy|shop|order|get|try|discover|click|learn|explore/i.test(text);
    const sentLen = text.split(/[.!?]+/).filter(Boolean).map(s => s.split(/\s+/).length);
    const avgSent = sentLen.length ? Math.round(sentLen.reduce((a,b)=>a+b,0)/sentLen.length) : 0;

    const scores = [
      { label: 'Readability', val: Math.min(100, Math.max(40, 100 - Math.abs(avgSent - 15) * 4)) },
      { label: 'Length Score', val: Math.min(100, Math.round((words / 150) * 100)) },
      { label: 'Engagement', val: (hasEmoji ? 30 : 0) + (hasBullets ? 30 : 0) + (hasCTA ? 40 : 0) },
      { label: 'CTA Strength', val: hasCTA ? 85 : 20 },
    ];

    const panel = document.getElementById('seoPanel');
    const grid = document.getElementById('seoGrid');

    grid.innerHTML = scores.map(s => {
      const cls = s.val >= 70 ? 'score-high' : s.val >= 40 ? 'score-mid' : 'score-low';
      const barColor = s.val >= 70 ? 'var(--accent3)' : s.val >= 40 ? 'var(--gold)' : 'var(--danger)';
      return `<div class="seo-item">
        <div class="seo-label">${s.label}</div>
        <div class="seo-bar-wrap"><div class="seo-bar" style="width:0%;background:${barColor}" data-w="${s.val}%"></div></div>
        <div class="seo-score ${cls}">${s.val}/100</div>
      </div>`;
    }).join('');

    panel.style.display = 'block';
    setTimeout(() => {
      grid.querySelectorAll('.seo-bar').forEach(b => { b.style.width = b.dataset.w; });
    }, 100);
  }

    const blob = new Blob([text], { type: 'text/plain' });
    const a = document.createElement('a');
    a.href = URL.createObjectURL(blob);
