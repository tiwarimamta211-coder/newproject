function switchAuthTab(tab) {
  document.querySelectorAll('.auth-tab').forEach((t,i)=>{
    t.classList.toggle('active', (tab==='login'&&i===0)||(tab==='register'&&i===1));
  });
  document.getElementById('loginForm').style.display = tab==='login'?'block':'none';
  document.getElementById('registerForm').style.display = tab==='register'?'block':'none';
}

function doLogin() {
  const email = document.getElementById('loginEmail').value.trim();
  const pass = document.getElementById('loginPassword').value.trim();
  const found = USERS_DB.find(u=>u.email===email && u.password===pass);
  if (found) {
    state.user = found;
    document.getElementById('authScreen').classList.remove('active');
    document.getElementById('appScreen').classList.add('active');
    document.getElementById('welcomeName').textContent = found.name.split(' ')[0];
    document.getElementById('userAvatar').textContent = found.avatar;
    document.getElementById('currentPlanBadge').textContent = found.plan;
    document.getElementById('currentPlanBadge').className = 'plan-badge plan-' + found.plan.toLowerCase();
    initApp();
    showToast('Welcome back, ' + found.name.split(' ')[0] + '! 👋', 'success');
  } else {
    showToast('Invalid credentials. Try admin@connectpro.com / admin123', 'error');
  }
}

function doRegister() {
  const name = document.getElementById('regName').value.trim();
  const email = document.getElementById('regEmail').value.trim();
  if (!name || !email) { showToast('Please fill all fields','error'); return; }
  showToast('Account created! Please log in.','success');
  switchAuthTab('login');
  document.getElementById('loginEmail').value = email;
}

function doLogout() {
  state.user = null;
  document.getElementById('appScreen').classList.remove('active');
  document.getElementById('authScreen').classList.add('active');
  showToast('Logged out successfully','info');
}

function renderDownloadChart() {
  const days = ['Mon','Tue','Wed','Thu','Fri','Sat','Sun'];
  const values = [42,67,54,89,120,78,95];
  const max = Math.max(...values);
  const container = document.getElementById('downloadChart');
  container.innerHTML = days.map((d,i)=>`
    <div class="chart-bar" style="height:${(values[i]/max*100)}%;background:${i===4?'var(--accent)':'rgba(59,130,246,0.5)'};" title="${d}: ${values[i]} downloads">
      <div class="chart-bar-label">${d}</div>
    </div>`).join('');
}

function renderLineChart() {
  const svg = document.getElementById('lineChartSvg');
  if (!svg) return;
  const pts = [30,60,45,80,70,110,90,130,100,140,120,160];
  const w=400, h=140, pad=10;
  const xs = pts.map((_,i)=>pad+i*(w-2*pad)/(pts.length-1));
  const maxV=Math.max(...pts), minV=Math.min(...pts);
  const ys = pts.map(v=>pad+(1-(v-minV)/(maxV-minV))*(h-2*pad));
  const pathD = xs.map((x,i)=>`${i?'L':'M'}${x},${ys[i]}`).join(' ');
  const areaD = pathD+` L${xs[xs.length-1]},${h} L${xs[0]},${h} Z`;
  svg.innerHTML = `
    <defs><linearGradient id="lg" x1="0" y1="0" x2="0" y2="1">
      <stop offset="0%" stop-color="#3b82f6" stop-opacity="0.4"/>
      <stop offset="100%" stop-color="#3b82f6" stop-opacity="0"/>
    </linearGradient></defs>
    <path d="${areaD}" fill="url(#lg)"/>
    <path d="${pathD}" fill="none" stroke="#3b82f6" stroke-width="2" stroke-linecap="round"/>
    ${xs.map((x,i)=>`<circle cx="${x}" cy="${ys[i]}" r="3" fill="#3b82f6"/>`).join('')}`;
}

function updateWatermarkPreview() {
  const text = document.getElementById('wmText').value;
  const size = document.getElementById('wmSize').value;
  const opacity = document.getElementById('wmOpacity').value;
  const rotation = document.getElementById('wmRotation').value;
  document.getElementById('wmSizeVal').textContent = size+'px';
  document.getElementById('wmOpacityVal').textContent = opacity+'%';
  document.getElementById('wmRotationVal').textContent = rotation+'°';
  const overlay = document.getElementById('wmOverlay');
  if (!overlay) return;
  overlay.textContent = text;
  overlay.style.fontSize = size+'px';
  overlay.style.opacity = opacity/100;
  overlay.style.color = state.wmColor;

  const tiled = document.getElementById('wmTile')?.checked;
  if (tiled) {
    overlay.style.top = '0'; overlay.style.left = '0';
    overlay.style.width = '100%'; overlay.style.height = '100%';
    overlay.style.display = 'flex'; overlay.style.alignItems = 'center'; overlay.style.justifyContent = 'center';
    overlay.style.transform = `rotate(${rotation}deg)`;
  } else {
    applyWmPosition(overlay, state.wmPosition, rotation);
  }
}

function applyWmPosition(el, pos, rotation) {
  const positions = {
    tl:{top:'12px',left:'12px',transform:`rotate(${rotation}deg)`},
    tc:{top:'12px',left:'50%',transform:`translateX(-50%) rotate(${rotation}deg)`},
    tr:{top:'12px',right:'12px',transform:`rotate(${rotation}deg)`},
    ml:{top:'50%',left:'12px',transform:`translateY(-50%) rotate(${rotation}deg)`},
    center:{top:'50%',left:'50%',transform:`translate(-50%,-50%) rotate(${rotation}deg)`},
    mr:{top:'50%',right:'12px',transform:`translateY(-50%) rotate(${rotation}deg)`},
    bl:{bottom:'12px',left:'12px',transform:`rotate(${rotation}deg)`},
    bc:{bottom:'12px',left:'50%',transform:`translateX(-50%) rotate(${rotation}deg)`},
    br:{bottom:'12px',right:'12px',transform:`rotate(${rotation}deg)`},
  };
  const p = positions[pos]||positions.center;
  el.style.top=p.top||''; el.style.left=p.left||''; el.style.right=p.right||'';
  el.style.bottom=p.bottom||''; el.style.transform=p.transform;
}

function showToast(msg, type='info') {
  const icons = { success:'✅', error:'❌', info:'ℹ️' };
  const container = document.getElementById('toastContainer');
  const toast = document.createElement('div');
  toast.className = `toast toast-${type}`;
  toast.innerHTML = `<span>${icons[type]}</span><span>${msg}</span>`;
  container.appendChild(toast);
  setTimeout(()=>toast.remove(), 3500);
}
