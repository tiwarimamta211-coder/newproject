<script>
  // Sample mock script for demo mode
  function getMockScript(topic, platform, tone, duration) {
    var scenes = Math.max(3, Math.round(duration / 15));
    var sceneTime = Math.round(duration / scenes);
    return [
      {
        header: 'SCENE 1: THE HOOK [00:00 – 00:' + String(sceneTime).padStart(2,'0') + ']',
        visual: 'High-energy opening shot. Quick cuts, vibrant visuals. Platform: ' + platform + '.',
        dialogue: '"Are you leaving money on the table every single month? Most people don\'t even realise it — until it\'s too late."'
      },
      {
        header: 'SCENE 2: THE PROBLEM [00:' + String(sceneTime).padStart(2,'0') + ' – 00:' + String(sceneTime*2).padStart(2,'0') + ']',
        visual: 'Split screen: stressed person vs. relaxed professional. ' + tone + ' mood.',
        dialogue: '"The truth about ' + topic + ' is that most people are doing it wrong. Without the right approach, you\'re losing time, money, and opportunity every day."'
      },
      {
        header: 'SCENE 3: THE SOLUTION [00:' + String(sceneTime*2).padStart(2,'0') + ' – 00:' + String(sceneTime*3).padStart(2,'0') + ']',
        visual: 'Text overlays with 3 key points pop on screen. Background music builds.',
        dialogue: '"Here\'s what works: First, understand the fundamentals. Second, apply a proven system. Third, track your results. That\'s the ' + topic + ' formula that actually delivers."'
      },
      {
        header: 'SCENE 4: CALL TO ACTION [00:' + String(duration - sceneTime).padStart(2,'0') + ' – 00:' + String(duration).padStart(2,'0') + ']',
        visual: 'Host on camera, confident smile. Logo and CTA button appears on screen.',
        dialogue: '"Ready to master ' + topic + '? Click the link below and get started today. Don\'t wait — the best time to act is right now."'
      }
    ];
  }

  function buildScriptHTML(topic, platform, tone, lang, scenes) {
    var html = '<div class="script-title">'
      + '<h2>' + escHtml(topic) + '</h2>'
      + '<div class="script-meta">'
      + '<span class="meta-tag blue">' + escHtml(platform) + '</span>'
      + '<span class="meta-tag green">' + escHtml(tone) + '</span>'
      + '<span class="meta-tag amber">' + escHtml(lang) + '</span>'
      + '</div></div>';

    scenes.forEach(function(s) {
      html += '<div class="scene-block">'
        + '<div class="scene-header">' + escHtml(s.header) + '</div>'
        + '<div class="visual-desc">📷 ' + escHtml(s.visual) + '</div>'
        + '<div class="dialogue">🎙️ ' + escHtml(s.dialogue) + '</div>'
        + '</div>';
    });
    return html;
  }

  function generateScript() {
    var topic      = document.getElementById('topic').value.trim();
    var platform   = document.getElementById('platform').value;
    var tone       = document.getElementById('tone').value;
    var audience   = document.getElementById('audience').value.trim();
    var duration   = parseInt(document.getElementById('duration').value) || 60;
    var keyMsg     = document.getElementById('keyMessage').value.trim();
    var lang       = document.getElementById('lang').value;
    var apiKey     = document.getElementById('apiKey').value.trim();
    var outputCard = document.getElementById('outputCard');
    var genBtn     = document.getElementById('genBtn');
    var copyBtn    = document.getElementById('copyBtn');

    if (!topic) { alert('Please enter a video topic!'); return; }

    // Loading state
    outputCard.innerHTML = '<div class="loading"><div class="loading-icon">🎬</div><p>CineScript AI is composing your scenes...</p></div>';
    genBtn.disabled = true;
    genBtn.innerHTML = '<span class="spin">⏳</span> Generating...';
    copyBtn.style.display = 'none';

    if (!apiKey || !apiKey.startsWith('sk-ant-')) {
      // Demo mode
      setTimeout(function() {
        var scenes = getMockScript(topic, platform, tone, duration);
        outputCard.innerHTML = buildScriptHTML(topic, platform, tone, lang, scenes);
        finishGen(false);
      }, 1600);
      return;
    }

    // Build prompt
    var prompt = 'You are a professional video scriptwriter. Write a complete, production-ready video script with the following parameters:\n\n'
      + '- Topic: ' + topic + '\n'
      + '- Platform: ' + platform + '\n'
      + '- Tone: ' + tone + '\n'
      + '- Target Audience: ' + (audience || 'General audience') + '\n'
      + '- Duration: ' + duration + ' seconds\n'
      + '- Key Message / CTA: ' + (keyMsg || 'Engage the viewer') + '\n'
      + '- Language: ' + lang + '\n\n'
      + 'Format each scene exactly like this:\n'
      + 'SCENE [N]: [SCENE NAME] [[TIMECODE]]\n'
      + 'VISUAL: [Camera direction and visual description]\n'
      + 'DIALOGUE: "[Spoken words]"\n\n'
      + 'Write ' + Math.max(3, Math.round(duration/15)) + ' scenes. Be specific, creative, and actionable.';

    fetch('https://api.anthropic.com/v1/messages', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        model: 'claude-sonnet-4-20250514',
        max_tokens: 1500,
        messages: [{ role: 'user', content: prompt }]
      })
    })
    .then(function(r){ return r.json(); })
    .then(function(data) {
      if (data.content && data.content[0]) {
        var raw = data.content[0].text || '';
        // Parse scenes from raw text
        var scenes = parseScenes(raw);
        if (scenes.length > 0) {
          outputCard.innerHTML = buildScriptHTML(topic, platform, tone, lang, scenes);
        } else {
          // Fallback: show raw output in readable format
          outputCard.innerHTML =
            '<div class="script-title"><h2>' + escHtml(topic) + '</h2>'
            + '<div class="script-meta"><span class="meta-tag blue">' + escHtml(platform) + '</span>'
            + '<span class="meta-tag green">' + escHtml(lang) + '</span></div></div>'
            + '<div class="raw-output">' + escHtml(raw) + '</div>';
        }
        document.getElementById('apiStatus').className = 'api-status live';
        document.getElementById('apiStatus').innerText = '✅ Live AI';
        finishGen(true);
      } else {
        outputCard.innerHTML = '<div class="output-empty"><div>⚠️</div><p>' + (data.error ? data.error.message : 'API error. Try again.') + '</p></div>';
        finishGen(false);
      }
    })
    .catch(function() {
      outputCard.innerHTML = '<div class="output-empty"><div>❌</div><p>Network error. Check API key.</p></div>';
      finishGen(false);
    });
  }

  function parseScenes(text) {
    var scenes = [];
    var blocks = text.split(/SCENE\s+\d+/i).filter(function(b){ return b.trim(); });
    blocks.forEach(function(block, i) {
      var headerMatch = block.match(/^[:\s]*([^\n]+)/);
      var visualMatch = block.match(/VISUAL[:\s]+([^\n]+(?:\n(?!DIALOGUE)[^\n]+)*)/i);
      var dialogueMatch = block.match(/DIALOGUE[:\s]+"?([^"]+)"?/i);
      if (dialogueMatch || visualMatch) {
        scenes.push({
          header: 'SCENE ' + (i+1) + ': ' + (headerMatch ? headerMatch[1].trim() : 'Scene ' + (i+1)),
          visual: visualMatch  ? visualMatch[1].trim()  : 'Visual direction not specified.',
          dialogue: dialogueMatch ? dialogueMatch[1].trim() : block.trim().substring(0, 200)
        });
      }
    });
    return scenes;
  }

  function finishGen(success) {
    var genBtn = document.getElementById('genBtn');
    genBtn.disabled = false;
    genBtn.innerHTML = '✨ Generate Script';
    document.getElementById('copyBtn').style.display = 'block';
    document.getElementById('emptyState') && (document.getElementById('emptyState').style.display = 'none');
  }

  function copyScript() {
    var content = document.getElementById('outputCard').innerText;
    if (navigator.clipboard) {
      navigator.clipboard.writeText(content).then(function(){ alert('Script copied! ✅'); });
    } else {
      var ta = document.createElement('textarea');
      ta.value = content; document.body.appendChild(ta); ta.select();
      document.execCommand('copy'); document.body.removeChild(ta);
      alert('Script copied! ✅');
    }
  }

  function escHtml(t) {
    return String(t||'').replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;');
  }
</script>
