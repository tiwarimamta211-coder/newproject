<script>
// ═══════════════════════════════════════════════════════
//  STATE
// ═══════════════════════════════════════════════════════
const state = {
  formats: 0, validates: 0, minifies: 0, copies: 0, others: 0,
  totalLines: 0, totalChars: 0, totalErrors: 0, timeSaved: 0,
  history: [],
  hourly: new Array(12).fill(0),
  sparkData: [],
  diffVisible: false,
  autoMode: false,
  autoTimer: null,
  shortcutsVisible: false,
  lastInput: '', lastOutput: ''
};

// ═══════════════════════════════════════════════════════
//  FORMATTER ENGINE
// ═══════════════════════════════════════════════════════
function formatJS() {
  const input = document.getElementById('inputJS').value;
  if (!input.trim()) { showToast('Paste some JavaScript first!', 'err'); return; }

  hideError();
  setBusy(true);

  setTimeout(() => {
    try {
      const opts = getOpts();
      let result = formatCode(input, opts);

      if (opts.removeLogs) result = result.replace(/console\.(log|warn|error|info|debug)\([^)]*\);?\n?/g, '');
      if (opts.sortImports) result = sortImportLines(result);

      const mode = document.getElementById('outputMode').value;
      if (mode === 'minify') result = minifyCode(result);
      else if (mode === 'compact') result = compactCode(result);

      document.getElementById('outputJS').value = result;
      state.lastInput = input; state.lastOutput = result;

      updateOutputStats(result);
      updateQualityAnalysis(input);

      const lines = result.split('\n').length;
      state.formats++;
      state.totalLines += lines;
      state.totalChars += input.length;
      state.timeSaved += Math.ceil(input.length / 200);
      state.sparkData.push(input.length);
      if (state.sparkData.length > 20) state.sparkData.shift();
      state.hourly[new Date().getHours() % 12]++;

      addHistory(input, result, 'format');
      addTimeline('Formatted JS', 'format', lines + ' lines');
      updateDashboard();
      updateBars();
      drawSparkline();
      renderHeatmap();
      setBusy(false);
      showToast('Formatted successfully! (' + lines + ' lines)', 'ok');
    } catch(e) {
      showError('Parse error: ' + e.message);
      state.totalErrors++;
      updateDashboard();
      setBusy(false);
    }
  }, 80);
}

function getOpts() {
  return {
    indent: document.getElementById('useTabs').checked ? '\t' : ' '.repeat(+document.getElementById('indentSize').value),
    singleQuotes: document.getElementById('singleQuotes').checked,
    semis: document.getElementById('addSemis').checked,
    trailingCommas: document.getElementById('trailingCommas').checked,
    arrowParens: document.getElementById('arrowParens').checked,
    bracketSpacing: document.getElementById('bracketSpacing').checked,
    removeLogs: document.getElementById('removeLogs').checked,
    sortImports: document.getElementById('sortImports').checked,
    lineEnding: document.getElementById('lineEndings').value,
    printWidth: +document.getElementById('printWidth').value,
  };
}

function formatCode(code, opts) {
  // Validate first
  new Function(code); // throws on syntax error

  let result = code;
  const ind = opts.indent;

  // Normalize line endings
  result = result.replace(/\r\n/g,'\n').replace(/\r/g,'\n');

  // Quote normalization
  if (opts.singleQuotes) {
    result = result.replace(/"([^"\\]*(\\.[^"\\]*)*)"/g, (m,inner) => "'" + inner.replace(/\\"/g,'"').replace(/'/g,"\\'") + "'");
  } else {
    result = result.replace(/'([^'\\]*(\\.[^'\\]*)*)'/g, (m,inner) => '"' + inner.replace(/\\'/g,"'").replace(/"/g,'\\"') + '"');
  }

  // Run a robust pretty-printer
  result = prettyPrint(result, ind, opts);

  // Add/remove semicolons
  if (!opts.semis) {
    result = result.split('\n').map(l => l.trimEnd().replace(/;$/, '')).join('\n');
  } else {
    result = ensureSemis(result);
  }

  // Line endings
  if (opts.lineEnding === 'crlf') result = result.replace(/\n/g, '\r\n');
  else if (opts.lineEnding === 'cr') result = result.replace(/\n/g, '\r');

  return result;
}

function prettyPrint(code, ind, opts) {
  const lines = [];
  let depth = 0;
  let inStr = false, strChar = '', inML = false;
  let i = 0, cur = '';

  const flush = () => {
    const t = cur.trim();
    if (t) lines.push(ind.repeat(Math.max(0, depth)) + t);
    cur = '';
  };

  while (i < code.length) {
    const ch = code[i];
    const next = code[i+1] || '';

    if (!inStr && !inML && ch === '/' && next === '*') { inML = true; cur += ch; i++; continue; }
    if (inML) { cur += ch; if (ch === '*' && next === '/') { cur += next; i++; inML = false; } i++; continue; }
    if (!inStr && !inML && ch === '/' && next === '/') {
      flush();
      let c = ''; while (i < code.length && code[i] !== '\n') { c += code[i]; i++; }
      lines.push(ind.repeat(Math.max(0,depth)) + c.trim());
      continue;
    }
    if (!inStr && (ch === '"' || ch === "'" || ch === '`')) { inStr = true; strChar = ch; cur += ch; i++; continue; }
    if (inStr) {
      cur += ch;
      if (ch === '\\') { cur += next; i++; }
      else if (ch === strChar) inStr = false;
      i++; continue;
    }

    if (ch === '\n') { flush(); i++; continue; }
    if (ch === '{') {
      cur += opts.bracketSpacing && cur.trim() ? ' {' : '{';
      flush(); depth++;
    } else if (ch === '}') {
      flush(); depth = Math.max(0, depth - 1);
      cur = '}';
      if (next === ';' || next === ',' || next === ')') { cur += next; i++; }
      flush();
    } else if (ch === ';') {
      cur += ';'; flush();
    } else {
      cur += ch;
    }
    i++;
  }
  flush();
  return lines.filter((l,i,a) => !(l.trim()==='' && a[i-1]?.trim()==='')).join('\n');
}

function ensureSemis(code) {
  return code.split('\n').map((line, i, arr) => {
    const t = line.trimEnd();
    if (!t) return line;
    const last = t[t.length-1];
    const next = (arr[i+1]||'').trim();
    if (['{','}',',',';','(','['].includes(last)) return line;
    if (next.startsWith('}') || next.startsWith(')') || next.startsWith(']')) return line;
    if (/^\s*(\/\/|\/\*|\*|if|else|for|while|switch|function|class|const|let|var|return|import|export)/.test(t) && last === ')') return line;
    return t + ';';
  }).join('\n');
}

function minifyCode(code) {
  let r = code;
  r = r.replace(/\/\/[^\n]*/g,'').replace(/\/\*[\s\S]*?\*\//g,'');
  r = r.replace(/\s+/g,' ').replace(/\s*([{}();,=+\-*/<>!&|:?])\s*/g,'$1').trim();
  return r;
}

function compactCode(code) {
  return code.split('\n').filter(l=>l.trim()).map(l=>l.trim()).join(' ');
}

function sortImportLines(code) {
  const lines = code.split('\n');
  const imports = lines.filter(l => l.trim().startsWith('import '));
  const rest = lines.filter(l => !l.trim().startsWith('import '));
  imports.sort();
  return [...imports, '', ...rest].join('\n');
}

  setPct('qaFunctions',fns); setPct('qaVariables',vars); setPct('qaComments',coms); setPct('qaStrings',strs); setPct('qaImports',imps);
}

function toggleComment() {
  const ta = document.getElementById('inputJS');
  const s = ta.selectionStart;
  const lineStart = ta.value.lastIndexOf('\n', s-1) + 1;
  const line = ta.value.slice(lineStart);
  const lineEnd = line.indexOf('\n');
  const fullLine = lineEnd >= 0 ? line.slice(0, lineEnd) : line;
  const isCommented = fullLine.trim().startsWith('//');

  ta.value = ta.value.slice(0, lineStart) + newLine + ta.value.slice(lineStart + fullLine.length);

}

function renderDiff(a, b) {
  const aLines = a.split('\n');
  const bLines = b.split('\n');
  let html = '';
  const maxLen = Math.max(aLines.length, bLines.length);

  for (let i = 0; i < maxLen; i++) {
    const al = aLines[i], bl = bLines[i];
    if (al === undefined) html += `<div class="diff-line add">+ ${esc(bl)}</div>`;
    else if (bl === undefined) html += `<div class="diff-line rem">− ${esc(al)}</div>`;
    else if (al !== bl) {
      html += `<div class="diff-line rem">− ${esc(al)}</div>`;
      html += `<div class="diff-line add">+ ${esc(bl)}</div>`;
    } else {
      html += `<div class="diff-line ctx">  ${esc(al)}</div>`;
    }
  }
  document.getElementById('diffBody').innerHTML = html || '<div style="padding:8px;color:var(--ink4)">No differences.</div>';
}

function drawSparkline() {
  const canvas = document.getElementById('sparkline');
  const ctx = canvas.getContext('2d');
  const W = canvas.offsetWidth * devicePixelRatio || 300;
  const H = canvas.offsetHeight * devicePixelRatio || 60;
  canvas.width = W; canvas.height = H;
  ctx.clearRect(0,0,W,H);

  const data = state.sparkData;
  if (data.length < 2) return;
  const max = Math.max(...data, 1);
  const step = W / (data.length - 1);

  const isDark = document.body.classList.contains('dark');
  const amber = isDark ? '#f5a623' : '#d4870a';

  ctx.strokeStyle = amber;
  ctx.lineWidth = 2 * devicePixelRatio;
  ctx.lineJoin = 'round';
  ctx.lineCap = 'round';
  ctx.beginPath();
  data.forEach((v,i) => {
    const x = i * step;
    const y = H - (v/max) * (H * 0.8) - H * 0.1;
    i === 0 ? ctx.moveTo(x,y) : ctx.lineTo(x,y);
  });
  ctx.stroke();

  // Fill
  ctx.lineTo((data.length-1)*step, H);
  ctx.lineTo(0, H);
  const grad = ctx.createLinearGradient(0,0,0,H);
  grad.addColorStop(0, amber + '44');
  grad.addColorStop(1, amber + '00');
  ctx.fillStyle = grad;
  ctx.fill();
}

    const alpha = Math.round(intensity * 200);
    const isDark = document.body.classList.contains('dark');
    const color = isDark ? `rgba(245,166,35,${intensity*0.8})` : `rgba(212,135,10,${intensity*0.7})`;
    return `<div class="heat-cell" data-tip="${label}" style="background:${v>0?color:''}"></div>`;
  }).join('');
}

function setTheme(t) {
  document.body.classList.toggle('dark', t === 'dark');
  document.getElementById('themeLight').classList.toggle('active', t === 'light');
  document.getElementById('themeDark').classList.toggle('active', t === 'dark');
  localStorage.setItem('jsfmt-theme', t);
  setTimeout(() => { drawSparkline(); renderHeatmap(); }, 50);
}

function showToast(msg, type='info') {
  const wrap = document.getElementById('toastWrap');
  const t = document.createElement('div');
  t.className = 'toast ' + type;
  t.textContent = msg;
  wrap.appendChild(t);
  setTimeout(() => { t.style.opacity='0'; t.style.transform='translateX(16px)'; t.style.transition='all 0.3s'; setTimeout(()=>t.remove(),300); }, 3000);
}

function byteSize(s){ return new Blob([s]).size }
function formatBytes(b){ return b<1024?b+' B':b<1048576?(b/1024).toFixed(1)+' KB':(b/1048576).toFixed(2)+' MB' }
function esc(s){ return (s||'').replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;') }
function timeAgo(d){ const s=Math.floor((Date.now()-d)/1000); return s<60?'just now':s<3600?Math.floor(s/60)+'m ago':Math.floor(s/3600)+'h ago' }
function formatTime(d){ return d.toLocaleTimeString('en',{hour:'2-digit',minute:'2-digit',second:'2-digit'}) }

document.addEventListener('keydown', e => {
  const mod = e.ctrlKey || e.metaKey;
  if (mod && e.key === 'Enter')         { e.preventDefault(); formatJS(); }
  if (mod && e.shiftKey && e.key === 'C') { e.preventDefault(); copyOutput(); }
  if (mod && e.shiftKey && e.key === 'X') { e.preventDefault(); clearAll(); }
  if (mod && e.shiftKey && e.key === 'M') { e.preventDefault(); minifyDirect(); }
  if (mod && e.shiftKey && e.key === 'V') { e.preventDefault(); validateOnly(); }
  if (mod && e.shiftKey && e.key === 'D') { e.preventDefault(); toggleDiff(); }
  if (mod && e.shiftKey && e.key === 'S') { e.preventDefault(); loadSample(); }
  if (mod && e.key === 's')              { e.preventDefault(); downloadOutput(); }
});
