  // Indian cities mock
  var indianCities = {
    'mumbai':   { name:'Mumbai, IN',   temp:32, desc:'Humid & Hazy',    type:'Haze',   humidity:82, wind:18, visibility:4,  pressure:1006, sunrise:'6:28 AM', sunset:'7:10 PM', aqi:87  },

    'chennai':  { name:'Chennai, IN',  temp:34, desc:'Hot & Sunny',     type:'Clear',  humidity:70, wind:20, visibility:8,  pressure:1008, sunrise:'6:01 AM', sunset:'6:34 PM', aqi:72  },
    'kolkata':  { name:'Kolkata, IN',  temp:33, desc:'Muggy',           type:'Haze',   humidity:78, wind:14, visibility:5,  pressure:1005, sunrise:'5:24 AM', sunset:'6:12 PM', aqi:110 },
    'pune':     { name:'Pune, IN',     temp:29, desc:'Pleasant',        type:'Clouds', humidity:55, wind:15, visibility:10, pressure:1011, sunrise:'6:16 AM', sunset:'6:58 PM', aqi:48  },
    'hyderabad':{ name:'Hyderabad, IN',temp:31, desc:'Sunny',           type:'Clear',  humidity:50, wind:16, visibility:9,  pressure:1007, sunrise:'5:58 AM', sunset:'6:44 PM', aqi:66  }

  function setDateTime() {
    var now = new Date();
    document.getElementById('dateTime').innerText =
      now.toLocaleDateString('en-IN', { weekday:'long', day:'numeric', month:'long' }) + ' • ' +
      now.toLocaleTimeString('en-IN', { hour:'2-digit', minute:'2-digit' });
  }

  function renderWeather(d) {
    document.getElementById('cityName').innerText    = d.name;
    document.getElementById('mainTemp').innerText    = d.temp + '°C';
    document.getElementById('weatherDesc').innerText = d.desc;
    document.getElementById('weatherEmoji').innerText= weatherEmojis[d.type] || '🌤️';
    document.getElementById('humidity').innerText    = d.humidity + '%';
    document.getElementById('wind').innerText        = d.wind + ' km/h';
    document.getElementById('visibility').innerText  = d.visibility + ' km';
    document.getElementById('pressure').innerText    = d.pressure + ' hPa';
    document.getElementById('sunrise').innerText     = d.sunrise;
    document.getElementById('sunset').innerText      = d.sunset;

    // AQI
    var aqi = d.aqi || 42;
    var aqiStatus = aqi <= 50 ? 'Good' : aqi <= 100 ? 'Moderate' : aqi <= 150 ? 'Unhealthy (Sensitive)' : 'Unhealthy';
    var aqiColor  = aqi <= 50 ? '#10b981' : aqi <= 100 ? '#f59e0b' : aqi <= 150 ? '#f97316' : '#ef4444';
    document.getElementById('aqiLabel').innerText    = aqiStatus + ' (AQI: ' + aqi + ')';
    document.getElementById('aqiLabel').style.color  = aqiColor;
    document.getElementById('aqiBar').style.width    = Math.min(aqi / 200 * 100, 100) + '%';
    document.getElementById('aqiBar').style.background = 'linear-gradient(90deg,' + aqiColor + ',#818cf8)';

    // Forecast
    var grid = document.getElementById('forecastGrid');
    grid.innerHTML = '';
    var fc = d.forecast || generateForecast(d);
    fc.forEach(function(f) {
      var card = document.createElement('div');
      card.className = 'forecast-card';
      card.innerHTML =
        '<div class="fc-day">' + f.day + '</div>' +
        '<div class="fc-icon">' + (weatherEmojis[f.type] || '🌤️') + '</div>' +
        '<div class="fc-temp">' + f.temp + '°</div>' +
        '<div class="fc-desc">' + f.desc + '</div>';
      grid.appendChild(card);
    });
  }

  function generateForecast(base) {
    var today = new Date();
    var result = [];
    for (var i = 1; i <= 5; i++) {
      var d = new Date(today); d.setDate(today.getDate() + i);
      var types = ['Clear','Clouds','Rain','Clouds','Clear'];
      var descs = ['Sunny','Cloudy','Rainy','Partly Cloudy','Clear'];
      var t = Math.round(base.temp + (Math.random() * 6 - 3));
      result.push({ day: days[d.getDay()], temp: t, type: types[i-1], desc: descs[i-1] });
    }
    return result;
  }

  function searchCity() {
    var city = document.getElementById('cityInput').value.trim().toLowerCase();
    if (!city) return;

    if (API_KEY && API_KEY.length > 10) {
      fetchLive(city);
    } else {
      // Demo mode — check mock data
      var found = indianCities[city];
      if (found) {
        found.forecast = found.forecast || generateForecast(found);
        renderWeather(found);
      } else {
        // Generic mock for unknown cities
        var generic = JSON.parse(JSON.stringify(mockCities.default));
        generic.name = city.charAt(0).toUpperCase() + city.slice(1);
        generic.temp = Math.round(20 + Math.random() * 15);
        renderWeather(generic);
      }
      document.getElementById('cityInput').value = '';
    }
  }

  // Live API fetch (works when API_KEY is set)
  function fetchLive(city) {
    fetch('https://api.openweathermap.org/data/2.5/weather?q=' + encodeURIComponent(city) + '&units=metric&appid=' + API_KEY)
      .then(function(r){ return r.json(); })
      .then(function(data) {
        if (data.cod !== 200) { alert('City not found!'); return; }
        var d = {
          name: data.name + ', ' + data.sys.country,
          temp: Math.round(data.main.temp),
          desc: data.weather[0].description,
          type: data.weather[0].main,
          humidity: data.main.humidity,
          wind: Math.round(data.wind.speed * 3.6),
          visibility: Math.round((data.visibility || 10000) / 1000),
          pressure: data.main.pressure,
          sunrise: new Date(data.sys.sunrise * 1000).toLocaleTimeString('en-IN',{hour:'2-digit',minute:'2-digit'}),
          sunset:  new Date(data.sys.sunset  * 1000).toLocaleTimeString('en-IN',{hour:'2-digit',minute:'2-digit'}),
          aqi: 55
        };
        document.getElementById('apiNotice').className = 'api-notice live';
        document.getElementById('apiNotice').innerText = '✅ Live Data — OpenWeatherMap API Active';
        fetchForecastLive(data.coord.lat, data.coord.lon, d);
      })
      .catch(function(){ alert('Network error. Check your API key.'); });
  }

  function fetchForecastLive(lat, lon, currentData) {
    fetch('https://api.openweathermap.org/data/2.5/forecast?lat=' + lat + '&lon=' + lon + '&units=metric&appid=' + API_KEY)
      .then(function(r){ return r.json(); })
      .then(function(data) {
        var daily = data.list.filter(function(i){ return i.dt_txt.includes('12:00:00'); }).slice(0,5);
        currentData.forecast = daily.map(function(day) {
          var dt = new Date(day.dt * 1000);
          return { day: days[dt.getDay()], temp: Math.round(day.main.temp), type: day.weather[0].main, desc: day.weather[0].main };
        });
        renderWeather(currentData);
      });
  }

  // INIT
  setDateTime();

  renderWeather(mockCities.default);

  // Check if API key is set
  if (API_KEY && API_KEY.length > 10) {
    document.getElementById('apiNotice').className = 'api-notice live';
    document.getElementById('apiNotice').innerText = '✅ Live Mode — OpenWeatherMap API Active';
    fetchLive('new york');
  }
