<script>
(function(){
var sc='normal';
function fmt(n){return '₹'+Math.round(n).toLocaleString('en-IN');}
function toast(m){var t=document.getElementById('sf-toast');t.textContent=m;t.classList.add('show');setTimeout(function(){t.classList.remove('show');},2500);}

window.sfCalc=function(){
  var assets=Array.from(document.querySelectorAll('#sf-root .sf-asset')).reduce(function(s,i){return s+(parseFloat(i.value)||0);},0);
  var debt=parseFloat(document.getElementById('sf-debt').value)||0;
  var bis=Array.from(document.querySelectorAll('#sf-root .sf-bitem'));
  var base=bis.reduce(function(s,i){return s+(parseFloat(i.value)||0);},0);
  var tgt=parseInt(document.getElementById('sf-tgt').value)||18;
  document.getElementById('sf-tgt2').textContent=tgt+' months';
  document.getElementById('sf-tgt-lbl').textContent=tgt;
  var burn=sc==='lean'?base*0.7:sc==='growth'?base*1.5:base;
  var net=assets-debt;
  var rwy=burn>0?net/burn:999;
  var req=burn*tgt;
  var gap=Math.max(0,req-net);
  var pct=Math.min(100,(rwy/18)*100);
  document.getElementById('sf-cash').textContent=fmt(assets);
  document.getElementById('sf-burn').textContent=fmt(burn);
  document.getElementById('sf-runway').textContent=(rwy>=999?'∞':rwy.toFixed(1))+' mo';
  document.getElementById('sf-gap').textContent=fmt(gap);
  document.getElementById('sf-req').textContent=fmt(req);
  document.getElementById('sf-avail').textContent=fmt(net);
  document.getElementById('sf-gapd').textContent=gap>0?fmt(gap):'✅ Funded';
  var fill=document.getElementById('sf-fill');
  fill.style.width=pct+'%';
  fill.style.background=rwy<6?'var(--re)':rwy<12?'var(--ye)':'var(--gr)';
  var badge=document.getElementById('sf-badge');
  if(rwy<6){badge.textContent='🚨 Critical — Raise Immediately!';badge.className='sf-badge crit';}
  else if(rwy<12){badge.textContent='⚠️ Moderate — Plan Your Raise';badge.className='sf-badge warn';}
  else{badge.textContent='✅ Healthy Runway';badge.className='sf-badge ok';}
  var bkd=document.getElementById('sf-bkd');bkd.innerHTML='';
  var total=burn||1;
  bis.forEach(function(inp){
    var v=parseFloat(inp.value)||0;if(!v)return;
    var adj=sc==='lean'?v*0.7:sc==='growth'?v*1.5:v;
    var lbl=inp.getAttribute('data-lbl')||'Expense';
    var pct2=(adj/total*100).toFixed(0);
    bkd.innerHTML+='<div class="sf-bkr"><div class="sf-bkl">'+lbl+'</div><div class="sf-bkt"><div class="sf-bkf" style="width:'+pct2+'%"></div></div><div class="sf-bkv">'+fmt(adj)+'</div></div>';
  });
};

window.sfScene=function(s,el){
  sc=s;
  document.querySelectorAll('#sf-root .sf-sc2').forEach(function(c){c.classList.remove('on');});
  el.classList.add('on');sfCalc();
};

window.sfAdd=function(type){
  var lbl=prompt(type==='a'?'Asset का नाम:':'Expense का नाम:','New Item');
  if(!lbl)return;
  var div=document.createElement('div');div.className='sf-fg';
  if(type==='a'){
    div.innerHTML='<div class="sf-fl">'+lbl+'</div><div class="sf-row"><span class="sf-pfx">₹</span><input class="sf-fi pfx sf-asset" type="number" value="0" oninput="sfCalc()"/><button class="sf-rm" onclick="sfRm(this)">×</button></div>';
    document.getElementById('sf-alist').appendChild(div);
  } else {
    div.innerHTML='<div class="sf-fl">'+lbl+'</div><div class="sf-row"><span class="sf-pfx">₹</span><input class="sf-fi pfx sf-bitem" type="number" value="0" oninput="sfCalc()" data-lbl="'+lbl+'"/><button class="sf-rm" onclick="sfRm(this)">×</button></div>';
    document.getElementById('sf-blist').appendChild(div);
  }
  sfCalc();
};

window.sfRm=function(btn){btn.closest('.sf-fg').remove();sfCalc();};

window.sfReset=function(){
  var a=document.querySelectorAll('#sf-root .sf-asset');
  [500000,50000,100000].forEach(function(v,i){if(a[i])a[i].value=v;});
  var b=document.querySelectorAll('#sf-root .sf-bitem');
  [80000,20000].forEach(function(v,i){if(b[i])b[i].value=v;});
  document.getElementById('sf-debt').value=0;
  document.getElementById('sf-tgt').value=18;
  sc='normal';
  document.querySelectorAll('#sf-root .sf-sc2').forEach(function(c,i){c.classList.toggle('on',i===0);});
  sfCalc();toast('🔄 Reset हो गया!');
};

window.sfPDF=function(){
  if(typeof window.jspdf==='undefined'){toast('PDF library loading...');return;}
  var jsPDF=window.jspdf.jsPDF;
  var doc=new jsPDF();
  doc.setFontSize(18);doc.text('Startup Financial Health Report',14,18);
  doc.setFontSize(10);doc.setTextColor(100);
  doc.text('Date: '+new Date().toLocaleDateString('en-IN')+'  |  Scenario: '+sc.toUpperCase(),14,26);
  doc.autoTable({
    startY:32,
    head:[['Metric','Value']],
    body:[
      ['Total Cash Assets',document.getElementById('sf-cash').textContent],
      ['Monthly Burn Rate',document.getElementById('sf-burn').textContent],
      ['Runway',document.getElementById('sf-runway').textContent],
      ['Funding Gap',document.getElementById('sf-gap').textContent],
      ['Status',document.getElementById('sf-badge').textContent],
      ['Required Funding',document.getElementById('sf-req').textContent],
    ],
    theme:'grid',headStyles:{fillColor:[99,102,241]}
  });
  var rows=Array.from(document.querySelectorAll('#sf-root .sf-bitem')).map(function(i){return[i.getAttribute('data-lbl')||'Expense','₹'+Number(i.value).toLocaleString('en-IN')];});
  doc.autoTable({startY:doc.lastAutoTable.finalY+10,head:[['Expense','Monthly Amount']],body:rows,theme:'striped'});
  doc.save('Startup_Report.pdf');toast('📄 PDF Downloaded!');
};

sfCalc();
})();
