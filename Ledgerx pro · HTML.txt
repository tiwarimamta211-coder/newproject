function calcGST(){
  const d=parseInt(document.getElementById('g-d').value)||0;
  const rt=document.getElementById('g-rt').value;
  const up=parseFloat(document.getElementById('g-up').value)||0;
  const itc=parseFloat(document.getElementById('g-itc').value)||0;
  const ew=parseFloat(document.getElementById('g-ew').value)||0;
  const pd=rt==='nil'?20:50,mf=rt==='nil'?500:10000;
  const lf=Math.min(d*pd,mf),intr=(up*.18*d)/365;
  const ip=itc>0?Math.max(itc*.1,10000):0,ep=ew>0?Math.max(ew*.01,10000):0;
  const tot=lf+intr+ip+ep;
  S.gstRisk=tot;
  document.getElementById('gb-l').textContent=fmt(lf);
  document.getElementById('gb-i').textContent=fmt(intr);
  document.getElementById('gb-itc').textContent=fmt(ip);
  document.getElementById('gb-ew').textContent=fmt(ep);
  document.getElementById('gb-tot').textContent=fmt(tot);
  document.getElementById('g-amt').textContent=fmt(tot);
  const ae=document.getElementById('g-amt'),se=document.getElementById('g-st'),cb=document.getElementById('gstbdg'),cg=document.getElementById('chip-gst');
  cg.textContent=`GST: ${fmt(tot)}`;
  if(tot>50000){ae.style.color='var(--coral)';se.style.color='var(--coral)';se.textContent='🚨 CRITICAL — Immediate action required!';cg.className='chip neg';cb.style.display='inline-block';}
  else if(tot>5000){ae.style.color='var(--amber)';se.style.color='var(--amber)';se.textContent='⚠️ MODERATE RISK — File returns immediately';cg.className='chip warn';cb.style.display='inline-block';}
  else if(tot>0){ae.style.color='var(--mint)';se.style.color='var(--mint)';se.textContent='✅ LOW RISK — Standard late fees apply';cg.className='chip pos';cb.style.display='none';}
  else{ae.style.color='var(--sky)';se.style.color='var(--mint)';se.textContent='✅ No Risk Detected — Fully Compliant';cg.className='chip';cb.style.display='none';}
  document.getElementById('d-gst').textContent=fmt(tot);
}
