<script>
  // Mock base data
  var pollData = {
    coding:     450,
    reasoning:  320,
    multimodal: 210,
    agents:     440
  };

  // Session-only voted flag (no localStorage — Blogger-safe)
  var hasVoted = false;

  function init() {
    updateTotal();
  }

  function castVote(option) {
    if (hasVoted) return;
    pollData[option]++;
    hasVoted = true;
    document.getElementById('thankyou').style.display = 'block';
    showResults(option);
    updateTotal();
  }

  function showResults(votedOption) {
    document.getElementById('votingSection').style.display = 'none';
    var sec = document.getElementById('analyticsSection');
    sec.style.display = 'flex';

    var total = Object.values(pollData).reduce(function(a,b){ return a+b; }, 0);

    // Find winner
    var maxKey = Object.keys(pollData).reduce(function(a,b){ return pollData[a] > pollData[b] ? a : b; });

    Object.keys(pollData).forEach(function(key) {
      var pct = Math.round((pollData[key] / total) * 100);
      var bar = document.getElementById('bar-' + key);
      var valEl = document.getElementById('val-' + key);
      var cntEl = document.getElementById('cnt-' + key);

      // Add winner badge
      if (key === maxKey) {
        var label = document.querySelector('#row-' + key + ' .result-label');
        if (label && !label.querySelector('.winner-tag')) {
          var tag = document.createElement('span');
          tag.className = 'winner-tag';
          tag.innerText = '🏆 Leading';
          label.appendChild(tag);
        }
        bar.classList.add('winner');
      }

      cntEl.innerText = '(' + pollData[key].toLocaleString('en-IN') + ' votes)';

      setTimeout(function(p, b, v) {
        return function() {
          b.style.width = p + '%';
          v.innerText = p + '%';
        };
      }(pct, bar, valEl), 120);
    });
  }

  function updateTotal() {
    var total = Object.values(pollData).reduce(function(a,b){ return a+b; }, 0);
    document.getElementById('totalVotes').innerText = 'Total: ' + total.toLocaleString('en-IN') + ' votes';
  }

  function resetPoll() {
    hasVoted = false;
    // Reset counts back to base
    pollData = { coding: 450, reasoning: 320, multimodal: 210, agents: 440 };
    document.getElementById('votingSection').style.display = 'flex';
    document.getElementById('analyticsSection').style.display = 'none';
    document.getElementById('thankyou').style.display = 'none';
    // Reset bars
    ['coding','reasoning','multimodal','agents'].forEach(function(k){
      document.getElementById('bar-'+k).style.width = '0%';
      document.getElementById('val-'+k).innerText = '0%';
      document.getElementById('cnt-'+k).innerText = '';
    });
    updateTotal();
  }

  init();
</script>
