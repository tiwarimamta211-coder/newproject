function renderTransactions(data) {
  const body = document.getElementById('txnTableBody');
  if(!body) return;
  const catColors = {
    Revenue: { bg:'var(--accent-dim)', color:'var(--accent)' },

    Payroll: { bg:'var(--red-dim)', color:'var(--red)' },
  };
  body.innerHTML = data.map(t => {
    const c = catColors[t.cat] || { bg:'var(--bg4)', color:'var(--text3)' };
    const amtStr = (t.amount > 0 ? '+' : '') + '$' + Math.abs(t.amount).toLocaleString();
    const amtClass = t.amount > 0 ? 'pos' : 'neg';

    return `<div class="table-row">

      <div class="td" style="color:var(--text2);font-size:12px;">${t.date}</div>
      <div class="td mono" style="color:${amtColor};font-weight:500;">${amtStr}</div>
      <div class="td">
        <span class="status-dot" style="background:${statusColor};"></span>
        <span style="font-size:12px;color:var(--text2);">${t.status}</span>
      </div>
    </div>`;
  }).join('') || `<div style="padding:32px;text-align:center;color:var(--text3);">No transactions found.</div>`;
  document.getElementById('txnCount').textContent = `Showing ${data.length} of ${allTxns.length} transactions`;
}

function filterTransactions() {
  const search = document.getElementById('txnSearch')?.value.toLowerCase() || '';
  const type = document.getElementById('txnTypeFilter')?.value || '';
  const cat = document.getElementById('txnCatFilter')?.value || '';
  const filtered = allTxns.filter(t => {
    const matchSearch = !search || t.desc.toLowerCase().includes(search) || t.cat.toLowerCase().includes(search);
    const matchType = !type || t.type === type;
    const matchCat = !cat || t.cat === cat;
    return matchSearch && matchType && matchCat;
  });
  renderTransactions(filtered);
}

function addTransaction() {
  const desc = document.getElementById('txnDesc')?.value.trim();
  const amount = parseFloat(document.getElementById('txnAmount')?.value);
  const type = document.getElementById('txnType')?.value;
  const cat = document.getElementById('txnCat')?.value;
  const date = document.getElementById('txnDate')?.value;
  if(!desc || isNaN(amount)) { showToast('Please fill all required fields', 'error'); return; }
  const newTxn = {
    id: allTxns.length + 1,
    desc, cat, date: date || new Date().toISOString().split('T')[0],
    amount: type === 'expense' ? -Math.abs(amount) : Math.abs(amount),
    type, status: 'pending'
  };
  allTxns.unshift(newTxn);
  renderTransactions(allTxns);
  closeModal('addTxnModal');
  showToast('Transaction added successfully', 'success');
  document.getElementById('txnDesc').value = '';
  document.getElementById('txnAmount').value = '';
}

function getAIResponse(input) {
  const lower = input.toLowerCase();
  if(lower.includes('profit margin') || lower.includes('margin')) return AI_RESPONSES['profit margin'];
  if(lower.includes('q1') || lower.includes('q2') || lower.includes('expense')) return AI_RESPONSES['q1 vs q2'];
  if(lower.includes('owe') || lower.includes('client') || lower.includes('outstanding') || lower.includes('receivable')) return AI_RESPONSES['owe me money'];
  if(lower.includes('tax') || lower.includes('irs') || lower.includes('liability')) return AI_RESPONSES['tax'];
  if(lower.includes('category') || lower.includes('categories') || lower.includes('breakdown')) return AI_RESPONSES['expense categories'];
  return `<p>Great question! Based on your financial data for May 2026, here's what I can tell you:</p>
    <div class="highlight-box">Revenue: $84,291 | Expenses: $31,047 | Net Profit: $53,244</div>
    <p>I can analyze your <strong>profit margins</strong>, <strong>expense trends</strong>, <strong>outstanding invoices</strong>, or <strong>tax estimates</strong>. Try asking me something more specific about any of these areas!</p>`;
}

async function sendChat() {
  if(chatLoading) return;
  const input = document.getElementById('chatInput');
  const text = input.value.trim();
  if(!text) return;
  input.value = '';
  input.style.height = 'auto';
  appendMsg('user', `<p>${text}</p>`);
  chatLoading = true;
  document.getElementById('sendBtn').disabled = true;
  showTyping();
  await new Promise(r => setTimeout(r, 900 + Math.random()*600));
  removeTyping();
  const response = getAIResponse(text);
  appendMsg('ai', response);
  chatLoading = false;
  document.getElementById('sendBtn').disabled = false;
}

function sendSuggestion(chip) {
  const text = chip.textContent;
  document.getElementById('chatInput').value = text;
  sendChat();
}

function handleChatKey(e) {
  if(e.key === 'Enter' && !e.shiftKey) { e.preventDefault(); sendChat(); }
}

function autoResize(el) {
  el.style.height = 'auto';
  el.style.height = Math.min(el.scrollHeight, 120) + 'px';
}
