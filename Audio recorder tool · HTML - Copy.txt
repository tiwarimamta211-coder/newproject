function loadFromStorage() {
  try {
    const saved = localStorage.getItem('sf_recordings');
    if (saved) state.recordings = JSON.parse(saved);
    const credits = localStorage.getItem('sf_credits');
    if (credits) state.credits = parseInt(credits);
    const dl = localStorage.getItem('sf_dl');
    if (dl) state.downloadsLeft = parseInt(dl);
  } catch(e) {}
}

function saveToStorage() {
  try {
    // Only save metadata, not blobs (since blobs can't be serialized)
    const meta = state.recordings.map(r => ({
      id: r.id, name: r.name, duration: r.duration,
      size: r.size, format: r.format, createdAt: r.createdAt
    }));
    localStorage.setItem('sf_recordings', JSON.stringify(meta));
    localStorage.setItem('sf_credits', state.credits);
    localStorage.setItem('sf_dl', state.downloadsLeft);
  } catch(e) {}
}

// ─────────────────────────────────────────────
// MIC
// ─────────────────────────────────────────────
async function requestMicAccess() {
  if (state.micConnected) return;
  try {
    const constraints = {
      audio: {
        echoCancellation: state.features.echo,
        noiseSuppression: state.features.noise,
        sampleRate: 44100,
      }
    };
    state.stream = await navigator.mediaDevices.getUserMedia(constraints);
    state.micConnected = true;
    document.getElementById('micDot').classList.remove('off');

    // Get device name
    const devices = await navigator.mediaDevices.enumerateDevices();
    const mic = devices.find(d => d.kind === 'audioinput');
    document.getElementById('micName').textContent = mic?.label || 'Default Microphone';

    // Setup analyser
    state.audioCtx = new (window.AudioContext || window.webkitAudioContext)();
    state.analyser = state.audioCtx.createAnalyser();
    state.analyser.fftSize = 1024;
    state.source = state.audioCtx.createMediaStreamSource(state.stream);
    state.source.connect(state.analyser);

    showToast('Microphone connected!', 'success');
    startIdleVisualize();
  } catch(err) {
    showToast('Microphone access denied. Please allow in browser.', 'error');
  }
}

// RECORDING
// ─────────────────────────────────────────────
async function toggleRecord() {
  if (!state.micConnected) {
    await requestMicAccess();
    if (!state.micConnected) return;
  }
  if (state.isRecording) {
    stopRecording();
  } else {
    startRecording();
  }
}

function startRecording() {
  if (state.recordings.length >= FREE_RECS) {
    showToast('Free limit: 5 recordings. Upgrade for more!', 'warning');
    openUpgradeModal(); return;
  }
  if (state.credits < 5) {
    showToast('Not enough credits! Buy credits to record.', 'warning');
    openCreditsModal(); return;
  }

  state.audioChunks = [];
  const mimeType = getSupportedMime();

  try {
    state.mediaRecorder = new MediaRecorder(state.stream, { mimeType });
  } catch(e) {
    state.mediaRecorder = new MediaRecorder(state.stream);
  }

  state.mediaRecorder.ondataavailable = e => {
    if (e.data.size > 0) state.audioChunks.push(e.data);
  };

  state.mediaRecorder.onstop = () => saveRecording();
  state.mediaRecorder.start(100);

  state.isRecording = true;
  state.isPaused = false;
  state.elapsedSec = 0;

  // Deduct credits
  state.credits = Math.max(0, state.credits - 5);
  updateCreditsUI();

  // UI
  document.getElementById('btnRecord').classList.add('recording');
  document.getElementById('btnRecord').textContent = '⏹';
  document.getElementById('btnPause').disabled = false;
  document.getElementById('btnStop').disabled = false;
  document.getElementById('btnRestart').disabled = false;
  document.getElementById('statusDot').className = 'status-dot rec';
  document.getElementById('statusText').textContent = 'Recording...';
  document.getElementById('timerDisplay').className = 'timer-display recording';
  document.getElementById('visIdle').style.display = 'none';

  startTimer();
  startWaveform();
  showToast('Recording started! 5 credits used.', 'success');
}

function stopRecording() {
  if (!state.mediaRecorder) return;
  state.mediaRecorder.stop();
  state.isRecording = false;
  state.isPaused = false;
  stopTimer();
  stopWaveform();
  resetControls();
  showToast('Recording stopped & saved!', 'success');
}

function togglePause() {
  if (!state.isRecording) return;
  if (state.isPaused) {
    state.mediaRecorder.resume();
    state.isPaused = false;
    startTimer();
    startWaveform();
    document.getElementById('btnPause').querySelector('.ctrl-icon').textContent = '⏸';
    document.getElementById('statusDot').className = 'status-dot rec';
    document.getElementById('statusText').textContent = 'Recording...';
    document.getElementById('timerDisplay').className = 'timer-display recording';
    showToast('Resumed', 'success');
  } else {
    state.mediaRecorder.pause();
    state.isPaused = true;
    stopTimer();
    stopWaveform();
    document.getElementById('btnPause').querySelector('.ctrl-icon').textContent = '▶';
    document.getElementById('statusDot').className = 'status-dot paused';
    document.getElementById('statusText').textContent = 'Paused';
    document.getElementById('timerDisplay').className = 'timer-display paused';
    showToast('Paused', 'warning');
  }
}

function restartRecording() {
  if (state.isRecording) {
    state.mediaRecorder.stop();
    state.audioChunks = [];
    stopTimer();
    stopWaveform();
  }
  state.isRecording = false;
  state.isPaused = false;
  state.elapsedSec = 0;
  updateTimer();
  resetControls();
  showToast('Restarted — ready to record again', 'warning');
}

function skipBack() {
  state.elapsedSec = Math.max(0, state.elapsedSec - 5);
  updateTimer();
}

function resetControls() {
  const rec = document.getElementById('btnRecord');
  rec.classList.remove('recording', 'paused');
  rec.textContent = '⏺';
  document.getElementById('btnPause').disabled = true;
  document.getElementById('btnStop').disabled = true;
  document.getElementById('btnRestart').disabled = true;
  document.getElementById('btnPause').querySelector('.ctrl-icon').textContent = '⏸';
  document.getElementById('statusDot').className = 'status-dot';
  document.getElementById('statusText').textContent = 'Ready to Record';
  document.getElementById('timerDisplay').className = 'timer-display';
  document.getElementById('visIdle').style.display = '';
  document.getElementById('limitBarFill').style.width = '0%';
  document.getElementById('limitVal').textContent = '0:00 / 5:00';
}

function saveRecording() {
  const blob = new Blob(state.audioChunks, { type: getSupportedMime() });
  const url = URL.createObjectURL(blob);
  const id = Date.now();
  const rec = {
    id, url, blob,
    name: `Recording ${state.recordings.length + 1}`,
    duration: state.elapsedSec,
    size: blob.size,
    format: 'webm',
    createdAt: new Date().toISOString(),
  };
  state.recordings.unshift(rec);
  state.totalSeconds += state.elapsedSec;
  saveToStorage();
  renderRecordings();
  updateStats();
}

// ─────────────────────────────────────────────
// TIMER
// ─────────────────────────────────────────────
function startTimer() {
  state.timerInterval = setInterval(() => {
    state.elapsedSec++;
    updateTimer();

    // Free limit enforcement
    const maxSec = FREE_LIMIT_SEC;
    const pct = Math.min((state.elapsedSec / maxSec) * 100, 100);
    const fill = document.getElementById('limitBarFill');
    fill.style.width = pct + '%';
    fill.classList.toggle('warn', pct > 75);
    document.getElementById('limitVal').textContent =
      `${fmtShort(state.elapsedSec)} / ${fmtShort(maxSec)}`;

    if (state.elapsedSec >= maxSec) {
      showToast('Free 5-min limit reached! Upgrade for longer sessions.', 'warning');
      stopRecording();
      setTimeout(openUpgradeModal, 800);
    }
  }, 1000);
}

function stopTimer() { clearInterval(state.timerInterval); }

function updateTimer() {
  document.getElementById('timerDisplay').textContent = fmtTime(state.elapsedSec);
}

function fmtTime(sec) {
  const h = Math.floor(sec/3600).toString().padStart(2,'0');
  const m = Math.floor((sec%3600)/60).toString().padStart(2,'0');
  const s = (sec%60).toString().padStart(2,'0');
  return `${h}:${m}:${s}`;
}

function fmtShort(sec) {
  const m = Math.floor(sec/60);
  const s = (sec%60).toString().padStart(2,'0');
  return `${m}:${s}`;
}

// ─────────────────────────────────────────────
// WAVEFORM
// ─────────────────────────────────────────────
const canvas = document.getElementById('waveCanvas');
const ctx2d = canvas.getContext('2d');
let waveData = [];

function startWaveform() {
  if (!state.analyser) return;
  const draw = () => {
    if (!state.isRecording || state.isPaused) return;
    state.animFrame = requestAnimationFrame(draw);
    drawWave();
    drawVolMeter();
  };
  draw();
}

function stopWaveform() {
  cancelAnimationFrame(state.animFrame);
  // Draw flat line
  const w = canvas.offsetWidth, h = canvas.offsetHeight;
  canvas.width = w; canvas.height = h;
  ctx2d.clearRect(0,0,w,h);
  ctx2d.beginPath();
  ctx2d.moveTo(0, h/2);
  ctx2d.lineTo(w, h/2);
  ctx2d.strokeStyle = 'rgba(124,92,252,0.3)';
  ctx2d.lineWidth = 1.5;
  ctx2d.stroke();
  resetVolMeter();
}

function drawWave() {
  const w = canvas.offsetWidth, h = canvas.offsetHeight;
  canvas.width = w; canvas.height = h;
  const bufLen = state.analyser.fftSize;
  const data = new Uint8Array(bufLen);
  state.analyser.getByteTimeDomainData(data);

  ctx2d.clearRect(0,0,w,h);

  // Gradient line
  const grad = ctx2d.createLinearGradient(0,0,w,0);
  grad.addColorStop(0, 'rgba(124,92,252,0.5)');
  grad.addColorStop(0.5, 'rgba(124,92,252,1)');
  grad.addColorStop(1, 'rgba(34,211,238,0.7)');

  ctx2d.beginPath();
  ctx2d.lineWidth = 2;
  ctx2d.strokeStyle = grad;

  for (let i = 0; i < bufLen; i++) {
    const x = (i / bufLen) * w;
    const y = ((data[i] / 128) - 1) * (h/2) + h/2;
    i === 0 ? ctx2d.moveTo(x, y) : ctx2d.lineTo(x, y);
  }
  ctx2d.stroke();

  // Mirror fill
  ctx2d.beginPath();
  ctx2d.fillStyle = 'rgba(124,92,252,0.05)';
  for (let i = 0; i < bufLen; i++) {
    const x = (i/bufLen)*w;
    const y = ((data[i]/128)-1)*(h/2)+h/2;
    i===0 ? ctx2d.moveTo(x,y) : ctx2d.lineTo(x,y);
  }
  ctx2d.lineTo(w, h/2);
  ctx2d.lineTo(0, h/2);
  ctx2d.fill();
}

function buildIdleWave() {
  // Dotted idle animation
  const w = canvas.offsetWidth || 600;
  const h = canvas.offsetHeight || 160;
  canvas.width = w; canvas.height = h;
  let t = 0;
  function animIdle() {
    if (state.isRecording) return;
    requestAnimationFrame(animIdle);
    t += 0.02;
    ctx2d.clearRect(0,0,w,h);
    ctx2d.beginPath();
    ctx2d.setLineDash([4, 8]);
    ctx2d.strokeStyle = 'rgba(124,92,252,0.2)';
    ctx2d.lineWidth = 1.5;
    ctx2d.moveTo(0, h/2 + Math.sin(t)*3);
    for (let x=0;x<=w;x+=5) {
      const y = h/2 + Math.sin((x/w)*Math.PI*4 + t)*4;
      ctx2d.lineTo(x,y);
    }
    ctx2d.stroke();
    ctx2d.setLineDash([]);
  }
  animIdle();
}

function startIdleVisualize() {
  // Volume-responsive idle (before recording)
  if (!state.analyser) return;
  const tick = () => {
    if (state.isRecording) return;
    requestAnimationFrame(tick);
    drawVolMeter();
  };
  tick();
}

// VOL METER
// ─────────────────────────────────────────────
function buildVolMeter() {
  const el = document.getElementById('volMeter');
  el.innerHTML = '';
  for (let i=0;i<20;i++) {
    const bar = document.createElement('div');
    bar.className = 'vol-bar';
    bar.style.height = '4px';
    el.appendChild(bar);
  }
}

function drawVolMeter() {
  if (!state.analyser) return;
  const data = new Uint8Array(state.analyser.frequencyBinCount);
  state.analyser.getByteFrequencyData(data);
  const bars = document.querySelectorAll('.vol-bar');
  const step = Math.floor(data.length / bars.length);
  bars.forEach((bar, i) => {
    const val = data[i * step] / 255;
    const h = Math.max(4, val * 24);
    bar.style.height = h + 'px';
    const r = val > 0.7 ? 'var(--red)' : val > 0.4 ? 'var(--accent)' : 'var(--accent)';
    bar.style.background = r;
    bar.style.opacity = 0.4 + val * 0.6;
  });
}

function resetVolMeter() {
  document.querySelectorAll('.vol-bar').forEach(b => {
    b.style.height = '4px'; b.style.opacity = '0.3';
  });
}

      <div style="font-size:0.75rem;color:var(--text3);padding:4px 0;">

      </div>`}
    </div>
  `).join('');
}

// Audio players map
const players = {};

function playRec(id) {
  const rec = state.recordings.find(r => r.id == id);
  if (!rec || !rec.url) return;

  if (!players[id]) {
    players[id] = new Audio(rec.url);
    players[id].ontimeupdate = () => {
      const pct = (players[id].currentTime / players[id].duration) * 100;
      const fill = document.getElementById(`fill-${id}`);
      const time = document.getElementById(`time-${id}`);
      if (fill) fill.style.width = pct + '%';
      if (time) time.textContent = fmtTime(Math.floor(players[id].currentTime));
    };
    players[id].onended = () => {
      document.getElementById(`play-${id}`).textContent = '▶';
    };
  }

  const audio = players[id];
  const btn = document.getElementById(`play-${id}`);
  if (audio.paused) {
    // Pause all others
    Object.entries(players).forEach(([pid, p]) => {
      if (pid != id && !p.paused) {
        p.pause();
        const ob = document.getElementById(`play-${pid}`);
        if (ob) ob.textContent = '▶';
      }
    });
    audio.play();
    btn.textContent = '⏸';
  } else {
    audio.pause();
    btn.textContent = '▶';
  }
}

function seekRec(e, id) {
  const rec = state.recordings.find(r => r.id == id);
  if (!rec || !players[id]) return;
  const seek = document.getElementById(`seek-${id}`);
  const pct = e.offsetX / seek.offsetWidth;
  players[id].currentTime = pct * players[id].duration;
}

function downloadRec(id) {
  if (state.downloadsLeft <= 0) {
    showToast('Daily download limit reached! Buy credits or upgrade.', 'warning');
    openCreditsModal(); return;
  }
  const rec = state.recordings.find(r => r.id == id);
  if (!rec || !rec.url) { showToast('Audio not available in current session', 'error'); return; }

  const a = document.createElement('a');
  a.href = rec.url;
  a.download = `${rec.name}.${rec.format || 'webm'}`;
  a.click();

  state.downloadsLeft = Math.max(0, state.downloadsLeft - 1);
  saveToStorage();
  updateStats();
  updateDlLimit();
  showToast(`Downloaded! ${state.downloadsLeft} downloads left today.`, 'success');
}

function shareRec(id) {
  const rec = state.recordings.find(r => r.id == id);
  if (!rec) return;
  if (navigator.clipboard) {
    navigator.clipboard.writeText(`Recording: ${rec.name} (${fmtTime(rec.duration||0)})`);
    showToast('Recording info copied to clipboard!', 'success');
  }
}

function deleteRec(id) {
  const idx = state.recordings.findIndex(r => r.id == id);
  if (idx === -1) return;
  if (players[id]) { players[id].pause(); delete players[id]; }
  state.recordings.splice(idx, 1);
  saveToStorage();
  renderRecordings();
  updateStats();
  showToast('Recording deleted', 'warning');
}

function renameRec(id, name) {
  const rec = state.recordings.find(r => r.id == id);
  if (rec) { rec.name = name || rec.name; saveToStorage(); }
}

function clearAllRecordings() {
  if (!state.recordings.length) return;
  if (!confirm('Delete all recordings?')) return;
  Object.values(players).forEach(p => p.pause());
  state.recordings = [];
  saveToStorage();
  renderRecordings();
  updateStats();
  showToast('All recordings cleared', 'warning');
}

function handleFileImport(e) {
  const file = e.target.files[0];
  if (file) processImportedFile(file);
}

function processImportedFile(file) {
  if (file.size > 50 * 1024 * 1024) {
    showToast('File too large! Free limit: 50MB. Upgrade for 500MB.', 'error');
    return;
  }
  const url = URL.createObjectURL(file);
  const audio = new Audio(url);
  audio.addEventListener('loadedmetadata', () => {
    const rec = {
      id: Date.now(),
      url, blob: file,
      name: file.name.replace(/\.[^.]+$/, ''),
      duration: Math.floor(audio.duration),
      size: file.size,
      format: file.name.split('.').pop(),
      createdAt: new Date().toISOString(),
    };
    state.recordings.unshift(rec);
    saveToStorage();
    renderRecordings();
    updateStats();
    showToast(`Imported: ${rec.name}`, 'success');
    switchTab('record');
  });
}

// ─────────────────────────────────────────────
// MIME
// ─────────────────────────────────────────────
function getSupportedMime() {
  const types = ['audio/webm;codecs=opus','audio/webm','audio/ogg;codecs=opus','audio/ogg','audio/mp4'];
  for (const t of types) {
    if (MediaRecorder.isTypeSupported(t)) return t;
  }
  return '';
}

// ─────────────────────────────────────────────
// FORMAT HELPERS
// ─────────────────────────────────────────────
function fmtSize(bytes) {
  if (bytes < 1024) return bytes + ' B';
  if (bytes < 1024*1024) return (bytes/1024).toFixed(1) + ' KB';
  return (bytes/(1024*1024)).toFixed(2) + ' MB';
}

function fmtDate(iso) {
  if (!iso) return '';
  const d = new Date(iso);
  return d.toLocaleDateString('en-IN', { day:'2-digit', month:'short' });
}

// ─────────────────────────────────────────────
// TOAST
// ─────────────────────────────────────────────
function showToast(msg, type='success') {
  const area = document.getElementById('toast-area');
  const icons = {success:'✓', error:'✕', warning:'⚠'};
  const colors = {success:'var(--green)', error:'var(--red)', warning:'var(--amber)'};
  const t = document.createElement('div');
  t.className = `toast ${type}`;
  t.innerHTML = `<span style="color:${colors[type]};font-weight:700;">${icons[type]}</span><span>${msg}</span>`;
  area.appendChild(t);
  setTimeout(() => { t.classList.add('out'); setTimeout(() => t.remove(), 300); }, 3500);
}

document.getElementById('qualitySelect').addEventListener('change', function() {
  if (this.value !== '128') {
    showToast('High quality requires Pro. Upgrade to unlock!', 'warning');
    this.value = '128';
    openUpgradeModal();
  }
});

document.getElementById('durationSelect').addEventListener('change', function() {
  if (this.value !== '300') {
    showToast('Longer sessions require Pro. Upgrade to unlock!', 'warning');
    this.value = '300';
    openUpgradeModal();
  }
});

document.getElementById('channelsSelect').addEventListener('change', function() {
  if (this.value !== 'mono') {
    showToast('Stereo requires Pro. Upgrade to unlock!', 'warning');
    this.value = 'mono';
    openUpgradeModal();
  }
});

// Keyboard shortcuts
document.addEventListener('keydown', e => {
  if (['INPUT','TEXTAREA','SELECT'].includes(e.target.tagName)) return;
  if (e.code === 'Space') { e.preventDefault(); toggleRecord(); }
  if (e.code === 'KeyP') togglePause();
  if (e.code === 'KeyS') stopRecording();
  if (e.code === 'KeyR') restartRecording();
});
