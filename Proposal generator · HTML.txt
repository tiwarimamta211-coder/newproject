// ── NAVIGATION ──
function navigate(section) {
  document.querySelectorAll('.panel').forEach(p => p.classList.remove('active'));
  document.querySelectorAll('.sidebar-item').forEach(i => i.classList.remove('active'));
  document.querySelectorAll('.step-item').forEach(i => i.classList.remove('active'));
  document.querySelectorAll('.mobile-nav-item').forEach(i => i.classList.remove('active'));

  document.getElementById('panel-' + section).classList.add('active');
  document.getElementById('step-' + section).classList.add('active');

  const sidebarItems = document.querySelectorAll('.sidebar-item');
  const sectionMap = { info: 0, assets: 1, liabilities: 2, fees: 3, preview: 4 };
  if (sidebarItems[sectionMap[section]]) sidebarItems[sectionMap[section]].classList.add('active');

  const mobileItems = document.querySelectorAll('.mobile-nav-item');
  if (mobileItems[sectionMap[section]]) mobileItems[sectionMap[section]].classList.add('active');

  if (section === 'preview') buildPreview();
  updateSummary();
  window.scrollTo({ top: 0, behavior: 'smooth' });
}

// ── CURRENCY ──
function cur() { return document.getElementById('currency').value || '₹'; }
function fmt(n) { return cur() + Number(n).toLocaleString('en-IN', { minimumFractionDigits: 2, maximumFractionDigits: 2 }); }

// ── ASSETS ──
function addAsset() {
  const name = document.getElementById('assetName').value.trim();
  const val = parseFloat(document.getElementById('assetValue').value);
  if (!name) { toast('Please enter an asset name.', 'error'); return; }
  if (isNaN(val) || val < 0) { toast('Please enter a valid asset value.', 'error'); return; }
  state.assets.push({
    id: assetIdCounter++,
    name,
    category: document.getElementById('assetCategory').value,
    desc: document.getElementById('assetDesc').value.trim(),
    value: val
  });
  clearAssetForm();
  renderAssets();
  updateSummary();
  toast('Asset added successfully!', 'success');
}

function deleteAsset(id) {
  state.assets = state.assets.filter(a => a.id !== id);
  renderAssets();
  updateSummary();
  toast('Asset removed.', '');
}

function clearAssetForm() {
  ['assetName','assetValue','assetDesc'].forEach(id => document.getElementById(id).value = '');
  document.getElementById('assetCategory').selectedIndex = 0;
}

function renderAssets() {
  const tbody = document.getElementById('assetsTbody');
  const table = document.getElementById('assetsTable');
  const empty = document.getElementById('assetsEmpty');
  tbody.innerHTML = '';
  if (state.assets.length === 0) {
    table.style.display = 'none'; empty.style.display = 'block';
  } else {
    table.style.display = 'table'; empty.style.display = 'none';
    state.assets.forEach(a => {
      tbody.innerHTML += `<tr>
        <td><strong>${esc(a.name)}</strong></td>
        <td><span class="tag tag-asset">${esc(a.category)}</span></td>
        <td class="text-muted">${esc(a.desc || '—')}</td>
        <td class="amount">${fmt(a.value)}</td>
        <td class="actions"><button class="btn btn-danger btn-sm" onclick="deleteAsset(${a.id})">✕</button></td>
      </tr>`;
    });
  }
  const total = state.assets.reduce((s, a) => s + a.value, 0);
  document.getElementById('totalAssetsDisplay').textContent = fmt(total);
  document.getElementById('assetCountDisplay').textContent = state.assets.length;
  const max = state.assets.length ? Math.max(...state.assets.map(a => a.value)) : 0;
  document.getElementById('largestAssetDisplay').textContent = max ? fmt(max) : '—';
  document.getElementById('assetPill').textContent = state.assets.length + ' item' + (state.assets.length !== 1 ? 's' : '');
}

// ── PDF EXPORT ──
async function exportPDF() {
  const el = document.getElementById('preview-content');
  if (!el || !el.innerHTML.trim()) {
    buildPreview();
    await new Promise(r => setTimeout(r, 200));
  }

  const { jsPDF } = window.jspdf;
  const doc = new jsPDF({ orientation: 'portrait', unit: 'mm', format: 'a4' });

  const g = id => (document.getElementById(id) || {}).value || '';
  const firmName = g('firmName') || 'Proposal';
  const clientName = g('clientName') || 'Client';
  const propDate = g('proposalDate') || new Date().toISOString().split('T')[0];
  const propNo = g('proposalNo') || ('PROP-' + Date.now().toString().slice(-6));
  const advisorName = g('advisorName');
  const advisorEmail = g('advisorEmail');
  const intro = g('introText');
  const remarks = g('remarksText');
  const firmTag = g('firmTagline');
  const clientEmail = g('clientEmail');
  const validUntil = g('validUntil');

  const totalA = state.assets.reduce((s, a) => s + a.value, 0);
  const totalL = state.liabilities.reduce((s, l) => s + l.value, 0);
  const totalF = state.fees.reduce((s, f) => s + f.value, 0);
  const net = totalA - totalL;

  const margin = 18;
  const pageW = 210;
  const contentW = pageW - margin * 2;
  let y = 0;

  function checkPage(needed = 20) {
    if (y + needed > 270) { doc.addPage(); y = margin; }
  }

  // Header bar
  doc.setFillColor(15, 14, 12);
  doc.rect(0, 0, 210, 28, 'F');
  doc.setTextColor(247, 244, 238);
  doc.setFont('helvetica', 'bold');
  doc.setFontSize(18);
  doc.text(firmName, margin, 13);
  doc.setFont('helvetica', 'normal');
  doc.setFontSize(9);
  if (firmTag) doc.text(firmTag, margin, 20);
  doc.setFontSize(9);
  doc.text('Financial Proposal · ' + propNo, pageW - margin, 11, { align: 'right' });
  doc.text('Date: ' + propDate, pageW - margin, 17, { align: 'right' });
  if (validUntil) doc.text('Valid Until: ' + validUntil, pageW - margin, 23, { align: 'right' });

  y = 36;

  // To
  doc.setTextColor(138, 128, 112);
  doc.setFont('helvetica', 'bold');
  doc.setFontSize(8);
  doc.text('PREPARED FOR', margin, y);
  y += 5;
  doc.setTextColor(15, 14, 12);
  doc.setFontSize(14);
  doc.text(clientName, margin, y);
  y += 5;
  if (clientEmail) {
    doc.setFontSize(9); doc.setTextColor(138,128,112); doc.setFont('helvetica','normal');
    doc.text(clientEmail, margin, y); y += 5;
  }
  if (advisorName) {
    doc.setFontSize(9); doc.setTextColor(138,128,112);
    doc.text('Advisor: ' + advisorName + (advisorEmail ? '  ·  ' + advisorEmail : ''), margin, y);
    y += 5;
  }
  y += 4;

  // Intro
  if (intro) {
    doc.setFillColor(247, 244, 238);
    const lines = doc.splitTextToSize(intro, contentW - 8);
    const h = lines.length * 5 + 8;
    checkPage(h + 6);
    doc.rect(margin, y, contentW, h, 'F');
    doc.setFillColor(196, 120, 60);
    doc.rect(margin, y, 2, h, 'F');
    doc.setTextColor(74, 69, 64);
    doc.setFontSize(9.5); doc.setFont('helvetica','normal');
    doc.text(lines, margin + 6, y + 6);
    y += h + 10;
  }

  // Section helper
  function sectionTitle(title) {
    checkPage(14);
    doc.setTextColor(15, 14, 12);
    doc.setFont('helvetica', 'bold');
    doc.setFontSize(11);
    doc.text(title, margin, y);
    y += 1;
    doc.setDrawColor(200, 193, 182);
    doc.line(margin, y + 2, margin + contentW, y + 2);
    y += 8;
  }

  function tableHeader(cols, widths) {
    doc.setFillColor(15, 14, 12);
    doc.rect(margin, y, contentW, 7, 'F');
    doc.setTextColor(247, 244, 238);
    doc.setFont('helvetica', 'bold');
    doc.setFontSize(8);
    let x = margin + 3;
    cols.forEach((c, i) => {
      if (i === cols.length - 1) doc.text(c, margin + contentW - 3, y + 5, { align: 'right' });
      else doc.text(c, x, y + 5);
      x += widths[i];
    });
    y += 7;
  }

  function tableRow(cells, widths, isLast, colored) {
    checkPage(8);
    if (!isLast) {
      doc.setDrawColor(232, 225, 212);
      doc.line(margin, y + 6, margin + contentW, y + 6);
    }
    doc.setTextColor(colored || 15, colored ? 0 : 14, colored ? 0 : 12);
    doc.setFont('helvetica', 'normal');
    doc.setFontSize(9);
    let x = margin + 3;
    cells.forEach((c, i) => {
      if (i === cells.length - 1) doc.text(String(c), margin + contentW - 3, y + 5, { align: 'right' });
      else { const lines = doc.splitTextToSize(String(c), widths[i] - 4); doc.text(lines, x, y + 5); }
      x += widths[i];
    });
    y += 7;
  }

  function tableFooter(label, val) {
    checkPage(9);
    doc.setFillColor(247, 244, 238);
    doc.rect(margin, y, contentW, 8, 'F');
    doc.setDrawColor(15, 14, 12);
    doc.line(margin, y, margin + contentW, y);
    doc.setFont('helvetica', 'bold');
    doc.setFontSize(9.5);
    doc.setTextColor(15, 14, 12);
    doc.text(label, margin + 3, y + 6);
    doc.text(val, margin + contentW - 3, y + 6, { align: 'right' });
    y += 12;
  }

  // Assets
  sectionTitle('Assets');
  tableHeader(['Asset Name', 'Category', 'Value'], [80, 50, 44]);
  state.assets.forEach((a, i) => tableRow([a.name, a.category, fmt(a.value)], [80, 50, 44], i === state.assets.length - 1));
  if (!state.assets.length) { doc.setFontSize(9); doc.setTextColor(170,170,170); doc.text('No assets recorded.', margin+3, y+5); y+=10; }
  tableFooter('Total Assets', fmt(totalA));

  // Liabilities
  sectionTitle('Liabilities');
  tableHeader(['Liability', 'Type', 'Amount'], [80, 50, 44]);
  state.liabilities.forEach((l, i) => tableRow([l.name, l.type, fmt(l.value)], [80, 50, 44], i === state.liabilities.length - 1));
  if (!state.liabilities.length) { doc.setFontSize(9); doc.setTextColor(170,170,170); doc.text('No liabilities recorded.', margin+3, y+5); y+=10; }
  tableFooter('Total Liabilities', fmt(totalL));

  // Net Worth box
  checkPage(18);
  doc.setFillColor(15, 14, 12);
  doc.rect(pageW - margin - 70, y, 70, 16, 'F');
  doc.setTextColor(138, 128, 112);
  doc.setFontSize(7.5); doc.setFont('helvetica','bold');
  doc.text('NET WORTH', pageW - margin - 5, y + 6, { align: 'right' });
  doc.setFontSize(13); doc.setFont('helvetica','bold');
  doc.setTextColor(net >= 0 ? 126 : 240, net >= 0 ? 214 : 128, net >= 0 ? 165 : 128);
  doc.text(fmt(net), pageW - margin - 5, y + 13, { align: 'right' });
  y += 22;

  // Fees
  sectionTitle('Proposed Services & Fees');
  tableHeader(['Service', 'Frequency', 'Description', 'Fee'], [55, 35, 50, 34]);
  state.fees.forEach((f, i) => tableRow([f.name, f.freq, f.desc || '', fmt(f.value)], [55, 35, 50, 34], i === state.fees.length - 1));
  if (!state.fees.length) { doc.setFontSize(9); doc.setTextColor(170,170,170); doc.text('No fees defined.', margin+3, y+5); y+=10; }
  if (state.fees.length) tableFooter('Total Proposed Fees', fmt(totalF));

  // Remarks
  if (remarks) {
    sectionTitle('Remarks & Terms');
    doc.setFillColor(247, 244, 238);
    const lines = doc.splitTextToSize(remarks, contentW - 8);
    const h = lines.length * 5 + 8;
    checkPage(h + 4);
    doc.rect(margin, y, contentW, h, 'F');
    doc.setFont('helvetica','normal'); doc.setFontSize(9); doc.setTextColor(74,69,64);
    doc.text(lines, margin + 4, y + 6);
    y += h + 10;
  }

  // Signatures
  checkPage(50);
  y += 10;
  doc.setDrawColor(15, 14, 12);
  doc.line(margin, y + 2, margin + 2, y + 2);
  doc.line(margin, y + 2, margin + 80, y + 2);
  doc.line(margin + contentW - 80, y + 2, margin + contentW, y + 2);
  doc.setFont('helvetica','normal'); doc.setFontSize(8.5); doc.setTextColor(138,128,112);
  doc.text('Authorized Signatory — ' + firmName, margin, y + 8);
  if (advisorName) doc.text(advisorName, margin, y + 13);
  doc.text('Client Acceptance — ' + clientName, margin + contentW - 80, y + 8);
  y += 22;

  // Footer strip
  doc.setFillColor(15, 14, 12);
  doc.rect(0, 285, 210, 12, 'F');
  doc.setTextColor(138, 128, 112);
  doc.setFontSize(7.5); doc.setFont('helvetica','normal');
  doc.text('Generated with ProposalCraft · Confidential Document · ' + new Date().getFullYear(), 105, 292, { align: 'center' });

  const filename = `${clientName.replace(/\s+/g,'_')}_Proposal_${propNo}.pdf`;
  doc.save(filename);
  toast('PDF exported: ' + filename, 'success');
}

// ── RESET ──
function resetAll() {
  if (!confirm('Reset all data? This cannot be undone.')) return;
  state.assets = []; state.liabilities = []; state.fees = [];
  assetIdCounter = liabIdCounter = feeIdCounter = 1;
  ['firmName','firmTagline','advisorName','advisorEmail','clientName','clientEmail',
   'proposalDate','validUntil','proposalNo','introText','remarksText'].forEach(id => {
    const el = document.getElementById(id);
    if (el) el.value = '';
  });
  document.getElementById('currency').value = '₹';
  renderAssets(); renderLiabilities(); renderFees();
  navigate('info');
  toast('All data cleared.', '');
}

// ── HELPERS ──
function esc(str) {
  return String(str || '').replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;');
}

let toastTimer;
function toast(msg, type) {
  const el = document.getElementById('toast');
  el.textContent = (type === 'success' ? '✓ ' : type === 'error' ? '✕ ' : 'ℹ ') + msg;
  el.className = 'show ' + (type || '');
  clearTimeout(toastTimer);
  toastTimer = setTimeout(() => el.className = '', 2800);
}
