// BUILD FULL PROMPT
function buildFullPrompt() {
  const main = document.getElementById('mainPrompt').value.trim();
  const bg = document.getElementById('bgSelect').value;
  const styles = getActiveChips('styleChips');
  const lights = getActiveChips('lightChips');
  const neg = document.getElementById('negPrompt').value.trim();
  const angle = document.getElementById('angleSelect').value;
  const color = document.getElementById('colorAccent').value.trim();
  const detailVal = ['', 'draft quality', 'high quality, detailed', 'ultra high resolution, 8K quality'][document.getElementById('detailSlider').value];

  let parts = [];
  if (main) parts.push(main);
  if (bg) parts.push(bg);
  if (angle) parts.push(angle);
  if (styles.length) parts.push(styles.join(', ') + ' style');
  if (lights.length) parts.push(lights.join(', ') + ' lighting');
  if (color) parts.push(`${color} color accent`);
  parts.push(detailVal);
  parts.push('professional product photography, commercial grade');

  let prompt = parts.filter(Boolean).join(', ');
  if (neg) prompt += `\n\nNegative: ${neg}`;
  return prompt;
}

  const main = document.getElementById('mainPrompt').value.trim();
  if (!main) { showToast('Please write a product description first', '⚠', 3000); return; }

  btn.disabled = true;

  try {
    const resp = await fetch('https://api.anthropic.com/v1/messages', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json', 'x-api-key': apiKey, 'anthropic-version': '2023-06-01' },

        model: 'claude-sonnet-4-20250514',

        messages: [{
          role: 'user',

  const styles = getActiveChips('styleChips');
  const lights = getActiveChips('lightChips');
  const format = document.getElementById('formatSelect').value;

  // Create placeholder cards
  const placeholders = [];
  for (let i = 0; i < batchCount; i++) {
    const card = createLoadingCard(fullPrompt, styles, lights, format);
    placeholders.push(card);
    document.getElementById('galleryGrid').prepend(card);
  }

  updateCounts();

  // Generate each image (simulated with Claude text — in real deployment, connect to image gen API)

async function generateSingle(apiKey, prompt, card, idx, styles, lights, format) {
  const imgDisplay = card.querySelector('.img-display');
  const skeletonText = card.querySelector('.skeleton-text');
  
  // Cycle loading messages
  const loadMsgs = ['Building scene...', 'Lighting the set...', 'Rendering textures...', 'Final polish...'];
  let mi = 0;
  const msgInterval = setInterval(() => {
    if (skeletonText) skeletonText.textContent = loadMsgs[mi++ % loadMsgs.length];
  }, 1800);

  try {
    // Using Claude to generate a detailed image description + get an image
    // Since Claude itself is text-only, we generate a descriptive prompt and then use
    // a free image placeholder service to simulate the output visually.
    // In production, replace with DALL-E, Stability AI, or Replicate API.

    const resp = await fetch('https://api.anthropic.com/v1/messages', {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'x-api-key': apiKey,
        'anthropic-version': '2023-06-01'
      },
      body: JSON.stringify({
        model: 'claude-sonnet-4-20250514',
        max_tokens: 600,
        messages: [{
          role: 'user',
          content: `You are a creative director for luxury product photography. Given this prompt, create a vivid, detailed scene description (2-3 sentences) that captures exactly what this generated image would look like. Be specific about materials, lighting, composition, and mood. Also suggest a single specific color palette (3 hex codes) that fits the scene.\n\nPrompt: ${prompt}\n\nReturn JSON only: {"description": "...", "palette": ["#xxx","#xxx","#xxx"], "title": "short catchy title (4 words max)"}`
        }]
      })
    });

    const data = await resp.json();
    clearInterval(msgInterval);

    let parsed = { description: prompt, palette: ['#7c5cfc','#c8b4ff','#1a1a26'], title: 'Generated Image' };
    try {
      const raw = data.content?.[0]?.text || '{}';
      const clean = raw.replace(/```json|```/g, '').trim();
      parsed = JSON.parse(clean);
    } catch(e) {}

    // Create visual representation using canvas
    const imgObj = {
      id: Date.now() + idx,
      prompt: prompt,
      styles: styles,
      lights: lights,
      format: format,
      size: currentSize,
      description: parsed.description,
      palette: parsed.palette,
      title: parsed.title || 'Product Image',
    };

    generatedImages.push(imgObj);

    // Render card
    clearInterval(msgInterval);
    const canvas = document.createElement('canvas');
    const W = 512, H = currentSize === 'portrait' ? 768 : currentSize === 'landscape' ? 288 : 512;
    canvas.width = W; canvas.height = H;
    const ctx = canvas.getContext('2d');

    // Draw gradient background
    const [c1, c2, c3] = (parsed.palette || ['#7c5cfc','#1a1a26','#c8b4ff']).map(c => c || '#7c5cfc');
    const grad = ctx.createLinearGradient(0, 0, W, H);
    grad.addColorStop(0, c1);
    grad.addColorStop(0.5, c2);
    grad.addColorStop(1, c3);
    ctx.fillStyle = grad;
    ctx.fillRect(0, 0, W, H);

    // Add noise texture
    for (let x = 0; x < W; x += 4) {
      for (let y = 0; y < H; y += 4) {
        const noise = Math.random() * 0.06;
        ctx.fillStyle = `rgba(255,255,255,${noise})`;
        ctx.fillRect(x, y, 2, 2);
      }
    }

    // Draw product symbol (geometric abstract)
    ctx.save();
    ctx.globalAlpha = 0.6;
    const cx = W/2, cy = H/2;
    // Outer ring
    ctx.beginPath();
    ctx.arc(cx, cy, Math.min(W,H)*0.28, 0, Math.PI*2);
    ctx.strokeStyle = 'rgba(255,255,255,0.3)';
    ctx.lineWidth = 1;
    ctx.stroke();
    // Inner shape
    ctx.beginPath();
    ctx.arc(cx, cy, Math.min(W,H)*0.15, 0, Math.PI*2);
    ctx.fillStyle = 'rgba(255,255,255,0.15)';
    ctx.fill();
    // Star/sparkle
    for (let a = 0; a < 8; a++) {
      const angle = (a / 8) * Math.PI * 2;
      const r = a % 2 === 0 ? Math.min(W,H)*0.22 : Math.min(W,H)*0.10;
      const px = cx + Math.cos(angle) * r;
      const py = cy + Math.sin(angle) * r;
      if (a === 0) ctx.moveTo(px, py);
      else ctx.lineTo(px, py);
    }
    ctx.closePath();
    ctx.fillStyle = 'rgba(255,255,255,0.2)';
    ctx.fill();
    ctx.restore();

    // Title text
    ctx.font = `bold ${Math.floor(W*0.045)}px Syne, sans-serif`;
    ctx.fillStyle = 'rgba(255,255,255,0.9)';
    ctx.textAlign = 'center';
    ctx.fillText(parsed.title || 'Product Image', cx, H - 30);

    // ✦ mark
    ctx.font = `${Math.floor(W*0.055)}px sans-serif`;
    ctx.fillStyle = 'rgba(255,255,255,0.5)';
    ctx.fillText('✦', cx, cy * 0.45);

    const dataURL = canvas.toDataURL('image/' + (format === 'JPEG' ? 'jpeg' : format === 'PNG' ? 'png' : 'webp'), 0.92);
    imgObj.dataURL = dataURL;

    renderSuccessCard(card, imgObj);

  } catch(e) {
    clearInterval(msgInterval);
    renderErrorCard(card, e.message || 'Generation failed');
  }
}

function renderSuccessCard(card, imgObj) {
  card.classList.remove('loading');
  const ratio = imgObj.size === 'portrait' ? 'portrait' : imgObj.size === 'landscape' ? 'landscape' : '';
  const styleTags = imgObj.styles.slice(0,2).map(s => `<span class="img-tag style-tag">${escapeHtml(s)}</span>`).join('');
  const lightTags = imgObj.lights.slice(0,1).map(s => `<span class="img-tag">${escapeHtml(s)}</span>`).join('');

  card.innerHTML = `
    <div class="img-display ${ratio}">
      <img src="${imgObj.dataURL}" alt="Generated product image" loading="lazy" />
    </div>
    <div class="img-meta">
      <div class="img-prompt">${escapeHtml(imgObj.description || imgObj.prompt)}</div>
      <div class="img-tags">
        ${styleTags}${lightTags}
        <span class="img-tag">${escapeHtml(imgObj.format)}</span>
      </div>
    </div>
    <div class="img-actions">
      <button class="act-btn primary" onclick="viewModal(${imgObj.id})">🔍 View</button>
      <button class="act-btn" onclick="downloadImg(${imgObj.id})">↓ Save</button>
      <button class="act-btn" onclick="copyPromptText(${imgObj.id})">📋 Prompt</button>
    </div>
  `;
}

}

}

function closeModal(e) {
  if (e.target === document.getElementById('modal')) closeModalBtn();
}
function closeModalBtn() { document.getElementById('modal').classList.remove('open'); }

// DOWNLOAD
function downloadImg(id) {
  const img = generatedImages.find(i => i.id === id);
  if (!img || !img.dataURL) { showToast('Image not ready', '⚠'); return; }
  const a = document.createElement('a');
  a.href = img.dataURL;
  a.download = `pixelforge-${id}.${img.format.toLowerCase()}`;
  a.click();
  showToast('Image downloaded!', '↓');
}

function downloadAll() {
  if (!generatedImages.length) return;
  generatedImages.forEach(img => {
    if (img.dataURL) {
      const a = document.createElement('a');
      a.href = img.dataURL;
      a.download = `pixelforge-${img.id}.${img.format.toLowerCase()}`;
      a.click();
    }
  });
  showToast(`${generatedImages.length} images downloaded!`, '↓');
}

function copyPromptText(id) {
  const img = generatedImages.find(i => i.id === id);
  if (!img) return;
  navigator.clipboard.writeText(img.prompt).then(() => showToast('Prompt copied!', '📋'));
}

// CLEAR
function clearAll() {
  generatedImages = [];
  document.getElementById('galleryGrid').innerHTML = '';
  document.getElementById('galleryGrid').style.display = 'none';
  document.getElementById('emptyState').style.display = 'flex';
  updateCounts();
}

// COUNTS
function updateCounts() {
  const count = generatedImages.length;
  document.getElementById('countBadge').textContent = `${count} image${count !== 1 ? 's' : ''}`;
  document.getElementById('dlAllBtn').style.display = count > 0 ? 'flex' : 'none';
  document.getElementById('clearBtn').style.display = count > 0 ? 'flex' : 'none';
}

// HISTORY
function addHistory(prompt) {
  if (promptHistory.includes(prompt)) return;
  promptHistory.unshift(prompt);
  if (promptHistory.length > 5) promptHistory.pop();
  renderHistory();
}

function renderHistory() {
  const list = document.getElementById('historyList');
  if (!promptHistory.length) {
    list.innerHTML = '<div style="font-size:12px;color:var(--text3);">No recent prompts yet.</div>';
    return;
  }
  list.innerHTML = promptHistory.map(p =>
    `<div class="history-item" onclick="loadHistory(this)" title="${escapeHtml(p)}">${escapeHtml(p.substring(0, 60))}${p.length > 60 ? '…' : ''}</div>`
  ).join('');
}

function loadHistory(el) {
  document.getElementById('mainPrompt').value = el.title;
  showToast('Prompt loaded from history', '🕓');
}

// UTIL
function escapeHtml(str) {
  return String(str).replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;');
}
