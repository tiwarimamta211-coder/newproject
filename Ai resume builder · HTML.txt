function switchTab(tab, el) {
  document.querySelectorAll('.panel-tab').forEach(t => t.classList.remove('active'));
  el.classList.add('active');
  ['content', 'design', 'ai'].forEach(t => {
    const panel = document.getElementById('tab-' + t);
    if (panel) panel.classList.toggle('hidden', t !== tab);
  });
}

    div.innerHTML = `
      <div class="entry-item-header">
        <div>
          <div class="entry-item-title">${lang.name || 'Language'}</div>
          <div class="entry-item-sub">${lang.level}</div>
        </div>
        <div class="entry-actions">
          <button class="icon-btn del" onclick="removeLang(${idx})">✕</button>
        </div>
      </div>
      <div class="form-row">
        <div class="form-group">
          <label>Language</label>
          <input type="text" value="${lang.name}" placeholder="English" oninput="updateLang(${idx},'name',this.value)">

          <label>Proficiency</label>

          </select>
        </div>
      </div>
    `;
    list.appendChild(div);
  });
}

function updateLang(idx, field, val) {
  state.languages[idx][field] = val;
  updateResume();
}

function removeLang(idx) {
  state.languages.splice(idx, 1);
  renderLanguages();
  updateResume();
}

// ═══════════════════════════════════════════════
//  SKILLS TAGS
// ═══════════════════════════════════════════════
function addSkillTag(e) {
  if (e.key === 'Enter' || e.key === ',') {
    e.preventDefault();
    const val = e.target.value.trim().replace(/,$/, '');
    if (val && !state.skills.includes(val)) {
      state.skills.push(val);
      renderSkillTags();
      e.target.value = '';
      updateResume();
    }
  }
}

function renderSkillTags() {
  const wrap = document.getElementById('skills-wrap');
  const input = document.getElementById('skill-inp');
  wrap.innerHTML = '';
  state.skills.forEach((skill, i) => {
    const tag = document.createElement('div');
    tag.className = 'tag';
    tag.innerHTML = `${skill} <button class="tag-x" onclick="removeSkill(${i})">×</button>`;
    wrap.appendChild(tag);
  });
  wrap.appendChild(input);
}

function removeSkill(idx) {
  state.skills.splice(idx, 1);
  renderSkillTags();
  updateResume();
}

function updateResume() {

  page.className = 'resume-page tpl-' + state.template;
  page.style.setProperty('--accent-color', state.color);
  page.style.fontFamily = `'${state.font}', sans-serif`;

  const fn = g('fn'), ln = g('ln');
  const name = [fn, ln].filter(Boolean).join(' ') || 'Your Name';
  const initials = [(fn[0] || ''), (ln[0] || '')].join('').toUpperCase() || 'YN';

    email: g('email'), phone: g('phone'),

  };

    contact.email && `📧 ${contact.email}`,
    contact.phone && `📱 ${contact.phone}`,

}

// ═══════════════════════════════════════════════
//  COMPLETION + ATS
// ═══════════════════════════════════════════════
function updateCompletion() {
  const checks = [
    g('fn'), g('ln'), g('job-title'), g('email'), g('phone'),
    g('summary'), state.experience.length, state.education.length,
    state.skills.length
  ];
  const filled = checks.filter(Boolean).length;
  const pct = Math.round((filled / checks.length) * 100);
  document.getElementById('completion-pct').textContent = pct + '%';
  document.getElementById('completion-bar').style.width = pct + '%';
}

function updateATS() {
  const summaryLen = g('summary').length;
  const hasContact = !!(g('email') && g('phone'));
  const kw = Math.min(100, state.skills.length * 12 + (summaryLen > 100 ? 20 : 0));
  const fmt = hasContact ? 80 : 40 + (state.experience.length > 0 ? 20 : 0);
  const comp = Math.min(100, (state.experience.length * 15) + (state.education.length * 10) + (state.skills.length * 5) + (summaryLen > 50 ? 20 : 0));
  const total = Math.round((kw * 0.4) + (fmt * 0.3) + (comp * 0.3));

  document.getElementById('ats-num').textContent = total;
  document.getElementById('ats-kw').style.width = kw + '%';
  document.getElementById('ats-fmt').style.width = fmt + '%';
  document.getElementById('ats-comp').style.width = comp + '%';

  const arc = document.getElementById('ats-arc');
  const circumference = 163.36;
  arc.style.strokeDashoffset = circumference - (circumference * total / 100);
  arc.style.stroke = total > 70 ? '#4caf7d' : total > 40 ? '#c9a84c' : '#e05c5c';
}

// ═══════════════════════════════════════════════
//  AI FUNCTIONS (Claude API)
// ═══════════════════════════════════════════════
async function callClaude(prompt, systemPrompt = '') {
  try {
    const messages = [{ role: 'user', content: prompt }];
    const body = {
      model: 'claude-sonnet-4-20250514',
      max_tokens: 1000,
      messages
    };
    if (systemPrompt) body.system = systemPrompt;

    const res = await fetch('https://api.anthropic.com/v1/messages', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(body)
    });
    const data = await res.json();
    return data.content?.[0]?.text || '';
  } catch (e) {
    return null;
  }
}

async function aiEnhanceSummary() {
  const btn = event.target.closest('.ai-btn');
  btn.disabled = true;
  btn.textContent = '⏳ Enhancing...';
  const role = g('job-title');
  const existing = g('summary');
  const skills = state.skills.join(', ');

  const prompt = `Rewrite this professional resume summary to be compelling, ATS-optimized, and 3-4 sentences long. Role: ${role}. Current summary: "${existing || 'None provided'}". Skills: ${skills}. Write only the summary, no quotes, no preamble.`;
  const result = await callClaude(prompt, 'You are an expert resume writer. Write concise, powerful, ATS-friendly resume content.');

  if (result) {
    document.getElementById('summary').value = result;
    updateResume();

const chatHistory = [];

async function sendAIMsg(preset) {
  const inp = document.getElementById('ai-inp');
  const msg = preset || inp.value.trim();
  if (!msg) return;
  inp.value = '';

  appendMsg(msg, 'user');

  const typingEl = document.createElement('div');
  typingEl.className = 'ai-msg bot';
  typingEl.innerHTML = '<div class="ai-typing"><span></span><span></span><span></span></div>';
  document.getElementById('ai-messages').appendChild(typingEl);
  scrollChat();

  const resumeCtx = `Current resume data: Name: ${g('fn')} ${g('ln')}, Role: ${g('job-title')}, Skills: ${state.skills.join(', ')}, Experience: ${state.experience.map(e => e.role + ' at ' + e.company).join('; ')}, Education: ${state.education.map(e => e.degree + ' from ' + e.school).join('; ')}, Summary: ${g('summary')}`;

  chatHistory.push({ role: 'user', content: msg });

  let reply;
  try {
    const res = await fetch('https://api.anthropic.com/v1/messages', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        model: 'claude-sonnet-4-20250514',
        max_tokens: 1000,
        system: `You are an expert resume coach and career advisor. ${resumeCtx}. Give concise, actionable advice. Use bullet points when listing items. Keep responses under 200 words unless the user asks for something specific to generate.`,
        messages: chatHistory.slice(-10)
      })
    });
    const data = await res.json();
    reply = data.content?.[0]?.text;
  } catch (e) {}

  if (!reply) {
    const fallbacks = {
      'summary': 'Based on your profile, I recommend starting your summary with a strong value proposition highlighting your years of experience and top 2-3 skills. Then mention your key achievement, and end with what you bring to your next role.',
      'ats': `Your ATS score can be improved by:\n• Adding more industry-specific keywords\n• Quantifying your achievements (%, $, numbers)\n• Using standard section headings\n• Including action verbs at the start of each bullet`,
      'skill': 'For your role, consider adding: team leadership, problem-solving, project management, and any cloud certifications relevant to your field.',
      'default': 'I can help you improve your resume! Try asking me to write your summary, suggest skills, improve your bullet points, or optimize for a specific job description.'
    };
    const key = Object.keys(fallbacks).find(k => msg.toLowerCase().includes(k)) || 'default';
    reply = fallbacks[key];
  }

  chatHistory.push({ role: 'assistant', content: reply });
  typingEl.remove();
  appendMsg(reply, 'bot');
}

function appendMsg(text, type) {
  const div = document.createElement('div');
  div.className = `ai-msg ${type} fade-in`;
  div.innerHTML = text.replace(/\n/g, '<br>').replace(/\*\*(.*?)\*\*/g, '<strong>$1</strong>');
  document.getElementById('ai-messages').appendChild(div);
  scrollChat();
}

function scrollChat() {
  const msgs = document.getElementById('ai-messages');
  msgs.scrollTop = msgs.scrollHeight;
}

function autoResize(el) {
  el.style.height = 'auto';
  el.style.height = Math.min(el.scrollHeight, 100) + 'px';
}

async function downloadPDF() {
  showToast('⏳ Generating PDF…');
  const page = document.getElementById('resume-page');
  const origTransform = document.getElementById('resume-wrapper').style.transform;
  document.getElementById('resume-wrapper').style.transform = 'scale(1)';

  await new Promise(r => setTimeout(r, 200));

  try {
    const canvas = await html2canvas(page, { scale: 2, useCORS: true, logging: false });
    const imgData = canvas.toDataURL('image/jpeg', 0.95);
    const { jsPDF } = window.jspdf;
    const pdf = new jsPDF({ orientation: 'portrait', unit: 'mm', format: 'a4' });
    const w = 210, h = (canvas.height / canvas.width) * w;
    pdf.addImage(imgData, 'JPEG', 0, 0, w, h);
    const name = [g('fn'), g('ln')].filter(Boolean).join('_') || 'Resume';
    pdf.save(`${name}_Resume.pdf`);
    showToast('PDF downloaded! 📄', 'success');
  } catch (e) {
    showToast('PDF error. Try again.', 'error');
  }

  document.getElementById('resume-wrapper').style.transform = origTransform;
}

function loadFromStorage() {
  try {
    const saved = localStorage.getItem('resumeforge_data');
    if (!saved) return;
    const data = JSON.parse(saved);

    if (data.personal) {
      Object.entries({ fn: 'fn', ln: 'ln', jobTitle: 'job-title', email: 'email', phone: 'phone', city: 'city', country: 'country', linkedin: 'linkedin', portfolio: 'portfolio' }).forEach(([k, id]) => {
        const el = document.getElementById(id);
        if (el && data.personal[k]) el.value = data.personal[k];
      });
    }
    if (data.summary) document.getElementById('summary').value = data.summary;
    if (data.skills) { state.skills = data.skills; renderSkillTags(); }
    if (data.experience) { state.experience = data.experience; renderExperience(); }
    if (data.education) { state.education = data.education; renderEducation(); }
    if (data.projects) { state.projects = data.projects; renderProjects(); }
    if (data.certifications) { state.certifications = data.certifications; renderCertifications(); }
    if (data.languages) { state.languages = data.languages; renderLanguages(); }
    if (data.design) {
      if (data.design.template) state.template = data.design.template;
      if (data.design.color) state.color = data.design.color;
      if (data.design.font) state.font = data.design.font;
    }
  } catch (e) {}
}
