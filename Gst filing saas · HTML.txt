// ==================== TAX CALC ====================
function calcTax(taxable, rate, supply) {
  const t = parseFloat(taxable) || 0;
  const r = parseFloat(rate) / 100;
  const gst = t * r;
  if (supply === 'inter') return { cgst: 0, sgst: 0, igst: gst, total: t + gst };
  return { cgst: gst / 2, sgst: gst / 2, igst: 0, total: t + gst };
}
function calcSaleTax() {
  const t = document.getElementById('s-taxable').value;
  const r = document.getElementById('s-rate').value;
  const s = document.getElementById('s-supply').value;
  const c = calcTax(t, r, s);
  document.getElementById('s-cgst-disp').textContent = fmt(c.cgst);
  document.getElementById('s-sgst-disp').textContent = fmt(c.sgst);
  document.getElementById('s-igst-disp').textContent = fmt(c.igst);
  document.getElementById('s-total-disp').textContent = fmt(c.total);
}
function calcPurchaseTax() {
  const t = document.getElementById('p-taxable').value;
  const r = document.getElementById('p-rate').value;
  const s = document.getElementById('p-supply').value;
  const c = calcTax(t, r, s);
  document.getElementById('p-cgst-disp').textContent = fmt(c.cgst);
  document.getElementById('p-sgst-disp').textContent = fmt(c.sgst);
  document.getElementById('p-igst-disp').textContent = fmt(c.igst);
  document.getElementById('p-total-disp').textContent = fmt(c.total);
}

function updateDashboard() {
  const outputTax = state.sales.reduce((a, r) => a + r.cgst + r.sgst + r.igst, 0);
  const itc = state.purchases.filter(r => r.itc).reduce((a, r) => a + r.cgst + r.sgst + r.igst, 0);
  const net = Math.max(0, outputTax - itc);
  document.getElementById('dash-output').textContent = fmt(outputTax);
  document.getElementById('dash-itc').textContent = fmt(itc);
  document.getElementById('dash-net').textContent = fmt(net);
}

  document.getElementById('bal-total-liab').textContent = fmt(liabGST + liabOther);
}

function compute3B() {
  const outward = state.sales.reduce((a, r) => ({
    taxable: a.taxable + r.taxable,
    cgst: a.cgst + r.cgst, sgst: a.sgst + r.sgst, igst: a.igst + r.igst
  }), { taxable:0, cgst:0, sgst:0, igst:0 });
  const itcPurch = state.purchases.filter(r => r.itc).reduce((a, r) => ({
    cgst: a.cgst + r.cgst, sgst: a.sgst + r.sgst, igst: a.igst + r.igst

  document.getElementById('t4-total').textContent = fmt(itcTotal);

  document.getElementById('t5-sgst').textContent = fmt(netSGST);

function exportSalesCSV() {
  if (!state.sales.length) { toast('No sales data to export', 'error'); return; }
  const header = ['Invoice #','Date','Customer','GSTIN','Taxable','CGST','SGST','IGST','Total','Type'];
  const rows = state.sales.map(r => [r.inv, r.date, r.customer, r.gstin, r.taxable, r.cgst, r.sgst, r.igst, r.total, r.type]);
  downloadCSV([header, ...rows], 'Sales_Ledger.csv');
  toast('Sales CSV exported ✓', 'success');
}

function exportPurchaseCSV() {
  if (!state.purchases.length) { toast('No purchase data to export', 'error'); return; }
  const header = ['Bill #','Date','Supplier','GSTIN','Taxable','CGST','SGST','IGST','Total','ITC Eligible'];
  const rows = state.purchases.map(r => [r.bill, r.date, r.supplier, r.gstin, r.taxable, r.cgst, r.sgst, r.igst, r.total, r.itc?'Yes':'No']);
  downloadCSV([header, ...rows], 'Purchase_Ledger.csv');
  toast('Purchase CSV exported ✓', 'success');
}

function downloadCSV(rows, filename) {
  const csv = rows.map(r => r.map(v => `"${v}"`).join(',')).join('\n');
  const a = document.createElement('a');
  a.href = 'data:text/csv;charset=utf-8,' + encodeURIComponent(csv);
  a.download = filename; a.click();
}
