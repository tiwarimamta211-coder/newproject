import { useState, useRef, useEffect } from "react";

const SECTIONS = [
  { id: "hook", label: "🎯 Hook", desc: "Attention-grabbing intro" },
  { id: "intro", label: "📣 Introduction", desc: "Channel & topic intro" },
  { id: "main1", label: "📌 Part 1", desc: "First main segment" },
  { id: "main2", label: "📌 Part 2", desc: "Second main segment" },
  { id: "main3", label: "📌 Part 3", desc: "Third main segment" },
  { id: "casestudy", label: "💡 Case Study", desc: "Real example or story" },
  { id: "tips", label: "✅ Tips & Tricks", desc: "Actionable takeaways" },
  { id: "outro", label: "🎬 Outro + CTA", desc: "Subscribe & end screen" },
];

const TONES = ["Educational", "Entertaining", "Motivational", "Conversational", "Professional", "Storytelling"];
const DURATIONS = ["10 min", "15 min", "20 min", "30 min", "45 min", "60 min"];
const NICHES = ["Tech & AI", "Finance", "Health & Fitness", "Business", "Personal Development", "Cooking", "Travel", "Education", "Gaming", "Lifestyle"];

export default function YouTubeVideoGenerator() {
  const [topic, setTopic] = useState("");
  const [tone, setTone] = useState("Educational");
  const [duration, setDuration] = useState("20 min");
  const [niche, setNiche] = useState("Tech & AI");
  const [targetAudience, setTargetAudience] = useState("");
  const [keywords, setKeywords] = useState("");
  const [selectedSections, setSelectedSections] = useState(SECTIONS.map(s => s.id));
  const [generating, setGenerating] = useState(false);
  const [streamText, setStreamText] = useState("");
  const [generated, setGenerated] = useState(false);
  const [activeTab, setActiveTab] = useState("script");
  const [progress, setProgress] = useState(0);
  const [currentSection, setCurrentSection] = useState("");
  const [copied, setCopied] = useState(false);
  const outputRef = useRef(null);
  const progressRef = useRef(null);

  useEffect(() => {
    if (outputRef.current && streamText) {
      outputRef.current.scrollTop = outputRef.current.scrollHeight;
    }
  }, [streamText]);

  const toggleSection = (id) => {
    setSelectedSections(prev =>
      prev.includes(id) ? prev.filter(s => s !== id) : [...prev, id]
    );
  };

  const buildPrompt = () => {
    const sectionNames = SECTIONS.filter(s => selectedSections.includes(s.id)).map(s => s.label).join(", ");
    return `You are an expert YouTube scriptwriter. Write a COMPLETE, DETAILED, LONG-FORM YouTube video script for the following:

**Topic:** ${topic}
**Niche:** ${niche}
**Tone:** ${tone}
**Target Duration:** ${duration}
**Target Audience:** ${targetAudience || "General audience"}
**SEO Keywords:** ${keywords || "auto-select relevant keywords"}
**Sections to include:** ${sectionNames}

Write a FULL, READY-TO-READ script with:
- Exact words the YouTuber should say (not bullet points — full sentences)
- Natural transitions between sections
- Emotional hooks and storytelling elements
- B-roll suggestions in [brackets]
- Timestamps for each section
- End-screen and subscribe call-to-action

Also provide:
1. **Video Title** (3 options, SEO-optimized)
2. **Description** (full YouTube description with timestamps, links placeholder, hashtags)
3. **Thumbnail Ideas** (3 concepts)
4. **Tags** (20 SEO tags)

Start the script now. Make it ${duration} long with rich, detailed, engaging content.`;
  };

  const handleGenerate = async () => {
    if (!topic.trim()) return;
    setGenerating(true);
    setStreamText("");
    setGenerated(false);
    setProgress(0);
    setCurrentSection("Initializing...");

    const sectionLabels = SECTIONS.filter(s => selectedSections.includes(s.id)).map(s => s.label);
    let progStep = 0;
    const totalSteps = sectionLabels.length + 2;

    const progressInterval = setInterval(() => {
      progStep++;
      const pct = Math.min(Math.round((progStep / totalSteps) * 95), 95);
      setProgress(pct);
      if (progStep <= sectionLabels.length) {
        setCurrentSection(`Writing ${sectionLabels[progStep - 1] || ""}...`);
      } else {
        setCurrentSection("Finalizing script...");
      }
    }, 2800);

    try {
      const response = await fetch("https://api.anthropic.com/v1/messages", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({
          model: "claude-sonnet-4-20250514",
          max_tokens: 8000,
          stream: true,
          messages: [{ role: "user", content: buildPrompt() }],
        }),
      });

      const reader = response.body.getReader();
      const decoder = new TextDecoder();
      let fullText = "";

      while (true) {
        const { done, value } = await reader.read();
        if (done) break;
        const chunk = decoder.decode(value);
        const lines = chunk.split("\n");
        for (const line of lines) {
          if (line.startsWith("data: ")) {
            try {
              const data = JSON.parse(line.slice(6));
              if (data.type === "content_block_delta" && data.delta?.text) {
                fullText += data.delta.text;
                setStreamText(fullText);
              }
            } catch {}
          }
        }
      }

      clearInterval(progressInterval);
      setProgress(100);
      setCurrentSection("Done!");
      setGenerated(true);
    } catch (err) {
      clearInterval(progressInterval);
      setStreamText("❌ Error: " + err.message);
    } finally {
      setGenerating(false);
    }
  };

  const handleCopy = () => {
    navigator.clipboard.writeText(streamText);
    setCopied(true);
    setTimeout(() => setCopied(false), 2000);
  };

  const wordCount = streamText.split(/\s+/).filter(Boolean).length;
  const estimatedTime = Math.round(wordCount / 130);

  return (
    <div style={{
      minHeight: "100vh",
      background: "linear-gradient(135deg, #0a0a0f 0%, #12121e 50%, #0d0d1a 100%)",
      fontFamily: "'Sora', 'DM Sans', sans-serif",
      color: "#e8e8f0",
      padding: "0",
    }}>
      <style>{`
        @import url('https://fonts.googleapis.com/css2?family=Sora:wght@300;400;600;700;800&family=DM+Mono:wght@400;500&display=swap');
        * { box-sizing: border-box; }
        ::-webkit-scrollbar { width: 6px; }
        ::-webkit-scrollbar-track { background: #12121e; }
        ::-webkit-scrollbar-thumb { background: #ff4545; border-radius: 3px; }
        .card { background: rgba(255,255,255,0.04); border: 1px solid rgba(255,255,255,0.08); border-radius: 16px; }
        .badge { background: rgba(255,69,69,0.15); border: 1px solid rgba(255,69,69,0.3); color: #ff7070; padding: 3px 10px; border-radius: 20px; font-size: 11px; font-weight: 600; letter-spacing: 0.5px; }
        .section-pill { cursor: pointer; padding: 8px 14px; border-radius: 20px; font-size: 12px; font-weight: 600; transition: all 0.2s; border: 1px solid; user-select: none; }
        .section-pill.active { background: rgba(255,69,69,0.2); border-color: #ff4545; color: #ff7070; }
        .section-pill.inactive { background: rgba(255,255,255,0.04); border-color: rgba(255,255,255,0.1); color: #666; }
        .section-pill:hover { transform: translateY(-1px); }
        select, input, textarea { background: rgba(255,255,255,0.05) !important; border: 1px solid rgba(255,255,255,0.1) !important; border-radius: 10px; color: #e8e8f0 !important; padding: 10px 14px; font-family: 'Sora', sans-serif; font-size: 13px; outline: none; width: 100%; transition: border-color 0.2s; }
        select:focus, input:focus, textarea:focus { border-color: rgba(255,69,69,0.5) !important; }
        select option { background: #1a1a2e; color: #e8e8f0; }
        .gen-btn { background: linear-gradient(135deg, #ff4545, #ff6b35); border: none; color: white; padding: 14px 36px; border-radius: 12px; font-size: 15px; font-weight: 700; cursor: pointer; transition: all 0.2s; letter-spacing: 0.3px; font-family: 'Sora', sans-serif; }
        .gen-btn:hover:not(:disabled) { transform: translateY(-2px); box-shadow: 0 8px 30px rgba(255,69,69,0.4); }
        .gen-btn:disabled { opacity: 0.6; cursor: not-allowed; transform: none; }
        .tab { padding: 8px 20px; border-radius: 8px; cursor: pointer; font-size: 13px; font-weight: 600; transition: all 0.2s; }
        .tab.active { background: rgba(255,69,69,0.2); color: #ff7070; }
        .tab.inactive { color: #555; }
        .tab:hover { color: #aaa; }
        .progress-bar { height: 4px; background: rgba(255,255,255,0.08); border-radius: 2px; overflow: hidden; }
        .progress-fill { height: 100%; background: linear-gradient(90deg, #ff4545, #ff6b35); border-radius: 2px; transition: width 0.5s ease; }
        label { display: block; font-size: 11px; font-weight: 600; color: #888; letter-spacing: 0.8px; text-transform: uppercase; margin-bottom: 6px; }
        .script-output { white-space: pre-wrap; font-family: 'DM Mono', monospace; font-size: 13px; line-height: 1.8; color: #d0d0e0; }
        @keyframes pulse { 0%,100%{opacity:1} 50%{opacity:0.5} }
        .pulse { animation: pulse 1.5s ease infinite; }
        @keyframes float { 0%,100%{transform:translateY(0)} 50%{transform:translateY(-6px)} }
      `}</style>

      {/* Header */}
      <div style={{ borderBottom: "1px solid rgba(255,255,255,0.06)", padding: "20px 32px", display: "flex", alignItems: "center", gap: "16px", background: "rgba(0,0,0,0.3)", backdropFilter: "blur(10px)", position: "sticky", top: 0, zIndex: 100 }}>
        <div style={{ width: 40, height: 40, background: "linear-gradient(135deg, #ff4545, #ff6b35)", borderRadius: 12, display: "flex", alignItems: "center", justifyContent: "center", fontSize: 20, animation: "float 3s ease infinite" }}>▶</div>
        <div>
          <div style={{ fontSize: 18, fontWeight: 800, letterSpacing: "-0.5px" }}>YouTube Script Generator</div>
          <div style={{ fontSize: 11, color: "#666", letterSpacing: "0.5px" }}>AI-POWERED LONG-FORM VIDEO CREATOR</div>
        </div>
        <div style={{ marginLeft: "auto", display: "flex", gap: 8 }}>
          <span className="badge">GPT CLAUDE</span>
          <span className="badge">LONG-FORM</span>
        </div>
      </div>

      <div style={{ display: "grid", gridTemplateColumns: "380px 1fr", gap: 0, minHeight: "calc(100vh - 81px)" }}>
        {/* LEFT PANEL */}
        <div style={{ borderRight: "1px solid rgba(255,255,255,0.06)", padding: "24px", overflowY: "auto", maxHeight: "calc(100vh - 81px)" }}>
          <div style={{ display: "flex", flexDirection: "column", gap: 20 }}>

            {/* Topic */}
            <div>
              <label>Video Topic *</label>
              <textarea
                rows={3}
                placeholder="e.g. How to build a SaaS in 30 days with no coding experience..."
                value={topic}
                onChange={e => setTopic(e.target.value)}
                style={{ resize: "vertical" }}
              />
            </div>

            <div style={{ display: "grid", gridTemplateColumns: "1fr 1fr", gap: 12 }}>
              <div>
                <label>Niche</label>
                <select value={niche} onChange={e => setNiche(e.target.value)}>
                  {NICHES.map(n => <option key={n}>{n}</option>)}
                </select>
              </div>
              <div>
                <label>Duration</label>
                <select value={duration} onChange={e => setDuration(e.target.value)}>
                  {DURATIONS.map(d => <option key={d}>{d}</option>)}
                </select>
              </div>
            </div>

            <div>
              <label>Tone & Style</label>
              <div style={{ display: "flex", flexWrap: "wrap", gap: 8, marginTop: 4 }}>
                {TONES.map(t => (
                  <span key={t} className={`section-pill ${tone === t ? "active" : "inactive"}`} onClick={() => setTone(t)}>{t}</span>
                ))}
              </div>
            </div>

            <div>
              <label>Target Audience</label>
              <input placeholder="e.g. Beginner entrepreneurs aged 25-35" value={targetAudience} onChange={e => setTargetAudience(e.target.value)} />
            </div>

            <div>
              <label>SEO Keywords</label>
              <input placeholder="e.g. passive income, online business, 2025" value={keywords} onChange={e => setKeywords(e.target.value)} />
            </div>

            {/* Sections */}
            <div>
              <label>Video Sections</label>
              <div style={{ display: "flex", flexWrap: "wrap", gap: 8, marginTop: 4 }}>
                {SECTIONS.map(s => (
                  <span
                    key={s.id}
                    className={`section-pill ${selectedSections.includes(s.id) ? "active" : "inactive"}`}
                    onClick={() => toggleSection(s.id)}
                    title={s.desc}
                  >{s.label}</span>
                ))}
              </div>
              <div style={{ fontSize: 11, color: "#555", marginTop: 6 }}>{selectedSections.length} sections selected</div>
            </div>

            {/* Tip card */}
            <div className="card" style={{ padding: "14px 16px" }}>
              <div style={{ fontSize: 11, color: "#ff7070", fontWeight: 700, marginBottom: 6, letterSpacing: "0.5px" }}>💡 PRO TIPS</div>
              <div style={{ fontSize: 12, color: "#777", lineHeight: 1.7 }}>
                • Be specific with your topic for better scripts<br />
                • Add target audience for personalized language<br />
                • Include keywords for SEO-optimized titles<br />
                • 20-30 min scripts ≈ 2500–4000 words
              </div>
            </div>

            <button className="gen-btn" onClick={handleGenerate} disabled={generating || !topic.trim()} style={{ width: "100%" }}>
              {generating ? "⚡ Generating Script..." : "🎬 Generate Full Script"}
            </button>

            {generating && (
              <div>
                <div style={{ display: "flex", justifyContent: "space-between", marginBottom: 8, fontSize: 12 }}>
                  <span style={{ color: "#ff7070" }} className="pulse">{currentSection}</span>
                  <span style={{ color: "#555" }}>{progress}%</span>
                </div>
                <div className="progress-bar">
                  <div className="progress-fill" style={{ width: `${progress}%` }} />
                </div>
              </div>
            )}

            {generated && (
              <div className="card" style={{ padding: "12px 16px", borderColor: "rgba(100,255,100,0.2)", background: "rgba(100,255,100,0.05)" }}>
                <div style={{ fontSize: 12, color: "#7fff7f", fontWeight: 600 }}>✅ Script Generated!</div>
                <div style={{ fontSize: 11, color: "#666", marginTop: 4 }}>
                  ~{wordCount.toLocaleString()} words · ~{estimatedTime} min read aloud
                </div>
              </div>
            )}
          </div>
        </div>

        {/* RIGHT PANEL */}
        <div style={{ display: "flex", flexDirection: "column", maxHeight: "calc(100vh - 81px)" }}>
          {/* Tabs + actions */}
          <div style={{ padding: "16px 24px", borderBottom: "1px solid rgba(255,255,255,0.06)", display: "flex", alignItems: "center", gap: 8 }}>
            <div style={{ display: "flex", gap: 4, background: "rgba(255,255,255,0.03)", padding: 4, borderRadius: 10 }}>
              {["script", "structure"].map(tab => (
                <div key={tab} className={`tab ${activeTab === tab ? "active" : "inactive"}`} onClick={() => setActiveTab(tab)}>
                  {tab === "script" ? "📝 Script" : "🗂️ Structure"}
                </div>
              ))}
            </div>
            <div style={{ marginLeft: "auto", display: "flex", gap: 8 }}>
              {generated && (
                <>
                  <button onClick={handleCopy} style={{ background: "rgba(255,255,255,0.07)", border: "1px solid rgba(255,255,255,0.1)", color: copied ? "#7fff7f" : "#ccc", padding: "7px 16px", borderRadius: 8, cursor: "pointer", fontSize: 12, fontWeight: 600, fontFamily: "Sora, sans-serif" }}>
                    {copied ? "✅ Copied!" : "📋 Copy"}
                  </button>
                  <button onClick={() => { const blob = new Blob([streamText], { type: "text/plain" }); const a = document.createElement("a"); a.href = URL.createObjectURL(blob); a.download = `${topic.slice(0, 30)}_script.txt`; a.click(); }} style={{ background: "rgba(255,69,69,0.15)", border: "1px solid rgba(255,69,69,0.3)", color: "#ff7070", padding: "7px 16px", borderRadius: 8, cursor: "pointer", fontSize: 12, fontWeight: 600, fontFamily: "Sora, sans-serif" }}>
                    ⬇️ Download
                  </button>
                </>
              )}
            </div>
          </div>

          {/* Output area */}
          <div ref={outputRef} style={{ flex: 1, overflowY: "auto", padding: "24px" }}>
            {activeTab === "script" && (
              streamText ? (
                <div className="script-output">{streamText}{generating && <span className="pulse" style={{ color: "#ff4545" }}>▌</span>}</div>
              ) : (
                <div style={{ display: "flex", flexDirection: "column", alignItems: "center", justifyContent: "center", height: "100%", gap: 16, opacity: 0.4 }}>
                  <div style={{ fontSize: 64, animation: "float 3s ease infinite" }}>🎬</div>
                  <div style={{ fontSize: 16, fontWeight: 600, color: "#888" }}>Your script will appear here</div>
                  <div style={{ fontSize: 13, color: "#555", textAlign: "center", maxWidth: 300 }}>Fill in the details on the left and click Generate to create your full YouTube video script</div>
                </div>
              )
            )}
            {activeTab === "structure" && (
              <div style={{ display: "flex", flexDirection: "column", gap: 12 }}>
                <div style={{ fontSize: 13, color: "#777", marginBottom: 4 }}>Video structure for <strong style={{ color: "#ff7070" }}>{duration}</strong> duration</div>
                {SECTIONS.filter(s => selectedSections.includes(s.id)).map((s, i) => {
                  const mins = {
                    hook: "0:00–0:45", intro: "0:45–2:00", main1: "2:00–8:00", main2: "8:00–14:00",
                    main3: "14:00–20:00", casestudy: "20:00–25:00", tips: "25:00–28:00", outro: "28:00–30:00"
                  };
                  return (
                    <div key={s.id} className="card" style={{ padding: "14px 18px", display: "flex", alignItems: "center", gap: 14 }}>
                      <div style={{ width: 32, height: 32, background: "rgba(255,69,69,0.15)", borderRadius: 8, display: "flex", alignItems: "center", justifyContent: "center", fontSize: 16 }}>{s.label.split(" ")[0]}</div>
                      <div style={{ flex: 1 }}>
                        <div style={{ fontSize: 13, fontWeight: 700 }}>{s.label.split(" ").slice(1).join(" ")}</div>
                        <div style={{ fontSize: 11, color: "#666", marginTop: 2 }}>{s.desc}</div>
                      </div>
                      <div style={{ fontSize: 11, color: "#ff7070", fontFamily: "DM Mono, monospace", background: "rgba(255,69,69,0.08)", padding: "3px 8px", borderRadius: 6 }}>{mins[s.id] || "—"}</div>
                    </div>
                  );
                })}
                <div className="card" style={{ padding: "14px 18px", borderColor: "rgba(255,200,69,0.2)", background: "rgba(255,200,69,0.05)", marginTop: 8 }}>
                  <div style={{ fontSize: 12, color: "#ffc845", fontWeight: 600, marginBottom: 6 }}>📊 Estimated Stats</div>
                  <div style={{ display: "grid", gridTemplateColumns: "1fr 1fr 1fr", gap: 12 }}>
                    {[
                      ["Duration", duration],
                      ["Sections", selectedSections.length],
                      ["Est. Words", duration === "10 min" ? "~1300" : duration === "20 min" ? "~2600" : duration === "30 min" ? "~3900" : duration === "60 min" ? "~7800" : "~2000"],
                    ].map(([k, v]) => (
                      <div key={k} style={{ textAlign: "center" }}>
                        <div style={{ fontSize: 18, fontWeight: 800, color: "#ffc845" }}>{v}</div>
                        <div style={{ fontSize: 10, color: "#777", marginTop: 2 }}>{k}</div>
                      </div>
                    ))}
                  </div>
                </div>
              </div>
            )}
          </div>
        </div>
      </div>
    </div>
  );
}
