function handlePhoto(input) {
  const file = input.files[0];
  if (!file) return;
  const reader = new FileReader();
  reader.onload = e => {
    state.photo = e.target.result;
    const area = document.getElementById('photoArea');
    area.innerHTML = `<img src="${state.photo}" alt="Photo">`;
    render();
  };
  reader.readAsDataURL(file);
}

      <div class="r-section-title">Testimonial</div>
      <div style="font-size:.76rem;color:#555;line-height:1.55;font-style:italic">${testimonials}</div>
    </div>` : ''}
  ` : '';

  doc.innerHTML = `
    <div class="resume-header">
      <div>
        <div class="resume-name">${name}</div>
        ${tagline ? `<div class="resume-tagline">${tagline}</div>` : ''}
        ${contactItems ? `<div class="resume-contact">${contactItems}</div>` : ''}
      </div>
      ${photoHTML}
    </div>
    <div class="resume-body">
      <div class="resume-main">
        ${checked('showSummary') && summary ? `
          <div class="r-section">
            <div class="r-section-title">About</div>
            <div class="r-summary">${summary}</div>
          </div>
        ` : ''}
        ${servicesHTML}
        ${experienceHTML}
        ${projectsHTML}
      </div>
      <div class="resume-sidebar">
        ${skillsHTML}
        ${educationHTML}
        ${ratesHTML}
      </div>
    </div>
  `;
}

// ── PROGRESS ──
function updateProgress() {
  const fields = ['name','tagline','email','phone','location','summary','hourlyRate'];
  const filled = fields.filter(f => val(f)).length;
  const pct = Math.round((filled / fields.length) * 100);
  document.getElementById('progressBar').style.width = pct + '%';
}

// ── SAVE / LOAD ──
function saveData() {
  const data = {
    fields: {},
    state: { photo: state.photo, theme: state.theme, font: state.font,
             services: state.services, experience: state.experience,
             projects: state.projects, skills: state.skills, education: state.education },
    checks: ['showSummary','showServices','showExperience','showProjects','showSkills','showEducation','showRates'].reduce((a, id) => {
      a[id] = checked(id); return a;
    }, {})
  };
  ['name','tagline','email','phone','location','website','linkedin','github','summary','yearsExp','availability',
   'hourlyRate','projectRate','retainerRate','platforms','languages','testimonials'].forEach(id => {
    data.fields[id] = val(id);
  });
  localStorage.setItem('freelancecv', JSON.stringify(data));
  toast('✅ Saved to browser');
}
function loadData() {
  const raw = localStorage.getItem('freelancecv');
  if (!raw) { toast('⚠️ No saved data found'); return; }
  const data = JSON.parse(raw);
  Object.entries(data.fields).forEach(([id, v]) => {
    const el = document.getElementById(id);
    if (el) el.value = v;
  });
  if (data.state) {
    state = { ...state, ...data.state };
    document.querySelectorAll('.swatch').forEach(s => {
      s.classList.toggle('active', s.dataset.theme === state.theme);
    });
    document.querySelectorAll('.font-chip').forEach(f => {
      f.classList.toggle('active', f.dataset.font === state.font);
    });
  }
  if (data.checks) {
    Object.entries(data.checks).forEach(([id, v]) => {
      const el = document.getElementById(id);
      if (el) el.checked = v;
    });
  }
  renderServicesList(); renderExperienceList(); renderProjectsList();
  renderSkillsList(); renderEducationList();
  render();
  toast('📂 Data loaded');
}

function resetAll() {
  if (!confirm('Reset all data?')) return;
  localStorage.removeItem('freelancecv');
  location.reload();
}

// ── TOAST ──
function toast(msg) {
  const t = document.createElement('div');
  t.className = 'toast';
  t.textContent = msg;
  document.getElementById('toastContainer').appendChild(t);
  setTimeout(() => t.remove(), 3000);
}
