/* ===== GENERATE ===== */
async function generateImage() {
  const prompt = document.getElementById('prompt').value.trim();
  if (!prompt) { showToast('Please enter a prompt first!', 'error'); return; }

  const cost = modelCost * imageCount;
  if (credits < cost) {
    document.getElementById('modal-need').textContent = cost;
    document.getElementById('modal-have').textContent = credits;
    document.getElementById('credit-modal').classList.add('active');
    return;
  }

  // start UI
  setLoadingState(true);
  const msgSet = LOAD_MSGS[Math.floor(Math.random() * LOAD_MSGS.length)];
  document.getElementById('load-main').textContent = msgSet[0];
  document.getElementById('load-sub').textContent = msgSet[1];

  let prog = 0;
  const bar = document.getElementById('prog-bar');
  bar.style.width = '0%';
  clearInterval(progInterval);
  progInterval = setInterval(() => {
    prog = Math.min(prog + Math.random() * 7, 91);
    bar.style.width = prog + '%';
  }, 220);

  // Simulate generation (1.8–3.2s)
  await sleep(1900 + Math.random() * 1300);
  clearInterval(progInterval);
  bar.style.width = '100%';
  await sleep(280);

  // Deduct credits
  credits -= cost;
  totalUsed += cost;
  genCount++;

  const seed = parseInt(document.getElementById('seed-range').value);
  const actualSeed = seed === 0 ? Math.floor(Math.random() * 9999) : seed;

  // Use a reliable placeholder image
  const w = selectedRatio === '16:9' ? 800 : selectedRatio === '9:16' ? 450 : 800;
  const h = selectedRatio === '9:16' ? 800 : selectedRatio === '16:9' ? 450 : 800;
  const imgSrc = `https://picsum.photos/seed/${actualSeed % 200}/${w}/${h}`;

  lastSrc = imgSrc;
  lastPrompt = prompt;

  showResult(imgSrc, prompt, actualSeed);
  addToLog(imgSrc, prompt, cost);
  addToGallery(imgSrc);
  updateCredits();
  setLoadingState(false, true);
  showToast('✦ Image generated!', 'success');
}

function setLoadingState(loading, success = false) {
  const ph = document.getElementById('placeholder');
  const lc = document.getElementById('loading-container');
  const rw = document.getElementById('result-wrapper');
  const btn = document.getElementById('gen-btn');
  const label = document.getElementById('output-label');

  if (loading) {
    ph.style.display = 'none';
    rw.classList.remove('active');
    lc.classList.add('active');
    btn.disabled = true;
    label.textContent = 'Generating…';
  } else {
    lc.classList.remove('active');
    btn.disabled = false;
    label.textContent = 'Studio';
  }
}

function showResult(src, prompt, seed) {

  const box = document.getElementById('result-img-box');

  img.style.opacity = 0;
  img.src = src;
  img.onload = () => { img.style.transition = 'opacity .5s ease'; img.style.opacity = 1; };

  // Aspect ratio
  const [w,h] = selectedRatio.split(':').map(Number);
  box.style.aspectRatio = w / h;

  rw.classList.add('active');

  // Meta
  const modelName = document.getElementById('model-select').options[document.getElementById('model-select').selectedIndex].text.split('—')[0].trim();
  document.getElementById('meta-model').textContent = modelName;
  document.getElementById('meta-style').textContent = selectedStyle;
  document.getElementById('meta-ratio').textContent = selectedRatio;
  document.getElementById('meta-quality').textContent = document.getElementById('quality-select').value.toUpperCase();

}

function addToLog(src, prompt, cost) {
  const log = document.getElementById('usage-log');
  const empty = log.querySelector('.log-empty');
  if (empty) empty.remove();

  const now = new Date();
  const time = now.toLocaleTimeString([], {hour:'2-digit',minute:'2-digit'});

  log.insertBefore(item, log.firstChild);
}

function renderGallery() {

}

  const pct = Math.max(0, Math.round((credits / 300) * 100));
  document.getElementById('cr-fill').style.width = pct + '%';
  if (credits < 15 && credits > 0) showToast('⚡ Low credits! Consider upgrading.', 'warning');
  if (credits === 0) showToast('⚡ Out of credits! Please upgrade.', 'error');
}

function downloadImage() {
  if (!lastSrc) return;
  const a = document.createElement('a');
  a.href = lastSrc;
  a.download = `nexusai_${Date.now()}.jpg`;
  a.target = '_blank';
  a.click();
  savedCount++;
  updateCredits();
  showToast('⬇ Image saved!', 'success');
}
