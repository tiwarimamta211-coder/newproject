// ── CANDLE GENERATION ─────────────────────────────────────────────────────
function generatePattern(count, startPrice) {
  const pattern = document.getElementById('chartPattern').value;
  let price = startPrice, data = [];
  for(let i=0;i<count;i++){
    let t=i/count, open=price, close, high, low, vol;
    let move, bias=0;
    switch(pattern){
      case 'bull_trend': bias=0.012; break;
      case 'bear_trend': bias=-0.012; break;
      case 'consolidation': bias=0; break;
      case 'breakout': bias = i>count*0.7?0.025:0; break;
      case 'reversal': bias = i<count*0.5?-0.008:0.015; break;
      case 'head_shoulders':
        if(i<count*0.2) bias=0.018;
        else if(i<count*0.4) bias=-0.012;
        else if(i<count*0.6) bias=0.02;
        else if(i<count*0.8) bias=-0.015;
        else bias=-0.008;
        break;
      default: bias=0;
    }
    move = (Math.random()-0.48+bias)*0.025;
    close = open*(1+move);
    let range=Math.abs(close-open)*0.5+open*0.004;
    high=Math.max(open,close)+Math.random()*range;
    low=Math.min(open,close)-Math.random()*range;
    vol=50+Math.random()*200+(bias>0?80:0);
    data.push({open,close,high,low,vol});
    price=close;
  }
  return data;
}

function regenerateCandles() {

  const start = parseFloat(document.getElementById('startPrice').value)||22000;
  candles = generatePattern(count, start);
  redrawAll();
}

function redrawAll() {
  const W = canvas.width, H = canvas.height;
  ctx.clearRect(0,0,W,H);

  ctx.fillStyle = document.getElementById('bgColor').value;
  ctx.fillRect(0,0,W,H);

  // Chart area
  const pad = {left:W*0.08, right:W*0.06, top:H*0.12, bottom:H*0.08};
  const showVol = document.getElementById('showVolume').checked;
  volH = showVol ? H*0.15 : 0;
  chartX = pad.left; chartY = pad.top;

}

function drawGrid(W,H) {

  ctx.lineWidth = 0.5;
  const cols=12, rows=8;
  for(let i=1;i<cols;i++){ctx.beginPath();ctx.moveTo(W/cols*i,0);ctx.lineTo(W/cols*i,H);ctx.stroke();}

}

function drawCandleChart(x,y,w,h) {
  const data=candles, n=data.length;
  const prices=data.flatMap(c=>[c.high,c.low]);
  const minP=Math.min(...prices), maxP=Math.max(...prices);
  const range=maxP-minP, pad=range*0.08;
  const pxMin=minP-pad, pxMax=maxP+pad, pxRange=pxMax-pxMin;
  const toY=p=>y+h-(p-pxMin)/pxRange*h;
  const cw=w/n, gap=Math.max(1,cw*0.15), bw=cw-gap*2;

  const bullC=document.getElementById('bullColor').value;
  const bearC=document.getElementById('bearColor').value;
  const lblC=document.getElementById('labelColor').value;

  // Price levels
  const levels=5;
  ctx.font=`${Math.max(10,canvas.width*0.013)}px 'JetBrains Mono'`;
  ctx.fillStyle=lblC; ctx.textAlign='right';
  for(let i=0;i<=levels;i++){
    const p=pxMin+pxRange*(i/levels);
    const py=toY(p);
    ctx.fillText(formatPrice(p), x-6, py+4);
    ctx.strokeStyle=document.getElementById('gridColor').value;
    ctx.lineWidth=0.5; ctx.setLineDash([3,3]);
    ctx.beginPath();ctx.moveTo(x,py);ctx.lineTo(x+w,py);ctx.stroke();
    ctx.setLineDash([]);
  }

  // Candles
  data.forEach((c,i)=>{
    const cx=x+i*cw+gap;
    const isBull=c.close>=c.open;
    const col=isBull?bullC:bearC;
    const oY=toY(c.open), clY=toY(c.close), hiY=toY(c.high), loY=toY(c.low);

    // Wick
    ctx.strokeStyle=col; ctx.lineWidth=Math.max(1,bw*0.12); ctx.setLineDash([]);
    ctx.beginPath(); ctx.moveTo(cx+bw/2,hiY); ctx.lineTo(cx+bw/2,loY); ctx.stroke();

    // Body
    const bodyTop=Math.min(oY,clY), bodyH=Math.abs(oY-clY)||1;
    if(isBull) {
      ctx.fillStyle=col; ctx.fillRect(cx,bodyTop,bw,bodyH);
    } else {
      ctx.strokeStyle=col; ctx.lineWidth=Math.max(1,bw*0.1);
      ctx.fillStyle=col+'33'; ctx.fillRect(cx,bodyTop,bw,bodyH);
      ctx.strokeRect(cx,bodyTop,bw,bodyH);
    }
  });

  // Chart info label
  const sym=document.getElementById('symbolName').value||'SYMBOL';
  const tf=document.getElementById('timeframe').value;
  const lastC=data[data.length-1];
  const chg=((lastC.close-data[0].open)/data[0].open*100);
  const isUp=chg>=0;

  ctx.textAlign='left';
  ctx.font=`700 ${canvas.width*0.022}px 'Syne'`;
  ctx.fillStyle=document.getElementById('labelColor').value;
  ctx.fillText(sym+' · '+tf, x, y-canvas.height*0.03);

  ctx.font=`700 ${canvas.width*0.018}px 'JetBrains Mono'`;
  ctx.fillStyle=isUp?bullC:bearC;
  const priceStr=formatPrice(lastC.close)+'  '+(isUp?'+':'')+chg.toFixed(2)+'%';
  ctx.fillText(priceStr, x+canvas.width*0.25, y-canvas.height*0.03);
}

function drawVolume(x,y,w,h) {
  const data=candles, n=data.length;
  const maxVol=Math.max(...data.map(c=>c.vol));
  const cw=w/n, gap=Math.max(1,cw*0.15), bw=cw-gap*2;
  const bullC=document.getElementById('bullColor').value;
  const bearC=document.getElementById('bearColor').value;
  data.forEach((c,i)=>{
    const vH=(c.vol/maxVol)*h;
    const isBull=c.close>=c.open;
    ctx.fillStyle=(isBull?bullC:bearC)+'88';
    ctx.fillRect(x+i*cw+gap, y+h-vH, bw, vH);
  });
  ctx.font=`${Math.max(9,canvas.width*0.011)}px 'Syne'`;
  ctx.fillStyle=document.getElementById('labelColor').value;
  ctx.textAlign='left';
  ctx.fillText('VOLUME', x, y+canvas.height*0.016);
}

function drawMALines(x,y,w,h) {
  const data=candles;
  const prices=data.flatMap(c=>[c.high,c.low]);
  const minP=Math.min(...prices), maxP=Math.max(...prices);
  const range=maxP-minP, pad=range*0.08;
  const pxMin=minP-pad, pxMax=maxP+pad, pxRange=pxMax-pxMin;
  const toY=p=>y+h-(p-pxMin)/pxRange*h;
  const cw=w/data.length;
  const closes=data.map(c=>c.close);

  const drawMA=(period,color,label)=>{
    if(data.length<period) return;
    ctx.strokeStyle=color; ctx.lineWidth=Math.max(1,canvas.width*0.002);
    ctx.setLineDash([]);
    ctx.beginPath();
    for(let i=period-1;i<closes.length;i++){
      const avg=closes.slice(i-period+1,i+1).reduce((a,b)=>a+b,0)/period;
      const px=x+i*cw+cw/2, py=toY(avg);
      i===period-1?ctx.moveTo(px,py):ctx.lineTo(px,py);
    }
    ctx.stroke();
    // Label
    const last=closes.slice(-period).reduce((a,b)=>a+b,0)/period;
    ctx.font=`600 ${Math.max(9,canvas.width*0.011)}px 'Syne'`;
    ctx.fillStyle=color; ctx.textAlign='left';
    ctx.fillText('MA'+period+' '+formatPrice(last), x+w+4, toY(last)+4);
  };
  drawMA(20, document.getElementById('ma20Color').value, 'MA20');
  drawMA(50, document.getElementById('ma50Color').value, 'MA50');
}

function drawLogo(W,H) {
  const pos = document.getElementById('logoPosition').value;
  const sz = parseInt(document.getElementById('logoSize').value);
  const op = parseInt(document.getElementById('logoOpacity').value)/100;
  const pad = 20;

}

function drawLayer(l) {
  ctx.save();
  switch(l.type){
    case 'text':
      ctx.font=`${l.style||'normal'} ${l.weight||'800'} ${l.size||28}px 'Syne'`;
      const metrics=ctx.measureText(l.text);
      const th=l.size||28;
      if(l.bg){
        const op=((l.bgOpacity||70)/100);
        ctx.globalAlpha=op;
        ctx.fillStyle=l.bgColor||'#000000';
        ctx.fillRect(l.x-8,l.y-th,metrics.width+16,th+8);
        ctx.globalAlpha=1;
      }
      ctx.fillStyle=l.color||'#ffffff';
      ctx.fillText(l.text,l.x,l.y);
      if(l===selectedLayer){
        ctx.strokeStyle='#00d4aa';ctx.lineWidth=1.5;ctx.setLineDash([4,3]);
        ctx.strokeRect(l.x-10,l.y-th-4,metrics.width+20,th+16);
        ctx.setLineDash([]);
      }
      break;
    case 'arrow':
      drawArrowShape(l);
      break;
    case 'line':
      ctx.strokeStyle=l.color||'#f5a623';
      ctx.lineWidth=l.lineW||2;
      ctx.setLineDash(l.dash||[]);
      ctx.beginPath();ctx.moveTo(l.x1,l.y1);ctx.lineTo(l.x2,l.y2);ctx.stroke();
      ctx.setLineDash([]);
      if(l===selectedLayer) drawHandles(l.x1,l.y1,l.x2,l.y2);
      break;
    case 'rect':
      ctx.strokeStyle=l.color||'#00d4aa';
      ctx.lineWidth=l.lineW||2;
      ctx.fillStyle=l.fill||(l.color+'22')||'#00d4aa22';
      ctx.fillRect(l.x,l.y,l.w,l.h);
      ctx.strokeRect(l.x,l.y,l.w,l.h);
      if(l===selectedLayer){ctx.strokeStyle='#00d4aa';ctx.setLineDash([4,3]);ctx.strokeRect(l.x-3,l.y-3,l.w+6,l.h+6);ctx.setLineDash([]);}
      break;
    case 'circle':
      ctx.strokeStyle=l.color||'#00d4aa';
      ctx.lineWidth=l.lineW||2;
      ctx.fillStyle=l.fill||(l.color+'22')||'#00d4aa22';
      ctx.beginPath();ctx.ellipse(l.x+l.rx,l.y+l.ry,l.rx,l.ry,0,0,Math.PI*2);
      ctx.fill();ctx.stroke();
      break;
    case 'label_badge':
      drawBadge(l);
      break;
    case 'fib':
      drawFibLevels(l);
      break;
    case 'sr':
      drawSRLines(l);
      break;
    case 'branding':
      drawBrandingStrip(l);
      break;
    case 'callout':
      drawCallout(l);
      break;
    case 'watermark':
      ctx.globalAlpha=0.12;
      ctx.font=`700 ${canvas.width*0.035}px 'Syne'`;
      ctx.fillStyle='#ffffff';
      ctx.textAlign='center';
      ctx.save();
      ctx.translate(canvas.width/2,canvas.height/2);
      ctx.rotate(-Math.PI/8);
      for(let r=-2;r<=2;r++)for(let c=-2;c<=2;c++)
        ctx.fillText(l.text||'CHARTCRAFT',c*canvas.width*0.4,r*canvas.height*0.25);
      ctx.restore();
      ctx.globalAlpha=1;
      break;
  }
  ctx.restore();
}

function drawArrowShape(l) {
  const isUp=l.dir==='up';
  const col=isUp?document.getElementById('bullColor').value:document.getElementById('bearColor').value;
  const s=l.size||40;
  ctx.fillStyle=col;
  ctx.strokeStyle=col;
  ctx.lineWidth=1;
  ctx.beginPath();
  if(isUp) {
    ctx.moveTo(l.x,l.y);
    ctx.lineTo(l.x+s,l.y+s*0.8);
    ctx.lineTo(l.x+s*0.35,l.y+s*0.8);
    ctx.lineTo(l.x+s*0.35,l.y+s*1.5);
    ctx.lineTo(l.x+s*0.65,l.y+s*1.5);
    ctx.lineTo(l.x+s*0.65,l.y+s*0.8);
    ctx.lineTo(l.x+s,l.y+s*0.8);
  } else {
    ctx.moveTo(l.x,l.y+s*1.5);
    ctx.lineTo(l.x+s,l.y+s*0.7);
    ctx.lineTo(l.x+s*0.65,l.y+s*0.7);
    ctx.lineTo(l.x+s*0.65,l.y);
    ctx.lineTo(l.x+s*0.35,l.y);
    ctx.lineTo(l.x+s*0.35,l.y+s*0.7);
    ctx.lineTo(l.x,l.y+s*0.7);
  }
  ctx.closePath();ctx.fill();
  if(l===selectedLayer){ctx.strokeStyle='#fff';ctx.lineWidth=2;ctx.stroke();}
}

function drawBadge(l) {
  const pad=canvas.width*0.012;
  ctx.font=`700 ${l.size||canvas.width*0.022}px 'Syne'`;
  const tw=ctx.measureText(l.text).width;
  const bh=(l.size||canvas.width*0.022)*1.6;
  const bw=tw+pad*3;
  const r=bh/2;
  ctx.fillStyle=l.color||'#26c981';
  ctx.beginPath();
  ctx.roundRect(l.x,l.y-bh*0.75,bw,bh,r);
  ctx.fill();
  ctx.fillStyle=l.textColor||'#000000';
  ctx.fillText(l.text,l.x+pad*1.5,l.y);
}

function drawFibLevels(l) {
  const levels=[0,0.236,0.382,0.5,0.618,0.786,1];
  const colors=['#ef4444','#f5a623','#f5a623','#ffffff','#26c981','#26c981','#ef4444'];
  const W=canvas.width, H=canvas.height;
  const high=l.high||H*0.2, low=l.low||H*0.8;
  levels.forEach((fib,i)=>{
    const y=low-(low-high)*fib;
    ctx.strokeStyle=colors[i]; ctx.lineWidth=1;
    ctx.setLineDash([5,4]);
    ctx.beginPath();ctx.moveTo(l.x||0,y);ctx.lineTo((l.x||0)+W*0.85,y);ctx.stroke();
    ctx.setLineDash([]);
    ctx.font=`500 ${Math.max(9,W*0.013)}px 'JetBrains Mono'`;
    ctx.fillStyle=colors[i];
    ctx.textAlign='right';
    ctx.fillText(`${(fib*100).toFixed(1)}%`, (l.x||0)+W*0.85-4, y-3);
  });
}

function drawCallout(l) {
  const fs=Math.max(10,canvas.width*0.018);
  ctx.font=`700 ${fs}px 'Syne'`;
  const tw=ctx.measureText(l.text).width;
  const bw=tw+fs*1.2, bh=fs*1.8;
  const r=6;
  ctx.fillStyle=l.color||'#f5a623';
  ctx.beginPath();ctx.roundRect(l.x,l.y,bw,bh,r);ctx.fill();
  // Tail
  ctx.beginPath();ctx.moveTo(l.x+bw*0.2,l.y+bh);ctx.lineTo(l.x+bw*0.3,l.y+bh+bh*0.4);ctx.lineTo(l.x+bw*0.4,l.y+bh);ctx.fill();
  ctx.fillStyle='#000000';
  ctx.textAlign='left';
  ctx.fillText(l.text,l.x+fs*0.6,l.y+bh*0.68);
}

function drawHandles(x1,y1,x2,y2) {
  [[x1,y1],[x2,y2]].forEach(([hx,hy])=>{
    ctx.fillStyle='#00d4aa';
    ctx.beginPath();ctx.arc(hx,hy,5,0,Math.PI*2);ctx.fill();
  });
}

function formatPrice(p) {
  if(p>=10000) return p.toLocaleString('en-IN',{maximumFractionDigits:1});
  if(p>=100) return p.toFixed(2);
  return p.toFixed(4);
}

canvas.addEventListener('mousemove', e=>{
  const r=canvas.getBoundingClientRect();
  const scaleX=canvas.width/r.width, scaleY=canvas.height/r.height;
  const mx=(e.clientX-r.left)*scaleX, my=(e.clientY-r.top)*scaleY;
  document.getElementById('statusCursor').textContent=`${Math.round(mx)}, ${Math.round(my)}`;
});

canvas.addEventListener('mousedown', e=>{
  const r=canvas.getBoundingClientRect();
  const scaleX=canvas.width/r.width, scaleY=canvas.height/r.height;
  const mx=(e.clientX-r.left)*scaleX, my=(e.clientY-r.top)*scaleY;
  drawStart={x:mx,y:my};

  if(activeTool==='select'){
    selectedLayer=null;
    for(let i=layers.length-1;i>=0;i--){
      if(hitTest(layers[i],mx,my)){selectedLayer=layers[i];break;}
    }
    updateSelectedControls();
    redrawAll();
    return;
  }
  if(activeTool==='text'){
    addTextAtPoint(mx,my); setTool('select'); return;
  }
  if(['line','rect','circle'].includes(activeTool)) isDrawing=true;
});

canvas.addEventListener('mouseup', e=>{
  if(!isDrawing) return;
  const r=canvas.getBoundingClientRect();
  const scaleX=canvas.width/r.width, scaleY=canvas.height/r.height;
  const mx=(e.clientX-r.left)*scaleX, my=(e.clientY-r.top)*scaleY;
  if(Math.abs(mx-drawStart.x)<3&&Math.abs(my-drawStart.y)<3){isDrawing=false;return;}

  saveUndo();
  if(activeTool==='line'){
    layers.push({type:'line',x1:drawStart.x,y1:drawStart.y,x2:mx,y2:my,color:'#f5a623',lineW:2,dash:[]});
  } else if(activeTool==='rect'){
    const rx=Math.min(drawStart.x,mx),ry=Math.min(drawStart.y,my);
    layers.push({type:'rect',x:rx,y:ry,w:Math.abs(mx-rx),h:Math.abs(my-ry),color:'#00d4aa',lineW:2});
  } else if(activeTool==='circle'){
    const rx=Math.abs(mx-drawStart.x)/2, ry=Math.abs(my-drawStart.y)/2;
    const ox=Math.min(drawStart.x,mx), oy=Math.min(drawStart.y,my);
    layers.push({type:'circle',x:ox,y:oy,rx,ry,color:'#00d4aa',lineW:2});
  }
  isDrawing=false;
  setTool('select');
  redrawAll();
});

canvas.addEventListener('mousemove', e=>{
  if(!isDrawing) return;
  const r=canvas.getBoundingClientRect();
  const scaleX=canvas.width/r.width, scaleY=canvas.height/r.height;
  const mx=(e.clientX-r.left)*scaleX, my=(e.clientY-r.top)*scaleY;
  redrawAll();
  ctx.strokeStyle='#00d4aa'; ctx.lineWidth=1.5; ctx.setLineDash([4,3]);
  if(activeTool==='line'){ctx.beginPath();ctx.moveTo(drawStart.x,drawStart.y);ctx.lineTo(mx,my);ctx.stroke();}
  else if(activeTool==='rect'){const x=Math.min(drawStart.x,mx),y=Math.min(drawStart.y,my);ctx.strokeRect(x,y,Math.abs(mx-x),Math.abs(my-y));}
  else if(activeTool==='circle'){ctx.beginPath();ctx.ellipse((drawStart.x+mx)/2,(drawStart.y+my)/2,Math.abs(mx-drawStart.x)/2,Math.abs(my-drawStart.y)/2,0,0,Math.PI*2);ctx.stroke();}
  ctx.setLineDash([]);
});

function hitTest(l,mx,my) {
  if(l.type==='text'){
    ctx.font=`${l.style||'normal'} ${l.weight||'800'} ${l.size||28}px 'Syne'`;
    const tw=ctx.measureText(l.text).width;
    const th=l.size||28;
    return mx>=l.x-10&&mx<=l.x+tw+10&&my>=l.y-th-4&&my<=l.y+12;
  }
  if(l.type==='rect') return mx>=l.x&&mx<=l.x+l.w&&my>=l.y&&my<=l.y+l.h;
  if(l.type==='arrow') return mx>=l.x&&mx<=l.x+(l.size||40)&&my>=l.y&&my<=l.y+(l.size||40)*1.5;
  if(l.type==='label_badge'||l.type==='callout') return mx>=l.x&&mx<=l.x+200&&my>=l.y-30&&my<=l.y+20;
  return false;
}

function updateSelectedControls() {
  const el=document.getElementById('selectedLayerControls');
  if(!selectedLayer){el.textContent='No element selected.';el.style.color='var(--text-muted)';return;}
  el.style.color='var(--text-secondary)';
  el.innerHTML=`<div style="display:flex;flex-direction:column;gap:6px">
    <div style="font-size:11px;color:var(--text-secondary)">Selected: <b style="color:var(--text-primary)">${selectedLayer.type}</b></div>
    <button class="btn btn-danger" style="font-size:11px" onclick="deleteSelected()">🗑 Delete Layer</button>
    <button class="btn" style="font-size:11px" onclick="moveUp()">↑ Move Up</button>
    <button class="btn" style="font-size:11px" onclick="moveDown()">↓ Move Down</button>
  </div>`;
}

function deleteSelected() {
  if(!selectedLayer) return;
  saveUndo();
  layers=layers.filter(l=>l!==selectedLayer);
  selectedLayer=null;
  updateSelectedControls();
  redrawAll();
}

function moveUp() {
  if(!selectedLayer) return;
  const i=layers.indexOf(selectedLayer);
  if(i<layers.length-1){[layers[i],layers[i+1]]=[layers[i+1],layers[i]];}
  redrawAll();
}
function moveDown() {
  if(!selectedLayer) return;
  const i=layers.indexOf(selectedLayer);
  if(i>0){[layers[i],layers[i-1]]=[layers[i-1],layers[i]];}
  redrawAll();
}

  reader.onload=ev=>{
