function recalcTax() {
  const gross = parseFloat(document.getElementById('taxableVal').value) || 0;
  const disc = parseFloat(document.getElementById('discountVal').value) || 0;
  const rate = parseInt(document.getElementById('gstRate').value);
  const pos = document.getElementById('posState').value;
  const txnType = document.getElementById('txnType').value;

  const taxable = Math.max(gross - disc, 0);
  const isIGST = (pos === 'diff') || (txnType === 'Export');
  const totalTax = taxable * rate / 100;
  const half = totalTax / 2;
  const total = taxable + totalTax;

  document.getElementById('tc-gross').textContent = fmt(gross);
  document.getElementById('tc-disc').textContent = '— ' + fmt(disc);
  document.getElementById('tc-taxable').textContent = fmt(taxable);
  document.getElementById('tc-total').textContent = fmt(total);

  if (isIGST) {
    document.getElementById('cgst-row').style.display = 'none';
    document.getElementById('sgst-row').style.display = 'none';
    document.getElementById('igst-row').style.display = '';
    document.getElementById('tc-igst').textContent = fmt(totalTax);
    document.getElementById('igst-rate-lbl').textContent = rate + '%';
  } else {
    document.getElementById('cgst-row').style.display = '';
    document.getElementById('sgst-row').style.display = '';
    document.getElementById('igst-row').style.display = 'none';
    document.getElementById('tc-cgst').textContent = fmt(half);
    document.getElementById('tc-sgst').textContent = fmt(half);
    document.getElementById('cgst-rate-lbl').textContent = (rate/2) + '%';
    document.getElementById('sgst-rate-lbl').textContent = (rate/2) + '%';
  }
}

function saveSale() {
  const invNo = document.getElementById('invNo').value.trim();
  const invDate = document.getElementById('invDate').value;
  const txnType = document.getElementById('txnType').value;
  const partyName = document.getElementById('partyName').value.trim();
  const buyerGstin = document.getElementById('buyerGstin').value.trim();
  const posState = document.getElementById('posState').value;
  const itemDesc = document.getElementById('itemDesc').value.trim();
  const hsnCode = document.getElementById('hsnCode').value.trim();
  const gstRate = parseInt(document.getElementById('gstRate').value);
  const gross = parseFloat(document.getElementById('taxableVal').value) || 0;
  const disc = parseFloat(document.getElementById('discountVal').value) || 0;
  const payMode = document.getElementById('payMode').value;

  if (!partyName) { toast('⚠ Enter party name'); return; }
  if (!invDate) { toast('⚠ Select invoice date'); return; }
  if (gross <= 0) { toast('⚠ Enter taxable value'); return; }

  const taxable = Math.max(gross - disc, 0);
  const isIGST = posState === 'diff' || txnType === 'Export';
  const totalTax = taxable * gstRate / 100;
  const cgst = isIGST ? 0 : totalTax / 2;
  const sgst = isIGST ? 0 : totalTax / 2;
  const igst = isIGST ? totalTax : 0;
  const total = taxable + totalTax;

  const sale = {
    id: Date.now(),
    invNo, invDate, txnType, partyName, buyerGstin,
    posState, itemDesc, hsnCode, gstRate,
    gross, disc, taxable, cgst, sgst, igst, totalTax, total,
    payMode
  };

  // Check for duplicate invoice number
  if (sales.find(s => s.invNo === invNo)) {
    toast('⚠ Invoice number already exists');
    return;
  }

  sales.push(sale);
  invCounter++;
  persist();
  updateInvNumber();
  clearForm();
  toast('✦ Invoice ' + invNo + ' saved successfully');
  renderDashboard();
}

function renderBarChart() {
  const months = ['Apr','May','Jun','Jul','Aug','Sep','Oct','Nov','Dec','Jan','Feb','Mar'];
  const monthData = new Array(12).fill(0);
  sales.forEach(s => {
    const d = new Date(s.invDate);
    let m = d.getMonth(); // 0=Jan
    // FY starts April (month index 3)
    let fyIdx = (m - 3 + 12) % 12;
    monthData[fyIdx] += s.taxable;
  });
  const maxVal = Math.max(...monthData, 1);
  const W = 460, H = 180, padL = 10, padR = 10, padT = 10, padB = 30;
  const chartW = W - padL - padR;
  const chartH = H - padT - padB;
  const bw = chartW / 12;
  const gap = 4;

  let rects = '', labels = '', ylines = '';
  // Y grid lines
  for (let i = 0; i <= 4; i++) {
    const y = padT + chartH - (i/4) * chartH;
    ylines += `<line x1="${padL}" y1="${y}" x2="${W-padR}" y2="${y}" stroke="#e2e0d8" stroke-width="1"/>`;
  }
  months.forEach((m, i) => {
    const val = monthData[i];
    const bh = val ? (val / maxVal) * chartH : 2;
    const x = padL + i * bw + gap;
    const y = padT + chartH - bh;
    const w = bw - gap * 2;
    const hue = i === new Date().getMonth() ? '#c8a45a' : '#1a1d27';
    rects += `<rect x="${x}" y="${y}" width="${w}" height="${bh}" fill="${hue}" rx="2" opacity="0.85">
      <title>${m}: ${fmt(val)}</title></rect>`;
    labels += `<text x="${x + w/2}" y="${H - 8}" text-anchor="middle" font-size="9" fill="#8a8fa8" font-family="'DM Mono',monospace">${m}</text>`;
  });
  document.getElementById('barChart').innerHTML = ylines + rects + labels;
}

function renderPieChart() {
  const igstTotal = sales.reduce((s,r) => s + r.igst, 0);
  const cgstTotal = sales.reduce((s,r) => s + r.cgst, 0);
  const sgstTotal = sales.reduce((s,r) => s + r.sgst, 0);
  const total = igstTotal + cgstTotal + sgstTotal || 1;

  const slices = [
    { label: 'CGST', val: cgstTotal, color: '#1a6b4a' },
    { label: 'SGST', val: sgstTotal, color: '#c8a45a' },
    { label: 'IGST', val: igstTotal, color: '#1a4a7a' },
  ];

  let paths = '';
  let angle = -Math.PI / 2;
  const cx = 65, cy = 65, r = 52, ir = 30;

  slices.forEach(sl => {
    const frac = sl.val / total;
    const a2 = angle + frac * 2 * Math.PI;
    if (frac === 0) { angle = a2; return; }
    const x1 = cx + r * Math.cos(angle), y1 = cy + r * Math.sin(angle);
    const x2 = cx + r * Math.cos(a2), y2 = cy + r * Math.sin(a2);
    const ix1 = cx + ir * Math.cos(angle), iy1 = cy + ir * Math.sin(angle);
    const ix2 = cx + ir * Math.cos(a2), iy2 = cy + ir * Math.sin(a2);
    const large = frac > 0.5 ? 1 : 0;
    paths += `<path d="M${ix1},${iy1} L${x1},${y1} A${r},${r},0,${large},1,${x2},${y2} L${ix2},${iy2} A${ir},${ir},0,${large},0,${ix1},${iy1} Z"
      fill="${sl.color}" opacity="0.9"><title>${sl.label}: ${fmt(sl.val)}</title></path>`;
    angle = a2;
  });

  document.getElementById('pieChart').innerHTML = paths +
    `<text x="${cx}" y="${cy+4}" text-anchor="middle" font-size="10" fill="#3a3f50" font-family="'DM Mono',monospace">GST</text>`;

  const legend = document.getElementById('pieLegend');
  legend.innerHTML = slices.map(sl => `
    <div class="legend-item">
      <div class="legend-dot" style="background:${sl.color}"></div>
      <span>${sl.label}</span>
      <span class="legend-val">${fmt(sl.val)}</span>
    </div>`).join('');
}

function filterTable() {
  const q = document.getElementById('searchInput').value.toLowerCase();
  const type = document.getElementById('filterType').value;
  const rate = document.getElementById('filterRate').value;
  const filtered = sales.filter(r => {
    const matchQ = !q || r.invNo.toLowerCase().includes(q) || r.partyName.toLowerCase().includes(q) || (r.buyerGstin||'').toLowerCase().includes(q);
    const matchType = !type || r.txnType === type;
    const matchRate = !rate || String(r.gstRate) === rate;
    return matchQ && matchType && matchRate;
  });
  renderRegister(filtered);
}

function renderTaxBreakdown() {
  const totalCGST = sales.reduce((s,r) => s + r.cgst, 0);
  const totalSGST = sales.reduce((s,r) => s + r.sgst, 0);
  const totalIGST = sales.reduce((s,r) => s + r.igst, 0);
  const totalNet = totalCGST + totalSGST + totalIGST;

  document.getElementById('tb-cgst').textContent = fmt(totalCGST);
  document.getElementById('tb-sgst').textContent = fmt(totalSGST);
  document.getElementById('tb-igst').textContent = fmt(totalIGST);
  document.getElementById('tb-net').textContent = fmt(totalNet);

  // Slab-wise
  const slabs = [0, 5, 12, 18, 28];
  const slabData = {};
  slabs.forEach(s => slabData[s] = { txns:0, taxable:0, cgst:0, sgst:0, igst:0, tax:0, total:0 });

  sales.forEach(r => {
    const d = slabData[r.gstRate];
    if (!d) return;
    d.txns++; d.taxable += r.taxable; d.cgst += r.cgst; d.sgst += r.sgst;
    d.igst += r.igst; d.tax += r.totalTax; d.total += r.total;
  });

  const slabBody = document.getElementById('slabTbody');
  slabBody.innerHTML = slabs.map(s => {
    const d = slabData[s];
    if (!d.txns) return '';
    return `<tr>
      <td><strong>${s}%</strong></td>
      <td>${d.txns}</td>
      <td class="mono">${fmtN(d.taxable)}</td>
      <td class="mono">${d.cgst ? fmtN(d.cgst) : '—'}</td>
      <td class="mono">${d.sgst ? fmtN(d.sgst) : '—'}</td>
      <td class="mono">${d.igst ? fmtN(d.igst) : '—'}</td>
      <td class="mono" style="color:var(--ruby);font-weight:600">${fmtN(d.tax)}</td>
      <td class="mono" style="font-weight:600">${fmtN(d.total)}</td>
    </tr>`;
  }).join('') || `<tr><td colspan="8"><div class="empty-state"><div class="empty-icon">📊</div><div class="empty-label">No data</div></div></td></tr>`;

  // Category comparison
  const cats = { B2B: {count:0,taxable:0,tax:0,total:0}, B2C: {count:0,taxable:0,tax:0,total:0}, Export: {count:0,taxable:0,tax:0,total:0} };
  sales.forEach(r => {
    const c = cats[r.txnType] || cats.B2C;
    c.count++; c.taxable += r.taxable; c.tax += r.totalTax; c.total += r.total;
  });
  const grandTotal = Object.values(cats).reduce((s,c) => s + c.total, 0);

  document.getElementById('catTbody').innerHTML = Object.entries(cats).map(([name, c]) => `
    <tr>
      <td><span class="badge badge-${name.toLowerCase()}">${name}</span></td>
      <td>${c.count}</td>
      <td class="mono">${fmtN(c.taxable)}</td>
      <td class="mono">${fmtN(c.tax)}</td>
      <td class="mono" style="font-weight:600">${fmtN(c.total)}</td>
      <td class="mono">${pct(c.total, grandTotal)}</td>
    </tr>`).join('');
}

    let fyIdx = (d.getMonth() - 3 + 12) % 12;
    const key = months[fyIdx];
    if (!mData[key]) mData[key] = { invs:0, taxable:0, cgst:0, sgst:0, igst:0, tax:0, total:0 };
    mData[key].invs++; mData[key].taxable+=r.taxable; mData[key].cgst+=r.cgst;
    mData[key].sgst+=r.sgst; mData[key].igst+=r.igst; mData[key].tax+=r.totalTax; mData[key].total+=r.total;
  });

  const rows = months.filter(m => mData[m]).map(m => {
    const d = mData[m];
    return `<tr>
      <td><strong>${m}</strong></td>
      <td>${d.invs}</td>
      <td class="mono">${fmtN(d.taxable)}</td>
      <td class="mono">${fmtN(d.cgst)}</td>
      <td class="mono">${fmtN(d.sgst)}</td>
      <td class="mono">${fmtN(d.igst)}</td>
      <td class="mono" style="color:var(--ruby)">${fmtN(d.tax)}</td>
      <td class="mono" style="font-weight:600">${fmtN(d.total)}</td>
    </tr>`;
  });

  document.getElementById('monthTbody').innerHTML = rows.length ? rows.join('') :
    `<tr><td colspan="8"><div class="empty-state"><div class="empty-icon">📅</div><div class="empty-label">No monthly data</div></div></td></tr>`;
}

function downloadFile(content, filename, type) {
  const blob = new Blob([content], { type });
  const a = document.createElement('a');
  a.href = URL.createObjectURL(blob);
  a.download = filename;
  a.click();
  URL.revokeObjectURL(a.href);
}
