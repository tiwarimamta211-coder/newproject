<script>
  var fieldCount = 0;
  var leads = [];   // in-memory only — no localStorage

  var fieldConfig = {
    text:     { label: 'Short Text',    html: '<input type="text"   class="field-input" placeholder="Your answer">' },
    email:    { label: 'Email Address', html: '<input type="email"  class="field-input" placeholder="email@example.com">' },
    number:   { label: 'Number',        html: '<input type="number" class="field-input" placeholder="0">' },
    textarea: { label: 'Long Text',     html: '<textarea class="field-input" placeholder="Write here..." rows="3"></textarea>' },
    select:   { label: 'Dropdown',      html: '<select class="field-input"><option>Option 1</option><option>Option 2</option><option>Option 3</option></select>' },
    date:     { label: 'Date',          html: '<input type="date" class="field-input">' }
  };

  function addField(type) {
    document.getElementById('emptyState').style.display = 'none';
    fieldCount++;
    var id  = 'field-' + fieldCount;
    var cfg = fieldConfig[type];
    var wrapper = document.createElement('div');
    wrapper.className = 'form-field-wrapper';
    wrapper.id = id;
    wrapper.innerHTML =
      '<button class="delete-btn" onclick="removeField(\'' + id + '\')" title="Delete">✕</button>' +
      '<span class="field-label" contenteditable="true">' + cfg.label + '</span>' +
      cfg.html;
    document.getElementById('formCanvas').appendChild(wrapper);
  }

  function removeField(id) {
    var el = document.getElementById(id);
    if (el) el.remove();
    if (document.querySelectorAll('.form-field-wrapper').length === 0) {
      document.getElementById('emptyState').style.display = 'block';
    }
  }

  function submitForm() {
    var fields = document.querySelectorAll('.form-field-wrapper');
    if (fields.length === 0) { alert('Please add some fields first!'); return; }

    var submission = { id: Date.now(), time: new Date().toLocaleTimeString(), data: {} };
    fields.forEach(function(f) {
      var labelEl = f.querySelector('.field-label');
      var inputEl = f.querySelector('input, select, textarea');
      var key = labelEl ? labelEl.innerText.trim() : 'Field';
      var val = inputEl ? (inputEl.value || '(empty)') : '(empty)';
      submission.data[key] = val;
    });

    leads.unshift(submission);
    renderLeads();
    showToast();
  }

  function renderLeads() {
    var container = document.getElementById('leadsContainer');
    if (leads.length === 0) {
      container.innerHTML = '<div class="empty-leads">No leads yet.<br>Submit the form to capture data.</div>';
      return;
    }
    container.innerHTML = '';
    leads.forEach(function(lead, idx) {
      var card = document.createElement('div');
      card.className = 'lead-card';
      var rows = '';
      for (var key in lead.data) {
        rows += '<div class="lead-row"><strong>' + escHtml(key) + ':</strong> ' + escHtml(lead.data[key]) + '</div>';
      }
      card.innerHTML =
        '<div class="lead-id">Lead #' + (leads.length - idx) + '</div>' +
        rows +
        '<div class="lead-time">Captured at ' + lead.time + '</div>';
      container.appendChild(card);
    });
  }

  function clearLeads() {
    leads = [];
    renderLeads();
  }

  function showToast() {
    var t = document.getElementById('toast');
    t.style.display = 'block';
    setTimeout(function(){ t.style.display = 'none'; }, 2500);
  }

  function escHtml(t) {
    return String(t).replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;');
  }
</script>
