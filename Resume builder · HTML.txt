function switchTab(name, btn) {

/* ────────────────────────────────────────
   SKILLS / TOOLS / SOFT TAGS
──────────────────────────────────────── */
function handleSkillInput(e) { handleTagInput(e, state.skills, 'skills-tags', 'skills-input'); }
function handleToolInput(e) { handleTagInput(e, state.tools, 'tools-tags', 'tools-input'); }
function handleSoftInput(e) { handleTagInput(e, state.soft, 'soft-tags', 'soft-input'); }

function handleTagInput(e, arr, containerId, inputId) {
  if (e.key === 'Enter' || e.key === ',') {
    e.preventDefault();
    const val = e.target.value.trim().replace(/,$/, '');
    if (val && !arr.includes(val)) {
      arr.push(val);
      renderTagsContainer(arr, containerId, inputId);
      updateResume();
    }
    e.target.value = '';
  } else if (e.key === 'Backspace' && e.target.value === '' && arr.length > 0) {
    arr.pop();
    renderTagsContainer(arr, containerId, inputId);
    updateResume();
  }
}

function renderTagsContainer(arr, containerId, inputId) {
  const container = document.getElementById(containerId);
  const input = document.getElementById(inputId);
  container.innerHTML = '';
  arr.forEach((s, i) => {
    const tag = document.createElement('span');
    tag.className = 'tag';
    tag.innerHTML = `${s}<button class="tag-remove" onclick="removeTag(${i},'${containerId}','${inputId}')">×</button>`;
    container.appendChild(tag);
  });
  container.appendChild(input);
}

function removeTag(idx, containerId, inputId) {
  const map = {
    'skills-tags': state.skills,
    'tools-tags': state.tools,
    'soft-tags': state.soft,
  };
  map[containerId].splice(idx, 1);
  renderTagsContainer(map[containerId], containerId, inputId);
  updateResume();
}

        <div class="field"><label>Year</label><input value="${c.year}" placeholder="2022" oninput="state.certs[${i}].year=this.value;updateResume()"></div>
      </div>
    </div>`;
  });
}

function addLang() {
  state.langs.push({ lang: '', level: 'Fluent' });
  renderLangs();
}
function renderLangs() {
  const el = document.getElementById('lang-list');
  el.innerHTML = '';
  state.langs.forEach((l, i) => {
    const row = document.createElement('div');
    row.className = 'portfolio-item';
    row.innerHTML = `
      <input placeholder="Language" value="${l.lang}" style="flex:2" oninput="state.langs[${i}].lang=this.value;updateResume()">
      <select style="flex:1; background:var(--surface2); border:1px solid var(--border); color:var(--text); border-radius:6px; padding:8px; font-size:13px; font-family:inherit" onchange="state.langs[${i}].level=this.value;updateResume()">
        ${['Native','Fluent','Advanced','Intermediate','Basic'].map(o => `<option${l.level===o?' selected':''}>${o}</option>`).join('')}
      </select>
      <button class="btn btn-danger btn-sm btn-icon" onclick="removeItem('langs',${i})">×</button>`;
    el.appendChild(row);
  });
}

function removeItem(key, i) {
  state[key].splice(i, 1);
  const renders = { exps: renderExps, projs: renderProjs, portfolio: renderPortfolio, edu: renderEdu, certs: renderCerts, langs: renderLangs };
  renders[key]();
  updateResume();
}

function handlePhoto(e) {
  const file = e.target.files[0];
  if (!file) return;
  const reader = new FileReader();
  reader.onload = ev => { state.photo = ev.target.result; updateResume(); };
  reader.readAsDataURL(file);
}
function removePhoto() { state.photo = null; updateResume(); }

function changeZoom(delta) {

  applyZoom();
}

  document.getElementById('resume-page').style.transform = `scale(${zoom/100})`;
  document.getElementById('zoom-label').textContent = zoom + '%';
}

/* ────────────────────────────────────────
   COMPLETION METER
──────────────────────────────────────── */
function calcCompletion() {
  const fields = ['name','title','email','phone','location','summary'];
  let filled = fields.filter(f => document.getElementById(f)?.value?.trim()).length;
  if (state.skills.length > 0) filled++;
  if (state.exps.length > 0) filled++;
  const total = fields.length + 2;
  const pct = Math.round(filled / total * 100);
  document.getElementById('completion-pct').textContent = pct + '%';
  document.getElementById('progress-fill').style.width = pct + '%';
}

function gv(id) { return (document.getElementById(id)?.value || '').trim(); }

}

/* ────────────────────────────────────────
   RENDER RESUME
──────────────────────────────────────── */
function updateResume() {
  calcCompletion();
  const page = document.getElementById('resume-page');
  page.className = `resume-page tpl-${state.template}`;
  page.style.setProperty('--r-accent', state.accent);
  page.style.setProperty('--r-font', state.font);

  const name = gv('name') || 'Your Name';
  const title = gv('title') || 'Freelance Professional';
  const email = gv('email');
  const phone = gv('phone');
  const loc = gv('location');
  const website = gv('website');
  const linkedin = gv('linkedin');
  const github = gv('github');
  const summary = gv('summary');
  const rate = gv('rate');
  const avail = gv('availability');

  const photoHtml = state.photo
    ? `<img class="resume-photo" src="${state.photo}" alt="Profile">`
    : '';

  const contactHtml = [
    email ? `📧 ${email}` : '',

      <div class="exp-role">${p.name || 'Project Name'}${p.client ? ` — ${p.client}` : ''}</div>

  const allExpAndProj = expItems + (projItems ? `<div class="section-title" style="margin-top:20px">Freelance Projects</div>${projItems}` : '');

  const toolTags = state.tools.map(s => `<span class="skill-tag">${s}</span>`).join('');

  const eduHtml = state.edu.map(e =>

    `<div style="font-size:12px;margin-bottom:6px;color:#333"><strong>${c.name || 'Certification'}</strong> <span style="color:#888">· ${c.issuer || ''} ${c.year || ''}</span></div>`).join('');

  const langsHtml = state.langs.map(l =>
    `<div style="font-size:12px;margin-bottom:4px;color:#333">${l.lang || 'Language'} <span style="color:#888">· ${l.level}</span></div>`).join('');

  const serviceLines = gv('services').split('\n').filter(Boolean).map(s => `<div class="exp-desc">• ${s}</div>`).join('');
  const softTags = state.soft.map(s => `<span class="skill-tag">${s}</span>`).join('');

  /* ── Template HTML ── */
  if (state.template === 'modern') {
    page.innerHTML = `
      <div class="resume-header">
        ${photoHtml}
        <div>
          <div class="resume-name">${name}</div>
          <div class="resume-title">${title}</div>
          <div class="resume-contact">${contactHtml.map(c => `<span class="contact-item">${c}</span>`).join('')}</div>
        </div>
      </div>
      <div class="resume-body">
        <div class="resume-main">
          ${summary ? `<div class="section-title">Profile</div><div class="profile-text">${summary}</div>` : ''}
          ${state.exps.length > 0 || state.projs.length > 0 ? `<div class="section-title">Experience</div>${allExpAndProj}` : ''}
          ${state.edu.length > 0 ? `<div class="section-title">Education</div>${eduHtml}` : ''}
          ${portfolioHtml ? `<div class="section-title">Portfolio</div>${portfolioHtml}` : ''}
        </div>
        <div class="resume-side">
          ${state.skills.length > 0 ? `<div class="section-title">Skills</div>${skillTags}` : ''}
          ${state.tools.length > 0 ? `<div class="section-title">Tools</div>${toolTags}` : ''}
          ${state.soft.length > 0 ? `<div class="section-title">Soft Skills</div>${softTags}` : ''}
          ${serviceLines ? `<div class="section-title">Services</div>${serviceLines}` : ''}
          ${state.certs.length > 0 ? `<div class="section-title">Certifications</div>${certsHtml}` : ''}
          ${state.langs.length > 0 ? `<div class="section-title">Languages</div>${langsHtml}` : ''}
        </div>
      </div>`;

  } else if (state.template === 'minimal') {
    const buildRow = (label, content) => content ? `<div class="resume-row"><div class="section-title">${label}</div><div>${content}</div></div>` : '';
    page.innerHTML = `
      <div class="resume-header">
        <div class="resume-name">${name}</div>
        <div class="resume-title">${title}</div>
        <div class="resume-contact">${contactHtml.map(c => `<span>${c}</span>`).join('')}</div>
      </div>
      <div class="resume-body">
        ${buildRow('Profile', summary ? `<div class="profile-text">${summary}</div>` : '')}

        ${state.edu.length > 0 ? buildRow('Education', eduHtml) : ''}

        ${state.langs.length > 0 ? buildRow('Languages', langsHtml) : ''}
        ${portfolioHtml ? buildRow('Portfolio', portfolioHtml) : ''}
      </div>`;

  } else if (state.template === 'bold') {
    page.innerHTML = `
      <div class="resume-header">
        <div class="resume-name">${name}</div>
        <div class="resume-title">${title}</div>
        <div class="resume-contact">${contactHtml.map(c => `<span>${c}</span>`).join('')}</div>
      </div>
      <div class="resume-body">
        <div class="resume-side">
          ${state.skills.length > 0 ? `<div class="section-title">Skills</div>${state.skills.map(s => `<div class="skill-tag">${s}</div>`).join('')}` : ''}
          ${state.tools.length > 0 ? `<div class="section-title">Tools</div>${state.tools.map(s => `<div class="skill-tag">${s}</div>`).join('')}` : ''}
          ${state.edu.length > 0 ? `<div class="section-title">Education</div>${eduHtml}` : ''}
          ${state.certs.length > 0 ? `<div class="section-title">Certifications</div>${certsHtml}` : ''}
          ${state.langs.length > 0 ? `<div class="section-title">Languages</div>${langsHtml}` : ''}

        </div>
      </div>`;

  } else if (state.template === 'creative') {
    const skillBars = state.skills.slice(0,8).map(s =>
      `<div class="skill-bar"><div class="skill-bar-label"><span>${s}</span></div><div class="skill-bar-track"><div class="skill-bar-fill" style="width:${70+Math.random()*30|0}%"></div></div></div>`).join('');
    page.innerHTML = `
      <div class="resume-side">
        ${photoHtml}
        <div class="resume-name">${name}</div>
        <div class="resume-title">${title}</div>
        ${contactHtml.length ? `<div class="side-section"><div class="side-section-title">Contact</div>${contactHtml.map(c => `<div class="contact-line">→ ${c}</div>`).join('')}</div>` : ''}
        ${skillBars ? `<div class="side-section"><div class="side-section-title">Skills</div>${skillBars}</div>` : ''}
        ${state.tools.length ? `<div class="side-section"><div class="side-section-title">Tools</div><div>${state.tools.map(t=>`<span style="font-size:11px;color:rgba(255,255,255,0.7);display:block;margin-bottom:4px">• ${t}</span>`).join('')}</div></div>` : ''}
        ${state.langs.length ? `<div class="side-section"><div class="side-section-title">Languages</div>${langsHtml}</div>` : ''}
      </div>
      <div class="resume-main">
        ${summary ? `<div class="section-title">About Me</div><div class="profile-text">${summary}</div>` : ''}
        ${state.exps.length > 0 ? `<div class="section-title">Experience</div>${state.exps.map(e => `<div class="exp-item"><div class="exp-header"><div><div class="exp-role">${e.role || 'Role'}</div><div class="exp-company">${e.company||''}</div></div><div class="exp-period">${e.period||''}</div></div>${e.desc?`<div class="exp-desc">${e.desc}</div>`:''}</div>`).join('')}` : ''}
        ${state.projs.length > 0 ? `<div class="section-title">Projects</div>${state.projs.map(p=>`<div class="exp-item"><div class="exp-role">${p.name||'Project'}${p.client?` · ${p.client}`:''}</div><div class="exp-period">${p.period||''}</div>${p.desc?`<div class="exp-desc">${p.desc}</div>`:''}</div>`).join('')}` : ''}

/* ────────────────────────────────────────
   SAVE / LOAD
──────────────────────────────────────── */
function saveData() {
  const data = {
    state,
    fields: ['name','title','email','phone','location','website','linkedin','github','summary','rate','currency','availability','services'].reduce((o,k) => { o[k] = gv(k); return o; }, {})
  };
  localStorage.setItem('freelancecv', JSON.stringify(data));
  showToast('✓ Resume saved to browser!');
}

function loadData() {
  const raw = localStorage.getItem('freelancecv');
  if (!raw) return;
  try {
    const data = JSON.parse(raw);
    Object.assign(state, data.state);
    Object.entries(data.fields || {}).forEach(([k, v]) => {
      const el = document.getElementById(k);
      if (el) el.value = v;
    });
    renderTagsContainer(state.skills, 'skills-tags', 'skills-input');
    renderTagsContainer(state.tools, 'tools-tags', 'tools-input');
    renderTagsContainer(state.soft, 'soft-tags', 'soft-input');
    renderExps(); renderProjs(); renderPortfolio(); renderEdu(); renderCerts(); renderLangs();
    updateResume();
  } catch(e) {}
}
