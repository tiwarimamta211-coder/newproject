// ─── KEYWORDS ────────────────────────────────────────────────────────────────
document.getElementById('keywordInput').addEventListener('keydown', e => {
  if (e.key === 'Enter') addKeyword();
});

function addKeyword() {
  const input = document.getElementById('keywordInput');
  const val = input.value.trim();
  if (!val || keywords.includes(val)) { input.value = ''; return; }
  keywords.push(val);
  renderChips();
  input.value = '';
}

function removeKeyword(kw) {
  keywords = keywords.filter(k => k !== kw);
  renderChips();
}

function renderChips() {
  const container = document.getElementById('keywordChips');
  container.innerHTML = keywords.map(kw =>
    `<div class="keyword-chip">${kw}<span class="chip-remove" onclick="removeKeyword('${kw}')">×</span></div>`
  ).join('');
}

// ─── STATS ───────────────────────────────────────────────────────────────────
function updateStats() {
  document.getElementById('statBlogs').textContent = blogCount;
  document.getElementById('statWords').textContent = wordTotal >= 1000
    ? (wordTotal / 1000).toFixed(1) + 'k' : wordTotal;
}

async function generateBlog() {
  const topic = document.getElementById('topicInput').value.trim();
  if (!topic) { showToast('Please enter a blog topic!'); return; }

  const audience = document.getElementById('audienceInput').value.trim() || 'general readers';
  const context = document.getElementById('contextInput').value.trim();
  const length = document.getElementById('lengthSelect').value;
  const tone = document.getElementById('toneSelect').value;

  const wordTarget = length === 'short' ? 600 : length === 'medium' ? 1000 : 1500;

  document.getElementById('loadingOverlay').classList.add('active');
  document.getElementById('generateBtn').disabled = true;
  animateSteps();

  const kwList = keywords.length ? keywords.join(', ') : 'relevant industry terms';

  const prompt = `You are an expert SEO blog writer. Write a complete, publish-ready SEO blog post based on:

TOPIC: ${topic}
TARGET AUDIENCE: ${audience}
TONE: ${tone}
WORD COUNT TARGET: ~${wordTarget} words
TARGET KEYWORDS: ${kwList}
${context ? `EXTRA CONTEXT: ${context}` : ''}

Return ONLY a valid JSON object (no markdown, no backticks) with this exact structure:
{

// ─── UTILITIES ───────────────────────────────────────────────────────────────
function countWords(text) {
  return text.trim().split(/\s+/).filter(Boolean).length;
}

function showToast(msg) {
  const t = document.getElementById('toast');
  t.textContent = msg;
  t.classList.add('show');
  setTimeout(() => t.classList.remove('show'), 3000);
}

function copyText() {
  if (!lastBlogData) return;
  const plain = lastBlogData.body.replace(/<[^>]+>/g, '').trim();
  navigator.clipboard.writeText(`${lastBlogData.seoTitle}\n\n${plain}`);
  showToast('✓ Plain text copied!');
}

function copyMarkdown() {
  if (!lastBlogData) return;
  let md = lastBlogData.body
    .replace(/<h2>(.*?)<\/h2>/gi, '\n## $1\n')
    .replace(/<h3>(.*?)<\/h3>/gi, '\n### $1\n')
    .replace(/<p>(.*?)<\/p>/gi, '$1\n\n')
    .replace(/<strong>(.*?)<\/strong>/gi, '**$1**')
    .replace(/<blockquote>(.*?)<\/blockquote>/gi, '> $1\n')
    .replace(/<li>(.*?)<\/li>/gi, '- $1\n')
    .replace(/<[^>]+>/g, '').trim();
  navigator.clipboard.writeText(`# ${lastBlogData.seoTitle}\n\n${md}`);
  showToast('✓ Markdown copied!');
}

function downloadBlog() {
  if (!lastBlogData) return;
  const plain = lastBlogData.body.replace(/<[^>]+>/g, '');
  const content = [
    `TITLE: ${lastBlogData.seoTitle}`,
    `META TITLE: ${lastBlogData.metaTitle}`,
    `META DESC: ${lastBlogData.metaDescription}`,
    `URL SLUG: ${lastBlogData.urlSlug}`,
    `KEYWORDS: ${(lastBlogData.keywordsUsed || []).join(', ')}`,
    `\n─────────────────────────────────\n`,
    plain
  ].join('\n');
  const blob = new Blob([content], { type: 'text/plain' });
  const a = document.createElement('a');
  a.href = URL.createObjectURL(blob);
  a.download = `${lastBlogData.urlSlug || 'blog-post'}.txt`;
  a.click();
  showToast('✓ Downloaded!');
}
