// ── API STREAM ─────────────────────────────────────────────────────
async function streamAPI(apiKey,prompt,onChunk,onDone,onError){
  try{
    const res=await fetch("https://api.anthropic.com/v1/messages",{
      method:"POST",
      headers:{"Content-Type":"application/json","x-api-key":apiKey,"anthropic-version":"2023-06-01"},
      body:JSON.stringify({model:"claude-sonnet-4-20250514",max_tokens:8000,stream:true,messages:[{role:"user",content:prompt}]}),
    });
    if(!res.ok){const e=await res.json();throw new Error(e.error?.message||"API Error");}
    const reader=res.body.getReader();const dec=new TextDecoder();
    while(true){
      const{done,value}=await reader.read();if(done)break;
      for(const line of dec.decode(value).split("\n")){
        if(line.startsWith("data: ")){
          try{const d=JSON.parse(line.slice(6));if(d.type==="content_block_delta"&&d.delta?.text)onChunk(d.delta.text);}catch{}
        }
      }
    }
    onDone();
  }catch(e){onError(e.message);}
}

            </div>
            <div style={{marginLeft:"auto",display:"flex",gap:8}}>
              {done&&<button onClick={downloadAll} style={{padding:"10px 20px",background:"linear-gradient(135deg,#68d391,#38a169)",border:"none",borderRadius:11,color:"white",fontWeight:800,fontSize:13,cursor:"pointer",fontFamily:"'Satoshi',sans-serif"}}>⬇️ Download All</button>}
              <button onClick={()=>setShowModal(true)} style={{padding:"9px 18px",background:apiKey?"rgba(99,179,237,0.1)":"rgba(255,100,50,0.15)",border:`1.5px solid ${apiKey?"rgba(99,179,237,0.3)":"rgba(255,100,50,0.4)"}`,borderRadius:11,color:apiKey?"#63b3ed":"#ff6432",fontWeight:700,fontSize:12,cursor:"pointer",fontFamily:"'Satoshi',sans-serif"}}>
                {apiKey?"🔑 Key Set ✓":"🔑 API Key"}
              </button>
            </div>
          </div>

          {/* INPUT FORM */}
          <div style={{display:"grid",gridTemplateColumns:"1fr 1fr 1fr",gap:10,marginBottom:10}}>
            <div style={{gridColumn:"1/-1"}}>
              <label>Video Topic *</label>
              <input
                placeholder="e.g. GST filing mistakes जो small business करते हैं — या — 5 ways to grow YouTube channel fast"
                value={topic}
                onChange={e=>setTopic(e.target.value)}
                onKeyDown={e=>e.key==="Enter"&&!generating&&handleGenerate()}
                style={{fontSize:15,padding:"14px 18px"}}
              />
            </div>
            <div>
              <label>Niche</label>
              <select value={niche} onChange={e=>setNiche(e.target.value)}>
                {NICHES.map(n=><option key={n}>{n}</option>)}
              </select>
            </div>
            <div>
              <label>Duration</label>
              <select value={duration} onChange={e=>setDuration(e.target.value)}>
                {DURATIONS.map(d=><option key={d}>{d}</option>)}
              </select>
            </div>
            <div>
              <label>Language</label>
              <select value={lang} onChange={e=>setLang(e.target.value)}>
                {LANGS.map(l=><option key={l}>{l}</option>)}
              </select>
            </div>
            <div>
              <label>Tone</label>
              <select value={tone} onChange={e=>setTone(e.target.value)}>
                {TONES.map(t=><option key={t}>{t}</option>)}
              </select>
            </div>
            <div>
              <label>Target Audience</label>
              <input placeholder="e.g. Small business owners" value={audience} onChange={e=>setAudience(e.target.value)}/>
            </div>
            <div>
              <label>SEO Keywords</label>
              <input placeholder="e.g. GST, tax filing, 2025" value={keywords} onChange={e=>setKeywords(e.target.value)}/>
            </div>
          </div>

          {/* BIG GENERATE BUTTON */}
          <button className="big-btn" onClick={handleGenerate} disabled={generating||!topic.trim()}
            style={{width:"100%",padding:"18px",fontSize:18}}>
            {generating?(
              <span style={{display:"flex",alignItems:"center",justifyContent:"center",gap:12}}>
                <span style={{width:20,height:20,border:"3px solid white",borderTopColor:"transparent",borderRadius:"50%",display:"inline-block",animation:"spin 0.7s linear infinite"}}/>
                Package बन रहा है... {progress}% ({completedCount}/12 sections)
              </span>
            ):"🚀 ONE CLICK — पूरा Package Generate करो"}
          </button>

          {/* PROGRESS BAR */}
          {generating&&(
            <div style={{marginTop:12}}>
              <div className="progress-track">
                <div className="progress-fill" style={{width:`${progress}%`,background:"linear-gradient(90deg,#ff6432,#ff9500,#f6c90e)"}}/>
              </div>
              <div style={{display:"flex",justifyContent:"space-between",marginTop:6,fontSize:11}}>
                <span style={{color:"#ff6432",animation:"pulse 1.5s infinite"}}>
                  {currentSec&&SECTIONS.find(s=>s.id===currentSec)?`${SECTIONS.find(s=>s.id===currentSec).icon} ${SECTIONS.find(s=>s.id===currentSec).label} लिख रहे हैं...`:"शुरू हो रहा है..."}
                </span>
                <span style={{color:"#444"}}>{totalWords.toLocaleString()} words generated</span>
              </div>
              {/* Section Pills */}
              <div style={{display:"flex",flexWrap:"wrap",gap:4,marginTop:10}}>
                {SECTIONS.map(s=>{
                  const isDone=sections[s.id]&&sections[s.id].length>100;
                  const isActive=currentSec===s.id;
                  return(
                    <span key={s.id} className="tag-pill" style={{
                      background:isDone?`${s.color}20`:isActive?`${s.color}12`:"rgba(255,255,255,0.03)",
                      border:`1px solid ${isDone?s.color+"55":isActive?s.color+"33":"rgba(255,255,255,0.06)"}`,
                      color:isDone?s.color:isActive?s.color+"aa":"#333",
                      animation:isActive?"pulse 1s infinite":"none",
                    }}>
                      {isDone?"✓ ":isActive?"⚡ ":""}{s.short}
                    </span>
                  );
                })}
              </div>
            </div>
          )}

          {/* DONE STATS */}
          {done&&(
            <div style={{marginTop:14,display:"grid",gridTemplateColumns:"repeat(4,1fr)",gap:10,animation:"fadeUp 0.4s ease"}}>
              {[
                ["📦","12 Sections","Complete Package"],
                ["📝",`${totalWords.toLocaleString()}`,"Total Words"],
                ["⚡","Instant","Ready to Use"],
                ["💾","Download","All as .txt"],
              ].map(([icon,val,label])=>(
                <div key={label} className="stat-box">
                  <div style={{fontSize:20,marginBottom:4}}>{icon}</div>
                  <div style={{fontSize:18,fontWeight:900,color:"#ff6432"}}>{val}</div>
                  <div style={{fontSize:10,color:"#444",marginTop:2}}>{label}</div>
                </div>
              ))}
            </div>
          )}

          {error&&<div style={{marginTop:12,padding:"12px 16px",background:"rgba(252,129,129,0.08)",border:"1px solid rgba(252,129,129,0.2)",borderRadius:10,fontSize:12,color:"#fc8181"}}>❌ {error}</div>}
        </div>
      </div>

      {/* SECTIONS OUTPUT */}
      <div style={{maxWidth:900,margin:"0 auto",padding:"24px"}}>
        {(generating||done)&&(
          <div style={{display:"flex",flexDirection:"column",gap:12}}>
            {SECTIONS.map((s,i)=>(
              <SectionCard
                key={s.id}
                section={s}
                content={sections[s.id]||""}
                index={i}
                isGenerating={generating}
                currentSection={currentSec}
              />
            ))}
          </div>
        )}

        {!generating&&!done&&(
          <div style={{textAlign:"center",padding:"60px 20px",opacity:0.4}}>
            <div style={{fontSize:80,marginBottom:16,animation:"float 3s ease infinite"}}>🎬</div>
            <div style={{fontSize:22,fontWeight:900,color:"#666",marginBottom:8}}>Ready to Create!</div>
            <div style={{fontSize:14,color:"#444",marginBottom:24}}>Topic डालें और One Click में 12 sections का पूरा package पाएं</div>
            <div style={{display:"flex",justifyContent:"center",flexWrap:"wrap",gap:8}}>
              {SECTIONS.map(s=>(
                <span key={s.id} style={{fontSize:12,background:"rgba(255,255,255,0.03)",border:"1px solid rgba(255,255,255,0.07)",padding:"5px 12px",borderRadius:20,color:"#444"}}>
                  {s.icon} {s.label}
                </span>
              ))}
            </div>
          </div>
        )}
      </div>
    </div>
  );
}

ReactDOM.createRoot(document.getElementById("root")).render(<App/>);
