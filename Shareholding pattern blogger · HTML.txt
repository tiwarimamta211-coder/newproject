<script>
  var stakeholders = [];

  function sync() {
    var c = document.getElementById('compName').value.trim();
    var q = document.getElementById('quarterDate').value.trim();
    document.getElementById('dr_compName').innerText = c || 'Corporate Entity Name';
    document.getElementById('dr_quarter').innerText = q ? 'Period: ' + q : 'Period: Not Set';
  }

  function addShareholder() {
    var name     = document.getElementById('shName').value.trim();
    var shares   = parseInt(document.getElementById('shShares').value) || 0;
    var category = document.getElementById('shCategory').value;

    if (!name || shares <= 0) {
      alert('Please enter a valid name and number of shares.');
      return;
    }

    stakeholders.push({ id: Date.now(), name: name, shares: shares, category: category });
    document.getElementById('shName').value   = '';
    document.getElementById('shShares').value = '';
    updateReport();
  }

  function removeShareholder(id) {
    stakeholders = stakeholders.filter(function(s) { return s.id !== id; });
    updateReport();
  }

  function fmt(n) { return n.toLocaleString('en-IN'); }

  function updateReport() {
    var total = stakeholders.reduce(function(sum, s) { return sum + s.shares; }, 0);
    var promoterTotal = stakeholders
      .filter(function(s) { return s.category === 'Promoter'; })
      .reduce(function(sum, s) { return sum + s.shares; }, 0);
    var publicTotal = total - promoterTotal;

    document.getElementById('resTotalShares').innerText  = fmt(total);
    document.getElementById('tabTotalShares').innerText  = fmt(total);
    document.getElementById('resPromoterPct').innerText  = total > 0 ? ((promoterTotal / total) * 100).toFixed(2) + '%' : '0.00%';
    document.getElementById('resPublicPct').innerText    = total > 0 ? ((publicTotal   / total) * 100).toFixed(2) + '%' : '0.00%';

    // MPS Compliance: Public shareholding must be >= 25%
    var mpsEl = document.getElementById('resMPS');
    if (total === 0) {
      mpsEl.innerText = '–';
      mpsEl.className = '';
    } else if ((publicTotal / total) * 100 >= 25) {
      mpsEl.innerText = '✔ Compliant';
      mpsEl.className = 'mps-ok';
    } else {
      mpsEl.innerText = '✘ Non-Compliant';
      mpsEl.className = 'mps-bad';
    }

    var tbody = document.getElementById('shTableBody');
    if (stakeholders.length === 0) {
      tbody.innerHTML = '<tr><td colspan="5" class="empty-state"><div>📋</div>No shareholders added yet.<br>Use the form on the left to begin.</td></tr>';
      return;
    }

    tbody.innerHTML = stakeholders.map(function(s) {
      var pct = total > 0 ? ((s.shares / total) * 100).toFixed(2) : '0.00';
      return '<tr>' +
        '<td class="bold">' + s.name + '</td>' +
        '<td><span class="cat-badge">' + s.category + '</span></td>' +
        '<td style="text-align:right;font-family:monospace">' + fmt(s.shares) + '</td>' +
        '<td style="text-align:right;color:#4f46e5;font-weight:700">' + pct + '%</td>' +
        '<td style="text-align:center"><button class="del-btn" onclick="removeShareholder(' + s.id + ')">🗑</button></td>' +
        '</tr>';
    }).join('');
  }
</script>
