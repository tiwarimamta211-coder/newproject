  updateSlideCounter(i);
  updateProgress(i);
  runSlideAnimation(i);
}

function updateProgress(i) {
  const pct = (i / (totalSlides-1)) * 100;
  document.getElementById('progress-bar').style.width = pct + '%';
}

function clearAllTimers() {
  slideTimers.forEach(t => clearTimeout(t));
  slideTimers = [];
}

function addTimer(fn, delay) {
  slideTimers.push(setTimeout(fn, delay));
}

function nextSlide() {
  if(currentSlide < totalSlides - 1) showSlide(currentSlide + 1);
}
function prevSlide() {
  if(currentSlide > 0) showSlide(currentSlide - 1);
}

function toggleAuto() {
  if(isAutoPlaying) {
    isAutoPlaying = false;
    clearInterval(autoInterval);
    document.getElementById('play-pause-btn').textContent = '▶ Play';
  } else {
    isAutoPlaying = true;
    startAuto();
    document.getElementById('play-pause-btn').textContent = '⏸ Pause';
  }
}

function startAuto() {
  clearInterval(autoInterval);
  autoInterval = setInterval(() => {
    if(currentSlide < totalSlides - 1) nextSlide();
    else { clearInterval(autoInterval); }
  }, SLIDE_DURATION);
}

function restartDemo() {
  clearInterval(autoInterval);
  clearAllTimers();
  clearInterval(timerInterval);
  timerStart = Date.now();
  startTimer();
  isAutoPlaying = true;
  document.getElementById('play-pause-btn').textContent = '⏸ Pause';
  showSlide(0);
  startAuto();
}

function startTimer() {
  timerInterval = setInterval(() => {
    const elapsed = Math.floor((Date.now() - timerStart) / 1000);
    const m = String(Math.floor(elapsed/60)).padStart(2,'0');
    const s = String(elapsed%60).padStart(2,'0');
    document.getElementById('timer-text').textContent = `${m}:${s}`;
  }, 1000);
}
