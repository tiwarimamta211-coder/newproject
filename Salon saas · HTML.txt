function uid(){return Date.now().toString(36)+Math.random().toString(36).slice(2,6);}
function today(){return new Date().toISOString().split('T')[0];}
function fmt(n){return'₹'+Number(n||0).toLocaleString('en-IN');}
function fmtDate(d){if(!d)return'—';const[y,m,dd]=d.split('-');return`${dd}/${m}/${y}`;}
function statusBadge(s){const m={'Confirmed':'badge-gold','Completed':'badge-success','Cancelled':'badge-danger','Pending':'badge-rose'};return`<span class="badge ${m[s]||'badge-rose'}">${s}</span>`;}
function typeBadge(t){return t==='income'?`<span class="badge badge-success">Income</span>`:`<span class="badge badge-danger">Expense</span>`;}

function renderDashboard(){
  const todayStr=today();
  const todayBookings=bookings.filter(b=>b.date===todayStr);
  document.getElementById('st-today').textContent=todayBookings.length;
  document.getElementById('st-clients').textContent=clients.length;
  document.getElementById('st-staff').textContent=staff.length;
  const mon=new Date().toISOString().slice(0,7);
  const rev=bookings.filter(b=>b.date&&b.date.startsWith(mon)&&b.status==='Completed').reduce((s,b)=>s+Number(b.amount),0);
  document.getElementById('st-revenue').textContent=fmt(rev);
  // bar chart — last 7 months
  const chart=document.getElementById('revenueChart');chart.innerHTML='';
  const months=[];for(let i=6;i>=0;i--){const d=new Date();d.setMonth(d.getMonth()-i);months.push(d.toISOString().slice(0,7));}
  const maxRev=Math.max(...months.map(m=>finance.filter(f=>f.type==='income'&&f.date&&f.date.startsWith(m)).reduce((s,f)=>s+Number(f.amount),0)),1);
  months.forEach(m=>{
    const r=finance.filter(f=>f.type==='income'&&f.date&&f.date.startsWith(m)).reduce((s,f)=>s+Number(f.amount),0);
    const pct=Math.round((r/maxRev)*100);
    const lbl=m.slice(5,7)+'/'+m.slice(2,4);
    chart.innerHTML+=`<div class="bar-wrap"><div class="bar" style="height:${Math.max(pct,3)}%"></div><div class="bar-label">${lbl}</div></div>`;
  });
  // today's schedule
  const tl=document.getElementById('todayList');
  if(todayBookings.length===0){tl.innerHTML='<p style="color:var(--text3);font-size:13px;">No bookings today.</p>';}
  else{tl.innerHTML=todayBookings.map(b=>`<div class="mini-list-item"><div><div class="mini-name">${b.client}</div><div class="mini-sub">${b.service} · ${b.time}</div></div>${statusBadge(b.status)}</div>`).join('');}
  // recent bookings
  const rb=document.getElementById('recentBookings');
  const recent=[...bookings].reverse().slice(0,5);
  rb.innerHTML=recent.length?recent.map(b=>`<div class="mini-list-item"><div><div class="mini-name">${b.client}</div><div class="mini-sub">${b.service} · ${fmtDate(b.date)}</div></div><div style="text-align:right;"><div style="font-size:13px;color:var(--gold);">${fmt(b.amount)}</div>${statusBadge(b.status)}</div></div>`).join(''):'<p style="color:var(--text3);font-size:13px;">No bookings yet.</p>';
  // top services
  const sc={};bookings.forEach(b=>{if(b.service)sc[b.service]=(sc[b.service]||0)+1;});
  const top=Object.entries(sc).sort((a,b)=>b[1]-a[1]).slice(0,5);
  const ts=document.getElementById('topServices');
  ts.innerHTML=top.length?top.map(([s,c])=>`<div class="mini-list-item"><div class="mini-name">${s}</div><span class="badge badge-gold">${c} bookings</span></div>`).join(''):'<p style="color:var(--text3);font-size:13px;">No data yet.</p>';
}

let calYear=new Date().getFullYear(), calMonth=new Date().getMonth();
function renderCalendar(){
  const mn=['January','February','March','April','May','June','July','August','September','October','November','December'];
  document.getElementById('calMonthLabel').textContent=mn[calMonth]+' '+calYear;
  const grid=document.getElementById('calGrid');grid.innerHTML='';
  const first=new Date(calYear,calMonth,1).getDay();
  const days=new Date(calYear,calMonth+1,0).getDate();
  const prevDays=new Date(calYear,calMonth,0).getDate();
  const todayStr=today();
  for(let i=0;i<42;i++){
    let d,month='cur';
    if(i<first){d=prevDays-(first-i-1);month='prev';}
    else if(i-first>=days){d=i-first-days+1;month='next';}
    else{d=i-first+1;}
    const dateStr=month==='cur'?`${calYear}-${String(calMonth+1).padStart(2,'0')}-${String(d).padStart(2,'0')}`:'';
    const dayBookings=dateStr?bookings.filter(b=>b.date===dateStr):[];
    const isToday=dateStr===todayStr;
    grid.innerHTML+=`<div class="cal-day${month!=='cur'?' other-month':''}${isToday?' today':''}">
      <div class="cal-day-num">${d}</div>
      ${dayBookings.slice(0,2).map(b=>`<div class="cal-event">${b.time} ${b.client}</div>`).join('')}
      ${dayBookings.length>2?`<div style="font-size:10px;color:var(--text3);margin-top:2px;">+${dayBookings.length-2} more</div>`:''}
    </div>`;
  }
}
function changeMonth(dir){calMonth+=dir;if(calMonth>11){calMonth=0;calYear++;}else if(calMonth<0){calMonth=11;calYear--;}renderCalendar();}

function renderFinance(){
  const inc=finance.filter(f=>f.type==='income').reduce((s,f)=>s+Number(f.amount),0);
  const exp=finance.filter(f=>f.type==='expense').reduce((s,f)=>s+Number(f.amount),0);

  const net=inc-exp;

  netEl.style.color=net>=0?'var(--success)':'var(--danger)';

    <td>${i+1}</td><td>${f.desc}</td><td>${f.category}</td>
    <td>${fmtDate(f.date)}</td><td>${typeBadge(f.type)}</td>
    <td style="color:${f.type==='income'?'var(--success)':'var(--danger)'};">${f.type==='income'?'+':'−'}${fmt(f.amount)}</td>
    <td><button class="btn btn-danger btn-sm" onclick="deleteItem('finance','${f.id}',renderFinance)">Del</button></td>
  </tr>`).join('')||`<tr><td colspan="7" style="text-align:center;color:var(--text3);padding:32px;">No transactions.</td></tr>`;

function renderReports(){
  document.getElementById('rpt-bookings').textContent=bookings.length;
  const comp=bookings.filter(b=>b.status==='Completed').length;
  document.getElementById('rpt-complete').textContent=bookings.length?Math.round(comp/bookings.length*100)+'%':'0%';
  const rev=bookings.filter(b=>b.status==='Completed').reduce((s,b)=>s+Number(b.amount),0);
  document.getElementById('rpt-avg').textContent=comp?fmt(Math.round(rev/comp)):'₹0';
  const staffRev={};bookings.filter(b=>b.status==='Completed').forEach(b=>{staffRev[b.staff]=(staffRev[b.staff]||0)+Number(b.amount);});
  const topStaff=Object.entries(staffRev).sort((a,b)=>b[1]-a[1])[0];
  document.getElementById('rpt-topstaff').textContent=topStaff?topStaff[0]:'—';
  // status breakdown
  const statuses=['Confirmed','Completed','Cancelled','Pending'];
  const statusCount={};statuses.forEach(s=>statusCount[s]=bookings.filter(b=>b.status===s).length);
  document.getElementById('statusBreakdown').innerHTML=statuses.map(s=>`<div class="mini-list-item"><span>${s}</span>${statusBadge(s)} <b style="color:var(--text);">${statusCount[s]}</b></div>`).join('');
  // service revenue
  const srvRev={};bookings.filter(b=>b.status==='Completed').forEach(b=>{srvRev[b.service]=(srvRev[b.service]||0)+Number(b.amount);});
  const top5=Object.entries(srvRev).sort((a,b)=>b[1]-a[1]).slice(0,5);
  document.getElementById('serviceRevenue').innerHTML=top5.length?top5.map(([s,r])=>`<div class="mini-list-item"><div class="mini-name">${s}</div><b style="color:var(--gold);">${fmt(r)}</b></div>`).join(''):'<p style="color:var(--text3);font-size:13px;">No data.</p>';
}

function deleteItem(key,id,renderFn){
  if(!confirm('Are you sure you want to delete this record?'))return;
  if(key==='bookings')bookings=bookings.filter(x=>x.id!==id);
  else if(key==='clients')clients=clients.filter(x=>x.id!==id);
  else if(key==='services')services=services.filter(x=>x.id!==id);
  else if(key==='staff')staff=staff.filter(x=>x.id!==id);
  else if(key==='inventory')inventory=inventory.filter(x=>x.id!==id);
  else if(key==='finance')finance=finance.filter(x=>x.id!==id);
  else if(key==='assets')assets=assets.filter(x=>x.id!==id);
  else if(key==='liabilities')liabilities=liabilities.filter(x=>x.id!==id);
  save(key,eval(key));renderFn();toast('Deleted.');
}
