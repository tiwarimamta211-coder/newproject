// ── HISTORY ─────────────────────────────────────────────────
const history = { states: [], idx: -1, max: 30 };
function saveState(label='Edit') {
  const data = canvas.toDataURL();
  if (history.idx < history.states.length - 1) history.states = history.states.slice(0, history.idx + 1);
  history.states.push({ data, label });
  if (history.states.length > history.max) history.states.shift();
  history.idx = history.states.length - 1;
  renderHistory();
}
function applyState(idx) {
  if (idx < 0 || idx >= history.states.length) return;
  history.idx = idx;
  const img = new Image();
  img.onload = () => { ctx.clearRect(0,0,canvas.width,canvas.height); ctx.drawImage(img,0,0); renderHistory(); };
  img.src = history.states[idx].data;
}
function undo() { applyState(history.idx - 1); }
function redo() { applyState(history.idx + 1); }

function renderHistory() {
  const list = document.getElementById('history-list');
  list.innerHTML = history.states.map((s,i) =>
    `<div class="history-item ${i===history.idx?'current':''}" data-i="${i}">
      <span>${i===history.idx?'●':'○'}</span>${s.label}
    </div>`
  ).join('');
  list.querySelectorAll('.history-item').forEach(el => {
    el.onclick = () => applyState(+el.dataset.i);
  });
}

// ── LAYERS ──────────────────────────────────────────────────
const layers = [];
let activeLayer = null;
function addLayer(name='Layer', src=null) {
  const id = Date.now();
  layers.push({ id, name, src });
  activeLayer = id;
  renderLayers();
}
function removeLayer(id) {
  const i = layers.findIndex(l=>l.id===id);
  if (i>-1) layers.splice(i,1);
  if (activeLayer===id) activeLayer = layers.length ? layers[layers.length-1].id : null;
  renderLayers();
}
function renderLayers() {
  const list = document.getElementById('layer-list');
  if (!layers.length) { list.innerHTML = '<div style="font-size:11px;color:var(--text-faint);padding:4px">No layers</div>'; return; }
  list.innerHTML = [...layers].reverse().map(l =>
    `<div class="layer-item ${l.id===activeLayer?'active':''}" data-id="${l.id}">
      <div class="layer-thumb"><canvas width="28" height="20" id="thumb-${l.id}"></canvas></div>
      <span class="layer-name">${l.name}</span>
      <button class="layer-del" data-del="${l.id}">✕</button>
    </div>`
  ).join('');
  list.querySelectorAll('.layer-item').forEach(el => {
    if (!el.dataset.del) return;
    el.onclick = (e) => {
      if (e.target.dataset.del) { removeLayer(+e.target.dataset.del); return; }
      activeLayer = +el.dataset.id; renderLayers();
    };
  });
  list.querySelectorAll('[data-del]').forEach(btn => {
    btn.onclick = (e) => { e.stopPropagation(); removeLayer(+btn.dataset.del); };
  });
}

// ── BACKGROUND PRESETS ──────────────────────────────────────
const bgPresets = [
  { type:'solid', value:'#1a1a2e' },
  { type:'gradient', value:'linear-gradient(135deg,#6366f1,#a855f7,#ec4899)' },
  { type:'gradient', value:'linear-gradient(135deg,#0ea5e9,#06b6d4)' },
  { type:'gradient', value:'linear-gradient(135deg,#10b981,#059669)' },
  { type:'gradient', value:'linear-gradient(135deg,#f59e0b,#ef4444)' },
  { type:'gradient', value:'linear-gradient(135deg,#1e293b,#0f172a)' },
  { type:'gradient', value:'linear-gradient(135deg,#6d28d9,#1e1b4b)' },
  { type:'solid', value:'#f8fafc' },
];
let activeBg = 0;
function renderBgPresets() {
  const el = document.getElementById('bg-presets');
  el.innerHTML = bgPresets.map((p,i) =>
    `<div class="bg-preset ${i===activeBg?'active':''}" data-i="${i}" style="background:${p.value}"></div>`

  el.querySelectorAll('.bg-preset').forEach(d => {

  });

renderBgPresets();

function applyBackground() {
  if (!state.imageLoaded) return;
  // Redraw background on the canvas behind the existing drawing
  // We store the original image and redraw each time
  redrawCanvas();
}

// ── CANVAS SETUP ────────────────────────────────────────────
let originalImage = null;
let overlayCanvas = document.createElement('canvas'); // annotations layer
let overlayCtx = overlayCanvas.getContext('2d');

function loadImage(src) {
  const img = new Image();
  img.onload = () => {
    const pad = state.bgPadding;
    const cw = img.width + pad*2;
    const ch = img.height + pad*2;
    canvas.width = cw;
    canvas.height = ch;
    overlayCanvas.width = cw;
    overlayCanvas.height = ch;
    originalImage = img;
    state.imageLoaded = true;
    document.getElementById('drop-overlay').classList.add('hidden');
    document.getElementById('sb-size').textContent = `${cw} × ${ch}`;
    addLayer('Background', src);
    redrawCanvas();
    saveState('Load Image');
  };
  img.src = src;
}

function redrawCanvas() {
  const pad = state.bgPadding;
  ctx.save();
  ctx.clearRect(0,0,canvas.width,canvas.height);

  // Background
  const bp = bgPresets[activeBg];
  if (bp.type==='solid') {
    ctx.fillStyle = bp.value;
    ctx.fillRect(0,0,canvas.width,canvas.height);
  } else {
    // Parse gradient
    const grad = ctx.createLinearGradient(0,0,canvas.width,canvas.height);
    const stops = extractGradientStops(bp.value);
    stops.forEach((s,i) => grad.addColorStop(i/(stops.length-1||1), s));
    ctx.fillStyle = grad;
    ctx.fillRect(0,0,canvas.width,canvas.height);
  }

  if (originalImage) {
    // Clip rounded rect for image
    const r = state.bgCorner;
    ctx.save();
    roundedRect(ctx, pad, pad, originalImage.width, originalImage.height, r);
    ctx.clip();

    // Apply filters
    ctx.filter = `brightness(${state.brightness}%) contrast(${state.contrast}%) saturate(${state.saturation}%) blur(${state.adjBlur}px)`;

    // Flip/rotate
    const cx = pad + originalImage.width/2;
    const cy = pad + originalImage.height/2;
    ctx.translate(cx, cy);
    ctx.rotate(state.rotation * Math.PI/180);
    ctx.scale(state.flipH?-1:1, state.flipV?-1:1);
    ctx.drawImage(originalImage, -originalImage.width/2, -originalImage.height/2);
    ctx.restore();
  }

  // Draw overlay (annotations)
  ctx.drawImage(overlayCanvas, 0, 0);
  ctx.restore();
}

function extractGradientStops(str) {
  const m = str.match(/#[0-9a-fA-F]{6}|#[0-9a-fA-F]{3}|rgba?\([^)]+\)/g);
  return m || ['#000','#fff'];
}
function roundedRect(ctx, x, y, w, h, r) {
  ctx.beginPath();
  ctx.moveTo(x+r, y);
  ctx.lineTo(x+w-r, y);
  ctx.quadraticCurveTo(x+w, y, x+w, y+r);
  ctx.lineTo(x+w, y+h-r);
  ctx.quadraticCurveTo(x+w, y+h, x+w-r, y+h);
  ctx.lineTo(x+r, y+h);
  ctx.quadraticCurveTo(x, y+h, x, y+h-r);
  ctx.lineTo(x, y+r);
  ctx.quadraticCurveTo(x, y, x+r, y);
  ctx.closePath();
}

// ── DRAWING ──────────────────────────────────────────────────
function getPos(e) {
  const r = canvas.getBoundingClientRect();
  const scaleX = canvas.width / r.width;
  const scaleY = canvas.height / r.height;
  const touch = e.touches?.[0] || e;
  return {
    x: (touch.clientX - r.left) * scaleX,
    y: (touch.clientY - r.top) * scaleY
  };
}

let snapshotBeforeDraw = null;

canvas.addEventListener('mousedown', startDraw);
canvas.addEventListener('mousemove', duringDraw);
canvas.addEventListener('mouseup', endDraw);
canvas.addEventListener('mouseleave', endDraw);
canvas.addEventListener('touchstart', e => { e.preventDefault(); startDraw(e); }, {passive:false});
canvas.addEventListener('touchmove', e => { e.preventDefault(); duringDraw(e); }, {passive:false});
canvas.addEventListener('touchend', e => { e.preventDefault(); endDraw(e); }, {passive:false});

function startDraw(e) {
  if (state.tool==='select') return;
  const {x,y} = getPos(e);
  state.drawing = true;
  state.startX = x; state.startY = y;
  state.lastX = x; state.lastY = y;
  state.drawPoints = [{x,y}];

  if (state.tool==='text') { placeText(x,y); return; }
  if (state.tool==='eyedrop') { eyedrop(x,y); return; }
  if (state.tool==='crop') { snapshotBeforeDraw = ctx.getImageData(0,0,canvas.width,canvas.height); return; }

  snapshotBeforeDraw = ctx.getImageData(0,0,canvas.width,canvas.height);
  setupStroke(overlayCtx);
  overlayCtx.beginPath();
  overlayCtx.moveTo(x,y);
}

function duringDraw(e) {
  if (!state.drawing) return;
  const {x,y} = getPos(e);
  document.getElementById('sb-cursor').textContent = `${Math.round(x)}, ${Math.round(y)}`;

  if (state.tool==='crop') { drawCropGuide(state.startX, state.startY, x, y); return; }
  if (state.tool==='eyedrop') { return; }
  if (state.tool==='text') { return; }

  const snap = snapshotBeforeDraw;
  overlayCtx.clearRect(0,0,overlayCanvas.width,overlayCanvas.height);
  if (snap) { /* overlay is separate, no need to restore */ }

  setupStroke(overlayCtx);

  switch(state.tool) {
    case 'draw':
    case 'highlight':
      state.drawPoints.push({x,y});
      overlayCtx.beginPath();
      overlayCtx.moveTo(state.drawPoints[0].x, state.drawPoints[0].y);
      for (let i=1;i<state.drawPoints.length;i++) overlayCtx.lineTo(state.drawPoints[i].x, state.drawPoints[i].y);
      overlayCtx.stroke();
      break;
    case 'eraser':
      state.drawPoints.push({x,y});
      overlayCtx.globalCompositeOperation='destination-out';
      overlayCtx.beginPath();
      overlayCtx.moveTo(state.drawPoints[0].x, state.drawPoints[0].y);
      for (let i=1;i<state.drawPoints.length;i++) overlayCtx.lineTo(state.drawPoints[i].x, state.drawPoints[i].y);
      overlayCtx.stroke();
      overlayCtx.globalCompositeOperation='source-over';
      break;
    case 'line':
      overlayCtx.beginPath();
      overlayCtx.moveTo(state.startX, state.startY);
      overlayCtx.lineTo(x,y);
      overlayCtx.stroke();
      break;
    case 'arrow':
      drawArrow(overlayCtx, state.startX, state.startY, x, y);
      break;
    case 'rect':
      overlayCtx.beginPath();
      overlayCtx.rect(state.startX, state.startY, x-state.startX, y-state.startY);
      overlayCtx.stroke();
      break;
    case 'ellipse':
      overlayCtx.beginPath();
      overlayCtx.ellipse(
        (state.startX+x)/2, (state.startY+y)/2,
        Math.abs(x-state.startX)/2, Math.abs(y-state.startY)/2,
        0, 0, Math.PI*2
      );
      overlayCtx.stroke();
      break;
    case 'blur':
      applyBlurRegion(state.startX, state.startY, x-state.startX, y-state.startY);
      break;
  }
  redrawCanvas();
  state.lastX = x; state.lastY = y;
}

function endDraw(e) {
  if (!state.drawing) return;
  state.drawing = false;
  const {x,y} = e.changedTouches ? getPos(e) : getPos(e);

  if (state.tool==='crop') {
    performCrop(state.startX, state.startY, x, y);
    return;
  }
  if (state.tool==='eyedrop' || state.tool==='text') return;

  saveState(toolLabel(state.tool));
}

function toolLabel(t) {
  return { draw:'Draw', line:'Line', arrow:'Arrow', rect:'Rectangle', ellipse:'Ellipse',
           blur:'Blur', highlight:'Highlight', eraser:'Eraser', text:'Text', crop:'Crop', eyedrop:'Pick Color' }[t] || t;
}

function setupStroke(c) {
  c.lineWidth = state.strokeSize;
  c.strokeStyle = state.color;
  c.globalAlpha = state.opacity;
  c.lineCap = 'round';
  c.lineJoin = 'round';
  if (state.tool==='highlight') {
    c.globalAlpha = 0.35;
    c.lineWidth = state.strokeSize * 3;
  }
  if (state.strokeStyle==='dashed') c.setLineDash([state.strokeSize*4, state.strokeSize*2]);
  else if (state.strokeStyle==='dotted') c.setLineDash([2, state.strokeSize*2]);
  else c.setLineDash([]);
}

function drawArrow(c, x1,y1,x2,y2) {
  const hw = Math.max(state.strokeSize*3, 10);
  const angle = Math.atan2(y2-y1,x2-x1);
  c.beginPath();
  c.moveTo(x1,y1);
  c.lineTo(x2,y2);
  c.stroke();
  c.beginPath();
  c.moveTo(x2,y2);
  c.lineTo(x2-hw*Math.cos(angle-Math.PI/6), y2-hw*Math.sin(angle-Math.PI/6));
  c.lineTo(x2-hw*Math.cos(angle+Math.PI/6), y2-hw*Math.sin(angle+Math.PI/6));
  c.closePath();
  c.fillStyle = state.color;
  c.fill();
}

function placeText(x,y) {
  state.drawing = false;
  const txt = state.textContent || 'Text';
  const weight = state.bold ? 'bold ' : '';
  const style2 = state.italic ? 'italic ' : '';
  overlayCtx.font = `${style2}${weight}${state.fontSize}px "${state.textFont}"`;
  overlayCtx.fillStyle = state.color;
  overlayCtx.globalAlpha = state.opacity;
  overlayCtx.fillText(txt, x, y);
  redrawCanvas();
  saveState('Text');
}

function eyedrop(x,y) {
  state.drawing = false;
  const px = ctx.getImageData(Math.floor(x),Math.floor(y),1,1).data;
  const hex = '#'+[px[0],px[1],px[2]].map(v=>v.toString(16).padStart(2,'0')).join('');
  state.color = hex;
  document.getElementById('custom-color').value = hex;
  document.querySelectorAll('.color-swatch').forEach(s=>s.classList.remove('active'));
  notify('Color picked: '+hex, 'success');
}

// ── BLUR REGION ──────────────────────────────────────────────
function applyBlurRegion(bx,by,bw,bh) {
  if (!bw||!bh) return;
  const x=Math.min(bx,bx+bw), y=Math.min(by,by+bh);
  const w=Math.abs(bw), h=Math.abs(bh);
  overlayCtx.save();
  overlayCtx.filter='blur(12px)';
  overlayCtx.drawImage(canvas,x,y,w,h,x,y,w,h);
  overlayCtx.restore();
}

// ── CROP ─────────────────────────────────────────────────────
let cropGuide = { x:0,y:0,w:0,h:0, active:false };
function drawCropGuide(x1,y1,x2,y2) {
  cropGuide = { x:Math.min(x1,x2),y:Math.min(y1,y2),w:Math.abs(x2-x1),h:Math.abs(y2-y1),active:true };
  // draw overlay
  overlayCtx.clearRect(0,0,overlayCanvas.width,overlayCanvas.height);
  overlayCtx.fillStyle='rgba(0,0,0,0.45)';
  overlayCtx.fillRect(0,0,overlayCanvas.width,overlayCanvas.height);
  overlayCtx.clearRect(cropGuide.x,cropGuide.y,cropGuide.w,cropGuide.h);
  overlayCtx.strokeStyle='#fff';
  overlayCtx.lineWidth=1;
  overlayCtx.setLineDash([4,4]);
  overlayCtx.strokeRect(cropGuide.x,cropGuide.y,cropGuide.w,cropGuide.h);
  overlayCtx.setLineDash([]);
  redrawCanvas();
}
function performCrop(x1,y1,x2,y2) {
  const cx=Math.min(x1,x2),cy=Math.min(y1,y2),cw=Math.abs(x2-x1),ch=Math.abs(y2-y1);
  if (cw<4||ch<4) { overlayCtx.clearRect(0,0,overlayCanvas.width,overlayCanvas.height); redrawCanvas(); return; }
  const data = ctx.getImageData(cx,cy,cw,ch);
  canvas.width=cw; canvas.height=ch;
  overlayCanvas.width=cw; overlayCanvas.height=ch;
  ctx.putImageData(data,0,0);
  document.getElementById('sb-size').textContent=`${cw} × ${ch}`;
  state.imageLoaded = true;
  // Update original image to cropped version

// ── PASTE ────────────────────────────────────────────────────
async function pasteFromClipboard() {
  try {
    const items = await navigator.clipboard.read();
    for (const item of items) {
      for (const type of item.types) {
        if (type.startsWith('image/')) {
          const blob = await item.getType(type);
          readFile(blob); return;
        }
      }
    }
    notify('No image in clipboard', 'error');
  } catch(e) { notify('Clipboard access denied', 'error'); }
}
document.getElementById('btn-paste').onclick = pasteFromClipboard;

document.addEventListener('paste', e => {
  const items = e.clipboardData?.items;
  if (!items) return;
  for (const item of items) {
    if (item.type.startsWith('image/')) { readFile(item.getAsFile()); return; }
  }
});

// Download / export
function exportCanvas() {
  const fmt = document.getElementById('export-format').value;
  const quality = (state.exportQuality||92)/100;
  const scale = parseFloat(document.getElementById('export-scale').value)||1;
  const link = document.createElement('a');

  if (scale===1) {
    link.download = `snapedit.${fmt}`;
    link.href = canvas.toDataURL(fmt==='png'?'image/png':'image/'+fmt, quality);
    link.click();
  } else {
    const tmp = document.createElement('canvas');
    tmp.width=canvas.width*scale; tmp.height=canvas.height*scale;
    const tc=tmp.getContext('2d');
    tc.scale(scale,scale);
    tc.drawImage(canvas,0,0);
    link.download = `snapedit@${scale}x.${fmt}`;
    link.href = tmp.toDataURL(fmt==='png'?'image/png':'image/'+fmt, quality);
    link.click();
  }
  notify('Downloaded!','success');
}

document.addEventListener('keydown', e => {
  if (e.ctrlKey||e.metaKey) {
    if (e.key==='z') { e.preventDefault(); undo(); }
    if (e.key==='y') { e.preventDefault(); redo(); }
    if (e.key==='s') { e.preventDefault(); exportCanvas(); }
  }
  const toolKeys = { v:'select', c:'crop', d:'draw', l:'line', a:'arrow', r:'rect',
                     e:'ellipse', t:'text', b:'blur', h:'highlight', x:'eraser' };
  if (!e.ctrlKey && !e.metaKey && toolKeys[e.key]) {
    const btn = document.querySelector(`[data-tool="${toolKeys[e.key]}"]`);
    if (btn) btn.click();
  }
});
