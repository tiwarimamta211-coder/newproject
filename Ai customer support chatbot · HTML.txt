// ─── AI RESPONSE ENGINE ───
function getResponse(text, fileName) {
  if(fileName) {
    return {
      response: `Thanks for sharing <strong>${escapeHtml(fileName)}</strong>! 📎 I've received your file. Our team will review it and get back to you shortly. Is there anything specific you'd like to highlight about this file?`,
      qr: ["It's a screenshot of the issue","It's an invoice","It's my ID proof"],
      satisfy: false
    };
  }

  // Greeting
  if(KNOWLEDGE.greetings.some(g => text.includes(g) || text === g)) {
    const hour = new Date().getHours();
    const greeting = hour < 12 ? "Good morning" : hour < 17 ? "Good afternoon" : "Good evening";
    return {
      response: `${greeting}! 👋 Great to have you here. I'm Nova, your AI support assistant. What can I help you with today?`,
      qr: ["Check my order","Billing issue","Return a product","Technical support"],
      satisfy: false
    };
  }

  // Thanks
  if(KNOWLEDGE.thanks.keys.some(k => text.includes(k))) {
    return {response: pick(KNOWLEDGE.thanks.responses), qr: ["I have another question","No, I'm done"], satisfy: true};
  }

  // Bye
  if(KNOWLEDGE.bye.keys.some(k => text.includes(k))) {
    return {response: pick(KNOWLEDGE.bye.responses), qr: null, satisfy: true};
  }

  // Topic matching
  const topics = ['billing','orders','returns','account','tech','shipping'];
  for(const t of topics) {
    if(KNOWLEDGE[t].keys.some(k => text.includes(k))) {
      return {
        response: pick(KNOWLEDGE[t].responses),
        qr: KNOWLEDGE[t].quickReplies || null,
        satisfy: true
      };
    }
  }

  // Order ID pattern
  if(/\b[A-Z]{2,4}[-]?\d{5,10}\b/.test(text.toUpperCase()) || /order.{0,10}(number|id|no)/i.test(text)) {
    return {
      response: `I see you're referencing an order. Let me look that up for you! 🔍<br><br>For the fastest results, please share your <strong>Order ID</strong> (format: ORD-XXXXX) and I'll check the status immediately.`,
      qr: ["Check delivery status","Cancel this order","Change delivery address"],
      satisfy: false
    };
  }

  // Fallback
  return {
    response: `I want to make sure I give you the best help possible! 🤔 Could you clarify a bit more? You can also pick a topic below:`,
    qr: ["Billing & Payments","Order Tracking","Returns & Refunds","Technical Support","Talk to a human agent"],
    satisfy: false
  };
}

function pick(arr) { return arr[Math.floor(Math.random()*arr.length)]; }

// ─── TAGS ───
function updateTags(text) {
  const map = {billing:["Billing","Payments"],orders:["Orders","Tracking","Shipping"],returns:["Returns","Refunds"],account:["Account","Security"],tech:["Technical","Bug"]};
  const cloud = document.getElementById('tagCloud');
  const existing = [...cloud.querySelectorAll('.tag')].map(t => t.textContent);
  for(const [key, tags] of Object.entries(map)) {
    if(KNOWLEDGE[key]?.keys.some(k => text.includes(k))) {
      tags.forEach(t => {
        if(!existing.includes(t)) {
          const span = document.createElement('span');
          span.className = 'tag';
          span.textContent = t;
          cloud.appendChild(span);
          existing.push(t);
        }
      });
    }
  }
}

// ─── EXPORT ───
function exportChat() {
  if(chatHistory.length === 0) { showToast("⚠️ No messages to export yet"); return; }
  let txt = `NovaSup AI — Chat Transcript\nDate: ${new Date().toLocaleString('en-IN')}\n${'─'.repeat(40)}\n\n`;
  chatHistory.forEach(m => { txt += `[${m.role.toUpperCase()}]: ${m.text.replace(/<[^>]+>/g,'')}\n\n`; });
  txt += `${'─'.repeat(40)}\nEnd of transcript`;
  const blob = new Blob([txt], {type:'text/plain'});
  const a = document.createElement('a');
  a.href = URL.createObjectURL(blob);
  a.download = `novasup-chat-${Date.now()}.txt`;
  a.click();
  showToast("✅ Chat exported successfully!");
}

}

}

function updateChatPreview(text) {
  const clean = text.replace(/<[^>]+>/g,'').substring(0,40);
  const chat = allChats.find(c => c.id === currentChatId);
  if(chat) { chat.preview = clean; renderChatList(); }
}

function renderChatList() {
  const list = document.getElementById('chatList');
  const icons = ["💬","📦","💳","🔑","🛠️","🚚"];
  list.innerHTML = allChats.map((c,i) => `
    <div class="chat-item ${c.id===currentChatId?'active':''}" onclick="switchChat(${c.id})">
      <div class="chat-item-icon">${icons[i%icons.length]}</div>
      <div class="chat-item-text">
        <div class="chat-item-title">${escapeHtml(c.title)}</div>
        <div class="chat-item-preview">${escapeHtml(c.preview)}</div>
      </div>
    </div>`).join('');
}

function switchChat(id) {
  currentChatId = id;
  renderChatList();
  if(window.innerWidth <= 900) toggleSidebar();
}

// ─── SIDEBAR TOGGLE ───
function toggleSidebar() {
  sidebarOpen = !sidebarOpen;
  document.getElementById('sidebar').classList.toggle('open', sidebarOpen);
}

// ─── TOAST ───
let toastTimer;
function showToast(msg) {
  const old = document.querySelector('.toast');
  if(old) old.remove();
  clearTimeout(toastTimer);
  const t = document.createElement('div');
  t.className = 'toast';
  t.innerHTML = msg;
  document.body.appendChild(t);
  toastTimer = setTimeout(() => t.remove(), 3500);
}
