<script>
// =========================================================
//  DATA STORE
// =========================================================
let entries = [];
let isBlocked = false;
let nextId = 1;

// =========================================================
//  TAB SWITCHING
// =========================================================
function switchTab(name) {
  document.querySelectorAll('.tab-btn').forEach((b,i) => {
    b.classList.toggle('active', ['entry','itc','comparison','export'][i] === name);
  });
  document.querySelectorAll('.panel').forEach(p => p.classList.remove('active'));
  document.getElementById('panel-' + name).classList.add('active');
  refreshAll();
}

// =========================================================
//  TOGGLE BLOCKED
// =========================================================
function toggleBlocked() {
  isBlocked = !isBlocked;
  document.getElementById('blockedToggle').classList.toggle('on', isBlocked);
  if (isBlocked) {
    document.getElementById('itcElig').value = 'blocked';
    notify('⚠️ Marked as blocked credit (Sec 17(5))');
  }
}

// =========================================================
//  AUTO CALCULATE TAX
// =========================================================
function autoCalcTax() {
  const val = parseFloat(document.getElementById('taxVal').value) || 0;
  const rateEl = document.getElementById('gstRate');
  const rate = rateEl.value === 'custom' ? 0 : parseFloat(rateEl.value) || 0;
  const type = document.getElementById('supplyType').value;

  const totalTax = (val * rate) / 100;

  if (type === 'inter') {
    document.getElementById('igst').value = totalTax.toFixed(2);
    document.getElementById('cgst').value = '0.00';
    document.getElementById('sgst').value = '0.00';
  } else {
    document.getElementById('igst').value = '0.00';
    document.getElementById('cgst').value = (totalTax / 2).toFixed(2);
    document.getElementById('sgst').value = (totalTax / 2).toFixed(2);
  }
}

// =========================================================
//  ADD ENTRY
// =========================================================
function addEntry() {
  const supName = document.getElementById('supName').value.trim();
  const supGSTIN = document.getElementById('supGSTIN').value.trim().toUpperCase();
  const invNo = document.getElementById('invNo').value.trim();
  const invDate = document.getElementById('invDate').value.trim();
  const desc = document.getElementById('desc').value.trim();
  const taxVal = parseFloat(document.getElementById('taxVal').value) || 0;
  const igst = parseFloat(document.getElementById('igst').value) || 0;
  const cgst = parseFloat(document.getElementById('cgst').value) || 0;
  const sgst = parseFloat(document.getElementById('sgst').value) || 0;
  const cess = parseFloat(document.getElementById('cess').value) || 0;
  const elig = document.getElementById('itcElig').value;
  const eligPct = Math.min(100, Math.max(0, parseFloat(document.getElementById('eligPct').value) || 100));

  if (!supName || taxVal === 0) {
    notify('❗ Supplier name and taxable value are required.', 'warn');
    return;
  }

  const totalTax = igst + cgst + sgst + cess;
  let itcEligible = 0;

  if (elig === 'full') itcEligible = totalTax;
  else if (elig === 'partial') itcEligible = (totalTax * eligPct) / 100;
  else itcEligible = 0;

  const entry = {
    id: nextId++,
    supName, supGSTIN, invNo, invDate, desc,
    taxVal, igst, cgst, sgst, cess,
    totalTax, elig, eligPct, itcEligible,
    gstRate: document.getElementById('gstRate').value,
    supplyType: document.getElementById('supplyType').value
  };

  entries.push(entry);
  renderTable();
  refreshAll();
  clearForm();
  notify('✅ Entry added successfully');
}

// =========================================================
//  RENDER TABLE
// =========================================================
function renderTable() {
  const body = document.getElementById('entriesBody');
  document.getElementById('entryCount').textContent = entries.length + ' ' + (entries.length === 1 ? 'entry' : 'entries');

  if (entries.length === 0) {
    body.innerHTML = `<tr><td colspan="13"><div class="empty-state"><div class="empty-icon">📭</div><p>No entries yet.</p></div></td></tr>`;
    return;
  }

  body.innerHTML = entries.map((e, i) => {
    const statusChip = e.elig === 'full'
      ? '<span class="chip chip-green">✓ Eligible</span>'
      : e.elig === 'partial'
      ? `<span class="chip chip-yellow">⚡ ${e.eligPct}%</span>`
      : '<span class="chip chip-red">✗ Blocked</span>';

    return `<tr>
      <td>${i + 1}</td>
      <td style="font-family:'DM Sans',sans-serif">${e.supName}<br><small style="color:var(--text-dim)">${e.supGSTIN || '—'}</small></td>
      <td>${e.invNo || '—'}</td>
      <td>${e.invDate || '—'}</td>
      <td style="font-family:'DM Sans',sans-serif; max-width:160px; overflow:hidden; text-overflow:ellipsis;">${e.desc || '—'}</td>
      <td>${fmt(e.taxVal)}</td>
      <td style="color:var(--igst)">${fmt(e.igst)}</td>
      <td style="color:var(--cgst)">${fmt(e.cgst)}</td>
      <td style="color:var(--sgst)">${fmt(e.sgst)}</td>
      <td style="color:var(--cess)">${fmt(e.cess)}</td>
      <td style="color:var(--accent); font-weight:600">${fmt(e.itcEligible)}</td>
      <td>${statusChip}</td>
      <td class="td-actions">
        <button class="icon-btn del" onclick="deleteEntry(${e.id})" title="Delete">🗑</button>
      </td>
    </tr>`;
  }).join('');
}

// =========================================================
//  DELETE ENTRY
// =========================================================
function deleteEntry(id) {
  entries = entries.filter(e => e.id !== id);
  renderTable();
  refreshAll();
  notify('🗑 Entry removed');
}

// =========================================================
//  REFRESH ALL
// =========================================================
function refreshAll() {
  refreshITC();
  refreshComparison();
  refreshSummaryPreview();
}

// =========================================================
//  ITC SUMMARY
// =========================================================
function refreshITC() {
  const eligible = entries.filter(e => e.elig !== 'blocked');
  const blocked = entries.filter(e => e.elig === 'blocked');

  const totalITC = entries.reduce((s,e) => s + e.totalTax, 0);
  const eligITC = entries.reduce((s,e) => s + e.itcEligible, 0);
  const blockedITC = blocked.reduce((s,e) => s + e.totalTax, 0);
  const totalTaxable = entries.reduce((s,e) => s + e.taxVal, 0);
  const opening = parseFloat(document.getElementById('openingITC').value) || 0;

  const igstTotal = eligible.reduce((s,e) => s + e.igst * (e.eligPct/100), 0);
  const cgstTotal = eligible.reduce((s,e) => s + e.cgst * (e.eligPct/100), 0);
  const sgstTotal = eligible.reduce((s,e) => s + e.sgst * (e.eligPct/100), 0);
  const cessTotal = eligible.reduce((s,e) => s + e.cess * (e.eligPct/100), 0);

  setEl('s-totalITC', '₹' + fmt(totalITC));
  setEl('s-eligITC', '₹' + fmt(eligITC));

// =========================================================
//  COMPARISON
// =========================================================
function refreshComparison() {
  const outIGST = parseFloat(document.getElementById('outIGST').value) || 0;
  const outCGST = parseFloat(document.getElementById('outCGST').value) || 0;
  const outSGST = parseFloat(document.getElementById('outSGST').value) || 0;
  const opening = parseFloat(document.getElementById('openingITC').value) || 0;
  const totalOut = outIGST + outCGST + outSGST;

  const eligible = entries.filter(e => e.elig !== 'blocked');
  const inIGST = eligible.reduce((s,e) => s + e.igst * (e.elig === 'partial' ? e.eligPct/100 : 1), 0);
  const inCGST = eligible.reduce((s,e) => s + e.cgst * (e.elig === 'partial' ? e.eligPct/100 : 1), 0);
  const inSGST = eligible.reduce((s,e) => s + e.sgst * (e.elig === 'partial' ? e.eligPct/100 : 1), 0);
  const inCESS = eligible.reduce((s,e) => s + e.cess * (e.elig === 'partial' ? e.eligPct/100 : 1), 0);
  const totalIn = inIGST + inCGST + inSGST + inCESS + opening;

  // Update comparison cards
  setEl('c-outIGST', '₹' + fmt(outIGST));
  setEl('c-outCGST', '₹' + fmt(outCGST));
  setEl('c-outSGST', '₹' + fmt(outSGST));
  setEl('c-totalOut', '₹' + fmt(totalOut));
  setEl('c-inIGST', '₹' + fmt(inIGST));
  setEl('c-inCGST', '₹' + fmt(inCGST));
  setEl('c-inSGST', '₹' + fmt(inSGST));
  setEl('c-inCESS', '₹' + fmt(inCESS));
  setEl('c-totalIn', '₹' + fmt(totalIn));

  // Net position
  const net = totalOut - totalIn;
  setEl('np-out', '₹' + fmt(totalOut));
  setEl('np-itc', '₹' + fmt(totalIn - opening));
  setEl('np-opening', '₹' + fmt(opening));
  setEl('np-net', '₹' + fmt(Math.abs(net)));

  // Banner
  const banner = document.getElementById('netResultBanner');
  if (totalOut === 0 && totalIn === 0) {
    banner.innerHTML = `<div class="result-banner neutral"><span class="result-icon">ℹ️</span><div>Enter output tax liability and purchase entries to see your net tax position.</div></div>`;
  } else if (net > 0) {
    banner.innerHTML = `<div class="result-banner negative"><span class="result-icon">💸</span><div>Net Tax Payable: <span class="result-val">₹${fmt(net)}</span><br><small style="font-size:12px; opacity:0.8">ITC is insufficient. Cash payment required.</small></div></div>`;
  } else if (net < 0) {
    banner.innerHTML = `<div class="result-banner positive"><span class="result-icon">🎉</span><div>ITC Surplus: <span class="result-val">₹${fmt(Math.abs(net))}</span><br><small style="font-size:12px; opacity:0.8">Carry forward to next period.</small></div></div>`;
  } else {
    banner.innerHTML = `<div class="result-banner neutral"><span class="result-icon">⚖️</span><div>Perfectly balanced. No cash payment needed.</div></div>`;
  }

  // Progress bar
  let pct = totalOut > 0 ? Math.min(100, (Math.min(totalIn, totalOut) / totalOut) * 100) : 0;
  document.getElementById('utilPct').textContent = pct.toFixed(1) + '%';
  document.getElementById('utilBar').style.width = pct + '%';

  // Head-wise table
  const headwiseData = [
    { head: 'IGST', out: outIGST, itc: inIGST },
    { head: 'CGST', out: outCGST, itc: inCGST },
    { head: 'SGST/UTGST', out: outSGST, itc: inSGST },
    { head: 'CESS', out: 0, itc: inCESS },
  ];

  const hwBody = document.getElementById('headwiseBody');
  hwBody.innerHTML = headwiseData.map(r => {
    const netR = r.out - r.itc;
    const chip = netR > 0
      ? `<span class="chip chip-red">Pay ₹${fmt(netR)}</span>`
      : netR < 0
      ? `<span class="chip chip-green">Surplus ₹${fmt(Math.abs(netR))}</span>`
      : `<span class="chip chip-yellow">Balanced</span>`;
    return `<tr>
      <td style="font-family:'DM Sans',sans-serif; font-weight:600">${r.head}</td>
      <td>${fmt(r.out)}</td>
      <td>${fmt(r.itc)}</td>
      <td style="font-weight:700">${fmt(Math.abs(netR))}</td>
      <td>${chip}</td>
    </tr>`;
  }).join('');
}

// =========================================================
//  ITC UTILISATION
// =========================================================
function calcUtilisation() {
  const eligible = entries.filter(e => e.elig !== 'blocked');
  const availIGST = eligible.reduce((s,e) => s + e.igst * (e.elig==='partial'?e.eligPct/100:1), 0);
  const availCGST = eligible.reduce((s,e) => s + e.cgst * (e.elig==='partial'?e.eligPct/100:1), 0);
  const availSGST = eligible.reduce((s,e) => s + e.sgst * (e.elig==='partial'?e.eligPct/100:1), 0);
  const availCESS = eligible.reduce((s,e) => s + e.cess * (e.elig==='partial'?e.eligPct/100:1), 0);

  const uII = parseFloat(document.getElementById('utilIGST_IGST').value) || 0;
  const uIC = parseFloat(document.getElementById('utilIGST_CGST').value) || 0;
  const uIS = parseFloat(document.getElementById('utilIGST_SGST').value) || 0;
  const uCC = parseFloat(document.getElementById('utilCGST_CGST').value) || 0;
  const uSS = parseFloat(document.getElementById('utilSGST_SGST').value) || 0;
  const uCs = parseFloat(document.getElementById('utilCESS').value) || 0;

  const totalUsed = uII + uIC + uIS + uCC + uSS + uCs;
  const totalAvail = availIGST + availCGST + availSGST + availCESS;
  const remaining = totalAvail - totalUsed;

  const el = document.getElementById('utilisationResult');
  el.innerHTML = `
    <div class="stats-row">
      <div class="stat-card"><div class="stat-label">Total ITC Available</div><div class="stat-value" style="color:var(--accent)">₹${fmt(totalAvail)}</div></div>
      <div class="stat-card"><div class="stat-label">Total ITC Utilised</div><div class="stat-value" style="color:var(--accent2)">₹${fmt(totalUsed)}</div></div>
      <div class="stat-card ${remaining<0?'orange':'accent'}"><div class="stat-label">Balance / (Excess Used)</div><div class="stat-value">₹${fmt(Math.abs(remaining))}</div><div class="stat-sub">${remaining<0?'⚠️ Over-utilised':'Available balance'}</div></div>
    </div>`;
}

function exportCSV() {
  if (entries.length === 0) { notify('⚠️ No entries to export', 'warn'); return; }
  const headers = ['#','Supplier','GSTIN','Invoice No','Date','Description','Taxable Value','IGST','CGST','SGST','CESS','Total Tax','ITC Eligible','Eligibility'];
  const rows = entries.map((e,i) => [
    i+1, `"${e.supName}"`, e.supGSTIN, e.invNo, e.invDate, `"${e.desc}"`,
    e.taxVal, e.igst, e.cgst, e.sgst, e.cess, e.totalTax, e.itcEligible, e.elig
  ]);
  const csv = [headers.join(','), ...rows.map(r => r.join(','))].join('\n');
  download('GST_ITC_Register.csv', csv, 'text/csv');
  notify('📊 CSV exported successfully');
}

function fmt(n) { return Number(n).toLocaleString('en-IN', {minimumFractionDigits:2, maximumFractionDigits:2}); }
function setEl(id, val) { const el = document.getElementById(id); if(el) el.textContent = val; }
function download(name, data, type) {
  const a = document.createElement('a');
  a.href = URL.createObjectURL(new Blob([data], {type}));
  a.download = name; a.click();
}

let notifTimer;
function notify(msg, type='success') {
  const el = document.getElementById('notification');
  const icon = document.getElementById('notifIcon');
  const text = document.getElementById('notifText');
  text.textContent = msg;
  icon.textContent = type === 'warn' ? '⚠️' : '✅';
  el.style.borderColor = type === 'warn' ? 'var(--warning)' : 'var(--accent)';
  el.classList.add('show');
  clearTimeout(notifTimer);
  notifTimer = setTimeout(() => el.classList.remove('show'), 3000);
}
