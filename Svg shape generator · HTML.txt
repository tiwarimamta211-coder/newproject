// ── STATE ─────────────────────────────────────────────────────────────────
let currentShape = 'blob';

function syncVal(id, valId) {
  document.getElementById(valId).textContent = document.getElementById(id).value;
}

function v(id) { return parseFloat(document.getElementById(id).value); }

function seededRand(seed) {
  let x = Math.sin(seed) * 10000;
  return x - Math.floor(x);
}

function makeRand(seed) {
  let s = seed;
  return () => { s += 0.618; return seededRand(s); };
}

// ── SHAPE FUNCTIONS ───────────────────────────────────────────────────────
function generateBlob(cx, cy, rx, ry, points, roundness, randomness, rand) {
  const n = Math.round(points);
  const rFactor = randomness / 100;
  const smooth = roundness / 100;
  const pts = [];

  for (let i = 0; i < n; i++) {
    const angle = (i / n) * Math.PI * 2 - Math.PI / 2;
    const r = 1 + (rand() - 0.5) * 2 * rFactor * 0.6;
    const px = cx + Math.cos(angle) * rx * r * (v('scaleX') / 100);
    const py = cy + Math.sin(angle) * ry * r * (v('scaleY') / 100);
    pts.push({ x: px, y: py });
  }

  if (smooth < 0.05) {
    return 'M' + pts.map(p => `${p.x.toFixed(2)},${p.y.toFixed(2)}`).join(' L') + ' Z';
  }

  // Smooth bezier through points
  let d = '';
  const len = pts.length;
  for (let i = 0; i < len; i++) {
    const p0 = pts[(i - 1 + len) % len];
    const p1 = pts[i];
    const p2 = pts[(i + 1) % len];
    const p3 = pts[(i + 2) % len];

    const t = smooth * 0.5;
    const cp1x = p1.x + (p2.x - p0.x) * t;
    const cp1y = p1.y + (p2.y - p0.y) * t;
    const cp2x = p2.x - (p3.x - p1.x) * t;
    const cp2y = p2.y - (p3.y - p1.y) * t;

    if (i === 0) d += `M ${p1.x.toFixed(2)},${p1.y.toFixed(2)} `;
    d += `C ${cp1x.toFixed(2)},${cp1y.toFixed(2)} ${cp2x.toFixed(2)},${cp2y.toFixed(2)} ${p2.x.toFixed(2)},${p2.y.toFixed(2)} `;
  }
  return d + 'Z';
}

function generateWave() {
  const amp = v('waveAmp');
  const freq = v('waveFreq');
  const fillH = v('waveFill') / 100;
  const smooth = v('waveSmooth') / 100;
  const rand = makeRand(currentSeed);
  const randomness = v('randomness') / 100;
  const W = svgW, H = svgH;
  const baseY = H * (1 - fillH);
  const steps = Math.round(freq * 8);
  const pts = [];

  for (let i = 0; i <= steps; i++) {
    const x = (i / steps) * W;
    const wave = Math.sin((i / steps) * Math.PI * 2 * freq) * amp;
    const noise = (rand() - 0.5) * amp * randomness * 1.5;
    pts.push({ x, y: baseY + wave + noise });
  }

  let d = `M 0,${H} L ${pts[0].x.toFixed(2)},${pts[0].y.toFixed(2)} `;
  if (smooth > 0.1) {
    for (let i = 0; i < pts.length - 1; i++) {
      const cp1x = pts[i].x + (pts[i + 1].x - pts[i].x) * smooth * 0.5;
      const cp1y = pts[i].y;
      const cp2x = pts[i + 1].x - (pts[i + 1].x - pts[i].x) * smooth * 0.5;
      const cp2y = pts[i + 1].y;
      d += `C ${cp1x.toFixed(2)},${cp1y.toFixed(2)} ${cp2x.toFixed(2)},${cp2y.toFixed(2)} ${pts[i + 1].x.toFixed(2)},${pts[i + 1].y.toFixed(2)} `;
    }
  } else {
    pts.forEach(p => d += `L ${p.x.toFixed(2)},${p.y.toFixed(2)} `);
  }
  d += `L ${W},${H} Z`;
  return d;
}

function generateSpiky(cx, cy, r, spikes, depth, randomness, rand) {
  const n = Math.round(spikes);
  const rFactor = randomness / 100;
  let d = '';
  for (let i = 0; i < n * 2; i++) {
    const angle = (i / (n * 2)) * Math.PI * 2 - Math.PI / 2;
    const isOuter = i % 2 === 0;
    const baseR = isOuter ? r : r * (1 - depth / 100);
    const noise = (rand() - 0.5) * rFactor * r * 0.4;
    const pr = baseR + noise;
    const px = cx + Math.cos(angle) * pr * (v('scaleX') / 100);
    const py = cy + Math.sin(angle) * pr * (v('scaleY') / 100);
    d += (i === 0 ? 'M ' : 'L ') + `${px.toFixed(2)},${py.toFixed(2)} `;
  }
  return d + 'Z';
}

function generatePolygon(cx, cy, r, sides, rotation, randomness, rand) {
  const n = Math.round(sides);
  const rFactor = randomness / 100;
  const rotRad = (rotation * Math.PI) / 180;
  const pts = [];
  for (let i = 0; i < n; i++) {
    const angle = (i / n) * Math.PI * 2 + rotRad - Math.PI / 2;
    const noise = (rand() - 0.5) * rFactor * r * 0.5;
    const pr = r + noise;
    pts.push({
      x: cx + Math.cos(angle) * pr * (v('scaleX') / 100),
      y: cy + Math.sin(angle) * pr * (v('scaleY') / 100)
    });
  }

  const cornerR = v('polyRadius');
  if (cornerR < 1) {
    return 'M ' + pts.map(p => `${p.x.toFixed(2)},${p.y.toFixed(2)}`).join(' L ') + ' Z';
  }

  // Rounded corners
  let d = '';
  for (let i = 0; i < n; i++) {
    const prev = pts[(i - 1 + n) % n];
    const curr = pts[i];
    const next = pts[(i + 1) % n];
    const dist = Math.hypot(next.x - curr.x, next.y - curr.y);
    const t = Math.min(cornerR / dist, 0.45);
    const p1 = { x: curr.x + (prev.x - curr.x) * t, y: curr.y + (prev.y - curr.y) * t };
    const p2 = { x: curr.x + (next.x - curr.x) * t, y: curr.y + (next.y - curr.y) * t };
    if (i === 0) d += `M ${p1.x.toFixed(2)},${p1.y.toFixed(2)} `;
    d += `L ${p1.x.toFixed(2)},${p1.y.toFixed(2)} Q ${curr.x.toFixed(2)},${curr.y.toFixed(2)} ${p2.x.toFixed(2)},${p2.y.toFixed(2)} `;
  }
  return d + 'Z';
}

function generateRing(cx, cy, outerR, innerRatio, segs, randomness, rand) {
  const rFactor = randomness / 100;
  const innerR = outerR * (innerRatio / 100);
  const segments = Math.round(segs);

  if (segments < 1) {
    // Smooth ring using path with inner cutout
    const noiseO = (rand() - 0.5) * rFactor * outerR * 0.15;
    return `M ${cx + outerR + noiseO},${cy} A ${outerR},${outerR} 0 1 0 ${cx - outerR},${cy} A ${outerR},${outerR} 0 1 0 ${cx + outerR},${cy} Z M ${cx + innerR},${cy} A ${innerR},${innerR} 0 1 1 ${cx - innerR},${cy} A ${innerR},${innerR} 0 1 1 ${cx + innerR},${cy} Z`;
  }

  let d = '';
  for (let s = 0; s < segments; s++) {
    const a1 = (s / segments) * Math.PI * 2;
    const a2 = ((s + 0.82) / segments) * Math.PI * 2;
    const x1o = cx + Math.cos(a1) * outerR, y1o = cy + Math.sin(a1) * outerR;
    const x2o = cx + Math.cos(a2) * outerR, y2o = cy + Math.sin(a2) * outerR;
    const x1i = cx + Math.cos(a2) * innerR, y1i = cy + Math.sin(a2) * innerR;
    const x2i = cx + Math.cos(a1) * innerR, y2i = cy + Math.sin(a1) * innerR;
    const large = a2 - a1 > Math.PI ? 1 : 0;
    d += `M ${x1o.toFixed(2)},${y1o.toFixed(2)} A ${outerR},${outerR} 0 ${large} 1 ${x2o.toFixed(2)},${y2o.toFixed(2)} L ${x1i.toFixed(2)},${y1i.toFixed(2)} A ${innerR},${innerR} 0 ${large} 0 ${x2i.toFixed(2)},${y2i.toFixed(2)} Z `;
  }
  return d;
}

function generateStarburst(cx, cy, outerR, innerRatio, points, roundness, randomness, rand) {
  const n = Math.round(points);
  const innerR = outerR * (innerRatio / 100);
  const rFactor = randomness / 100;
  const round = roundness / 100;

  const pts = [];
  for (let i = 0; i < n * 2; i++) {
    const angle = (i / (n * 2)) * Math.PI * 2 - Math.PI / 2;
    const isOuter = i % 2 === 0;
    const base = isOuter ? outerR : innerR;
    const noise = (rand() - 0.5) * rFactor * outerR * 0.3;
    pts.push({
      x: cx + Math.cos(angle) * (base + noise) * (v('scaleX') / 100),
      y: cy + Math.sin(angle) * (base + noise) * (v('scaleY') / 100)
    });
  }

  if (round < 0.05) {
    return 'M ' + pts.map(p => `${p.x.toFixed(2)},${p.y.toFixed(2)}`).join(' L ') + ' Z';
  }

  let d = '';
  const len = pts.length;
  for (let i = 0; i < len; i++) {
    const p0 = pts[(i - 1 + len) % len];
    const p1 = pts[i];
    const p2 = pts[(i + 1) % len];
    const p3 = pts[(i + 2) % len];
    const t = round * 0.35;
    const cp1x = p1.x + (p2.x - p0.x) * t;
    const cp1y = p1.y + (p2.y - p0.y) * t;
    const cp2x = p2.x - (p3.x - p1.x) * t;
    const cp2y = p2.y - (p3.y - p1.y) * t;
    if (i === 0) d += `M ${p1.x.toFixed(2)},${p1.y.toFixed(2)} `;
    d += `C ${cp1x.toFixed(2)},${cp1y.toFixed(2)} ${cp2x.toFixed(2)},${cp2y.toFixed(2)} ${p2.x.toFixed(2)},${p2.y.toFixed(2)} `;
  }
  return d + 'Z';
}

// ── MAIN GENERATE ─────────────────────────────────────────────────────────
function generate() {
  const rand = makeRand(currentSeed * 100 + Math.random() * 0.0001);
  // Use stable rand per seed
  const stableRand = makeRand(currentSeed * 137.508);

  const cx = svgW / 2, cy = svgH / 2;
  const rx = Math.min(svgW, svgH) * 0.38 * (v('scaleX') / 100);
  const ry = Math.min(svgW, svgH) * 0.38 * (v('scaleY') / 100);
  const r = Math.min(svgW, svgH) * 0.36;
  const randomness = v('randomness');

  let d = '';
  switch (currentShape) {
    case 'blob':
      d = generateBlob(cx, cy, rx, ry, v('blobPoints'), v('blobRound'), randomness, stableRand);
      break;
    case 'wave':
      d = generateWave();
      break;
    case 'spiky':
      d = generateSpiky(cx, cy, r, v('spikyCount'), v('spikyDepth'), randomness, stableRand);
      break;
    case 'polygon':
      d = generatePolygon(cx, cy, r, v('polySides'), v('polyRot'), randomness, stableRand);
      break;
    case 'ring':
      d = generateRing(cx, cy, r, v('ringInner'), v('ringSegs'), randomness, stableRand);
      break;
    case 'starburst':
      d = generateStarburst(cx, cy, r, v('starInner'), v('starPoints'), v('starRound'), randomness, stableRand);
      break;
  }

  lastPath = d;
  document.getElementById('mainShape').setAttribute('d', d);
  updateAppearance();
  updateCode();
}

  const rad = (angle * Math.PI) / 180;
  const x1 = (50 - Math.cos(rad) * 50).toFixed(1), y1 = (50 - Math.sin(rad) * 50).toFixed(1);
  const x2 = (50 + Math.cos(rad) * 50).toFixed(1), y2 = (50 + Math.sin(rad) * 50).toFixed(1);

  let defsInner = '';
  if (gradType === 'linear') {
    defsInner = `\n    <linearGradient id="grad" x1="${x1}%" y1="${y1}%" x2="${x2}%" y2="${y2}%">\n      <stop offset="0%" stop-color="${c1}"/>\n      <stop offset="100%" stop-color="${c2}"/>\n    </linearGradient>`;
  } else if (gradType === 'radial') {
    defsInner = `\n    <radialGradient id="grad" cx="50%" cy="50%" r="50%">\n      <stop offset="0%" stop-color="${c1}"/>\n      <stop offset="100%" stop-color="${c2}"/>\n    </radialGradient>`;
  }
  if (glowOn) {
    defsInner += `\n    <filter id="glow">\n      <feGaussianBlur stdDeviation="8" result="blur"/>\n      <feMerge><feMergeNode in="blur"/><feMergeNode in="SourceGraphic"/></feMerge>\n    </filter>`;
  }

  const fillAttr = gradType === 'solid' ? `fill="${c1}"` : `fill="url(#grad)"`;
  const defsStr = defsInner ? `  <defs>${defsInner}\n  </defs>\n` : '';
  const bgStr = bgOn ? `  <rect width="${svgW}" height="${svgH}" fill="${bgFill}"/>\n` : '';

  return `<svg width="${svgW}" height="${svgH}" viewBox="0 0 ${svgW} ${svgH}" xmlns="http://www.w3.org/2000/svg">\n${defsStr}${bgStr}  <path\n    d="${lastPath}"\n    ${fillAttr}\n    opacity="${opacity}"${strokeStr}${filterStr}\n  />\n</svg>`;
}

function buildCSSCode() {
  const svg = buildSVGCode();
  const encoded = encodeURIComponent(svg);
  return `.element {\n  background-image: url("data:image/svg+xml,${encoded}");\n  background-size: cover;\n  background-repeat: no-repeat;\n}`;
}

function buildJSXCode() {
  const c1 = document.getElementById('color1').value;
  const c2 = document.getElementById('color2').value;
  const opacity = v('shapeOpacity') / 100;
  const fillAttr = gradType === 'solid' ? `fill="${c1}"` : `fill="url(#grad)"`;
  return `export function Shape({ className }) {\n  return (\n    <svg\n      width="${svgW}"\n      height="${svgH}"\n      viewBox="0 0 ${svgW} ${svgH}"\n      className={className}\n      xmlns="http://www.w3.org/2000/svg"\n    >\n      <defs>\n        <linearGradient id="grad" x1="0%" y1="0%" x2="100%" y2="100%">\n          <stop offset="0%" stopColor="${c1}" />\n          <stop offset="100%" stopColor="${c2}" />\n        </linearGradient>\n      </defs>\n      <path\n        d="${lastPath}"\n        ${fillAttr}\n        opacity={${opacity}}\n      />\n    </svg>\n  );\n}`;
}

function updateCode() {
  let code = '';
  if (codeTab === 'svg') code = buildSVGCode();
  else if (codeTab === 'css') code = buildCSSCode();
  else code = buildJSXCode();
  document.getElementById('codeBlock').textContent = code;
}

function copyCode() {
  const text = document.getElementById('codeBlock').textContent;
  navigator.clipboard.writeText(text).then(() => {
    const toast = document.getElementById('copyToast');
    toast.classList.add('show');
    setTimeout(() => toast.classList.remove('show'), 2000);
  });
}

function downloadSVG() {
  const svg = buildSVGCode();
  const blob = new Blob([svg], { type: 'image/svg+xml' });
  const url = URL.createObjectURL(blob);
  const a = document.createElement('a');
  a.href = url; a.download = `morphe-${currentShape}-${Date.now()}.svg`;
  a.click(); URL.revokeObjectURL(url);
}

function setShape(shape, btn) {
  currentShape = shape;
  document.querySelectorAll('.shape-tab').forEach(b => b.classList.remove('active'));
  btn.classList.add('active');
  ['blob','wave','spiky','polygon','ring','starburst'].forEach(s => {
    const el = document.getElementById('ctrl-' + s);
    if (el) el.style.display = s === shape ? '' : 'none';
  });
  generate();
}
