        <button class="action-btn" onclick="resetAll()">
          🔄 Reset
        </button>
        <button class="action-btn" onclick="printPage()">
          🖨️ Print
        </button>
      </div>

    </div>
  </div>

  <!-- History -->
  <div class="card history-section">
    <div class="card-title">Calculation History</div>
    <div id="history-list">
      <div class="no-history">No calculations yet. Results will appear here.</div>
    </div>
    <button class="clear-history-btn" onclick="clearHistory()">✕ Clear History</button>
  </div>

    Consult a tax professional for official filings
  </div>

</div>

<script>
  // ── State ──────────────────────────────────────────────────────
  let mode = 'exclusive';      // exclusive | inclusive
  let txType = 'intra';        // intra | inter
  let selectedRate = 5;        // numeric GST %
  let history = JSON.parse(localStorage.getItem('gst_history') || '[]');

  // ── Setters ────────────────────────────────────────────────────
  function setMode(m) {
    mode = m;
    document.getElementById('btn-exclusive').classList.toggle('active', m === 'exclusive');
    document.getElementById('btn-inclusive').classList.toggle('active', m === 'inclusive');
    autoCalculate();
  }

  function setRate(r) {
    const customInput = document.getElementById('custom-rate-input');
    if (r === 'custom') {
      customInput.style.display = 'block';
      selectedRate = parseFloat(customInput.value) || 0;
    } else {
      customInput.style.display = 'none';
      selectedRate = r;
    }
    document.querySelectorAll('.rate-btn').forEach(b => b.classList.remove('active'));
    if (r !== 'custom') {
      const rates = [0, 5, 12, 18, 28, 40];
      const idx = rates.indexOf(r);
      if (idx !== -1) document.querySelectorAll('.rate-btn')[idx].classList.add('active');
    } else {
      document.getElementById('custom-rate-btn').classList.add('active');
    }
    autoCalculate();
  }

  function setTxType(t) {
    txType = t;
    document.getElementById('btn-intra').classList.toggle('active', t === 'intra');
    document.getElementById('btn-inter').classList.toggle('active', t === 'inter');
    autoCalculate();
  }

  // ── Core Calculation ───────────────────────────────────────────
  function getRate() {
    const customInput = document.getElementById('custom-rate-input');
    if (customInput.style.display !== 'none') {
      return parseFloat(customInput.value) || 0;
    }
    return selectedRate;
  }

  function compute(amount, rate, mode) {
    let base, gst, total;
    if (mode === 'exclusive') {
      base  = amount;
      gst   = base * (rate / 100);
      total = base + gst;
    } else {
      total = amount;
      base  = total / (1 + rate / 100);
      gst   = total - base;
    }
    return { base, gst, total };
  }

  function fmt(n) {
    return '₹' + n.toLocaleString('en-IN', { minimumFractionDigits: 2, maximumFractionDigits: 2 });
  }

  function calculate() {
    const amtVal = parseFloat(document.getElementById('amount').value);
    if (isNaN(amtVal) || amtVal < 0) {
      alert('Please enter a valid positive amount.');
      return;
    }

    const rate = getRate();
    const { base, gst, total } = compute(amtVal, rate, mode);

    // GST Split
    let cgst = 0, sgst = 0, igst = 0;
    if (txType === 'intra') {
      cgst = gst / 2;
      sgst = gst / 2;
    } else {
      igst = gst;
    }

    // Render
    document.getElementById('total-label').textContent =
      mode === 'exclusive' ? 'Total Amount Payable (incl. GST)' : 'Base Amount (excl. GST)';
    document.getElementById('total-amount').textContent =
      mode === 'exclusive' ? fmt(total) : fmt(base);

    document.getElementById('base-amount').textContent  = fmt(base);
    document.getElementById('gst-amount').textContent   = fmt(gst);
    document.getElementById('gst-rate-display').textContent = `@ ${rate}% GST`;

    // Split blocks
    const cgstBlock = document.getElementById('cgst-block');
    const sgstBlock = document.getElementById('sgst-block');
    const igstBlock = document.getElementById('igst-block');

    if (txType === 'intra') {
      cgstBlock.style.display = ''; sgstBlock.style.display = '';
      igstBlock.style.display = 'none';
      document.getElementById('cgst-val').textContent = fmt(cgst);
      document.getElementById('sgst-val').textContent = fmt(sgst);
      document.getElementById('cgst-pct').textContent = `${rate/2}%`;
      document.getElementById('sgst-pct').textContent = `${rate/2}%`;
      igstBlock.style.gridColumn = '';
    } else {
      cgstBlock.style.display = 'none'; sgstBlock.style.display = 'none';
      igstBlock.style.display = '';
      igstBlock.style.gridColumn = '1 / -1';
      document.getElementById('igst-val').textContent = fmt(igst);
      document.getElementById('igst-pct').textContent = `${rate}%`;
    }

    // Meta
    const now = new Date();
    document.getElementById('results-meta').textContent =
      `${txType === 'intra' ? 'Intra-State' : 'Inter-State'} · ${mode === 'exclusive' ? 'Exclusive' : 'Inclusive'} · ${now.toLocaleTimeString('en-IN')}`;

    // Show results
    const card = document.getElementById('results-card');
    card.classList.add('visible');
    card.scrollIntoView({ behavior: 'smooth', block: 'nearest' });

    // Add to history
    addHistory({ amtVal, rate, mode, txType, base, gst, total, cgst, sgst, igst, time: now.toLocaleString('en-IN') });
  }

  function autoCalculate() {
    const v = document.getElementById('amount').value;
    if (v && parseFloat(v) >= 0) calculate();
  }

  // ── History ────────────────────────────────────────────────────
  function addHistory(entry) {
    history.unshift(entry);
    if (history.length > 10) history.pop();
    localStorage.setItem('gst_history', JSON.stringify(history));
    renderHistory();
  }

  function renderHistory() {
    const el = document.getElementById('history-list');
    if (!history.length) {
      el.innerHTML = '<div class="no-history">No calculations yet. Results will appear here.</div>';
      return;
    }
    el.innerHTML = history.map((h, i) => `
      <div class="history-item" onclick="loadHistory(${i})">
        <span class="hi-base">₹${h.amtVal.toLocaleString('en-IN')}</span>
        <span class="hi-rate">${h.rate}% GST</span>
        <span class="hi-type">${h.txType === 'intra' ? 'Intra' : 'Inter'}</span>
        <span style="color:var(--muted);font-size:11px;">${h.time}</span>
        <span class="hi-total">Total: ${fmt(h.total)}</span>
      </div>
    `).join('');
  }

  function loadHistory(i) {
    const h = history[i];
    document.getElementById('amount').value = h.amtVal;
    setMode(h.mode);
    setTxType(h.txType);
    setRate(h.rate);
    calculate();
  }

  function clearHistory() {
    history = [];
    localStorage.removeItem('gst_history');
    renderHistory();
  }

  // ── PDF Download ───────────────────────────────────────────────
  function downloadPDF() {
    const { jsPDF } = window.jspdf;
    const doc = new jsPDF({ unit: 'mm', format: 'a4' });

    const amtVal = parseFloat(document.getElementById('amount').value) || 0;
    const rate   = getRate();
    const { base, gst, total } = compute(amtVal, rate, mode);
    const notes = document.getElementById('notes').value;
    const now   = new Date().toLocaleString('en-IN');

    const W = doc.internal.pageSize.getWidth();

    // Header bar
    doc.setFillColor(240, 165, 0);
    doc.rect(0, 0, W, 28, 'F');

    doc.setTextColor(13, 17, 23);
    doc.setFont('helvetica', 'bold');
    doc.setFontSize(22);
    doc.text('GST CALCULATOR', 14, 14);
    doc.setFontSize(10);
    doc.setFont('helvetica', 'normal');
    doc.text('India Tax Computation Summary', 14, 22);

    doc.setFont('helvetica', 'normal');
    doc.setFontSize(9);
    doc.setTextColor(13, 17, 23);
    doc.text(`Generated: ${now}`, W - 14, 14, { align: 'right' });

    // Body
    doc.setTextColor(30, 30, 30);
    let y = 40;

    function row(label, value, bold = false, color = [30,30,30]) {
      doc.setFont('helvetica', bold ? 'bold' : 'normal');
      doc.setFontSize(10);
      doc.setTextColor(...color);
      doc.text(label, 14, y);
      doc.text(value, W - 14, y, { align: 'right' });
      y += 8;
    }

    function section(title) {
      y += 4;
      doc.setFillColor(245, 245, 240);
      doc.rect(14, y - 5, W - 28, 8, 'F');
      doc.setFont('helvetica', 'bold');
      doc.setFontSize(9);
      doc.setTextColor(120, 120, 120);
      doc.text(title.toUpperCase(), 16, y);
      y += 8;
    }

    section('Transaction Details');
    row('Mode', mode === 'exclusive' ? 'GST Exclusive' : 'GST Inclusive');
    row('Transaction Type', txType === 'intra' ? 'Intra-State (CGST + SGST)' : 'Inter-State (IGST)');
    row('GST Rate', `${rate}%`);
    if (notes) row('Notes', notes);

    section('Amount Breakdown');
    row('Base Amount (excl. GST)', fmt(base));
    row('Total GST Amount', fmt(gst), false, [200, 120, 0]);

    if (txType === 'intra') {
      row(`  CGST (${rate/2}%)`, fmt(gst/2));
      row(`  SGST (${rate/2}%)`, fmt(gst/2));
    } else {
      row(`  IGST (${rate}%)`, fmt(gst));
    }

    y += 4;
    doc.setFillColor(240, 165, 0);
    doc.rect(14, y, W - 28, 0.5, 'F');
    y += 8;

    row('TOTAL AMOUNT (incl. GST)', fmt(total), true, [13, 17, 23]);

    // Footer
    const pageH = doc.internal.pageSize.getHeight();
    doc.setFillColor(240, 165, 0);
    doc.rect(0, pageH - 14, W, 14, 'F');
    doc.setFont('helvetica', 'normal');
    doc.setFontSize(8);
    doc.setTextColor(13, 17, 23);
    doc.text('This document is for reference only. Consult a tax professional for official filings.', W/2, pageH - 6, { align: 'center' });

    doc.save('GST-Calculation.pdf');
  }

  // ── Copy Summary ───────────────────────────────────────────────
  function copyResults() {
    const amtVal = parseFloat(document.getElementById('amount').value) || 0;
    const rate   = getRate();
    const { base, gst, total } = compute(amtVal, rate, mode);
    const lines = [
      `GST Calculation Summary`,
      `─────────────────────`,
      `Mode: ${mode === 'exclusive' ? 'GST Exclusive' : 'GST Inclusive'}`,
      `Type: ${txType === 'intra' ? 'Intra-State' : 'Inter-State'}`,
      `Rate: ${rate}%`,
      `Base Amount : ${fmt(base)}`,
      `GST Amount  : ${fmt(gst)}`,
      txType === 'intra' ? `  CGST (${rate/2}%): ${fmt(gst/2)}\n  SGST (${rate/2}%): ${fmt(gst/2)}` : `  IGST (${rate}%): ${fmt(gst)}`,
      `Total Amount: ${fmt(total)}`,
    ].join('\n');
    navigator.clipboard.writeText(lines).then(() => alert('✅ Copied to clipboard!'));
  }

  function printPage() { window.print(); }

  function resetAll() {
    document.getElementById('amount').value = '';
    document.getElementById('notes').value  = '';
    document.getElementById('results-card').classList.remove('visible');
    setMode('exclusive');
    setTxType('intra');
    setRate(5);
  }

  // ── Init ───────────────────────────────────────────────────────
  renderHistory();
  setRate(5);
</script>
