// ---- LOAD IMAGE ----
function loadImage(file) {
  if (file.size > 20 * 1024 * 1024) { showNotification('File too large (max 20MB)', 'error'); return; }
  if (!file.type.startsWith('image/')) { showNotification('Please upload an image file', 'error'); return; }

  const reader = new FileReader();
  reader.onload = e => {
    const img = new Image();
    img.onload = () => {
      originalWidth = img.width;
      originalHeight = img.height;
      originalImageData = img;

      document.getElementById('originalImg').src = e.target.result;
      document.getElementById('statusFileName').textContent = file.name.length > 20 ? file.name.substring(0, 18) + '…' : file.name;
      document.getElementById('statusDims').textContent = `${img.width} × ${img.height}px`;
      document.getElementById('statusSize').textContent = formatBytes(file.size);

      uploadZone.style.display = 'none';
      processingArea.classList.add('active');

      processImage();
    };
    img.src = e.target.result;
  };
  reader.readAsDataURL(file);
}

    if (progress > 100) progress = 100;
    document.getElementById('progressFill').style.width = progress + '%';

    if (step < steps.length && progress >= (step + 1) * (100 / steps.length)) {
      document.getElementById('processingText').textContent = steps[step++];
    }

    if (progress >= 100) {
      clearInterval(interval);
      setTimeout(() => finishProcessing(), 300);
    }
  }, 120);
}

function finishProcessing() {
  const img = originalImageData;
  const maxW = 600, maxH = 500;
  let w = img.width, h = img.height;
  if (w > maxW) { h = Math.round(h * maxW / w); w = maxW; }
  if (h > maxH) { w = Math.round(w * maxH / h); h = maxH; }

  resultCanvas.width = w;
  resultCanvas.height = h;
  ctx.clearRect(0, 0, w, h);

  const tmpCanvas = document.createElement('canvas');
  tmpCanvas.width = w; tmpCanvas.height = h;
  const tmpCtx = tmpCanvas.getContext('2d');
  tmpCtx.drawImage(img, 0, 0, w, h);
  const imgData = tmpCtx.getImageData(0, 0, w, h);
  const data = imgData.data;

  // Simulate background removal via flood-fill approximation
  // Sample corner colors as "background" heuristic
  const corners = [
    [0, 0], [w-1, 0], [0, h-1], [w-1, h-1],
    [Math.floor(w/2), 0], [0, Math.floor(h/2)], [w-1, Math.floor(h/2)], [Math.floor(w/2), h-1]
  ];

  const bgColors = corners.map(([x, y]) => {
    const i = (y * w + x) * 4;
    return [data[i], data[i+1], data[i+2]];
  });

  function colorDist(r1, g1, b1, r2, g2, b2) {
    return Math.sqrt((r1-r2)**2 + (g1-g2)**2 + (b1-b2)**2);
  }

  function isBg(r, g, b) {
    return bgColors.some(([br, bg, bb]) => colorDist(r, g, b, br, bg, bb) < 55);
  }

  maskData = new Uint8Array(w * h);
  for (let i = 0; i < w * h; i++) {
    const r = data[i*4], g = data[i*4+1], b = data[i*4+2];
    // More nuanced: check if near center (likely foreground) vs edges
    const px = i % w, py = Math.floor(i / w);
    const distFromEdge = Math.min(px, w-1-px, py, h-1-py) / (Math.min(w,h) * 0.5);
    const bgScore = bgColors.reduce((s, [br, bg, bb]) => s + (colorDist(r,g,b,br,bg,bb) < 60 ? 1 : 0), 0);
    maskData[i] = (bgScore >= 4 && distFromEdge < 0.35) ? 0 : 1;
  }

  // Smooth mask edges
  const smoothedMask = smoothMask(maskData, w, h, 2);
  applyMaskToCanvas(smoothedMask, imgData, w, h);

  saveToHistory();

  document.getElementById('processingOverlay').style.display = 'none';
  document.getElementById('resultBadge').textContent = 'DONE';
  document.getElementById('resultBadge').style.color = 'var(--success)';
  document.getElementById('processBtn').disabled = false;
  document.getElementById('downloadBtn').disabled = false;

  addToSessionHistory();
  setupBrushEvents();
  showNotification('Background removed successfully! ✓', 'success');
}

function smoothMask(mask, w, h, radius) {
  const result = new Float32Array(w * h);
  for (let y = 0; y < h; y++) {
    for (let x = 0; x < w; x++) {
      let sum = 0, cnt = 0;
      for (let dy = -radius; dy <= radius; dy++) {
        for (let dx = -radius; dx <= radius; dx++) {
          const nx = x+dx, ny = y+dy;
          if (nx >= 0 && nx < w && ny >= 0 && ny < h) {
            sum += mask[ny*w+nx]; cnt++;
          }
        }
      }
      result[y*w+x] = sum / cnt;
    }
  }
  return result;
}

function applyMaskToCanvas(mask, imgData, w, h) {
  const result = ctx.createImageData(w, h);
  const src = imgData.data;
  const feather = parseInt(document.getElementById('featherRadius').value);

  for (let i = 0; i < w * h; i++) {
    const alpha = Math.min(1, Math.max(0, mask[i]));
    result.data[i*4]   = src[i*4];
    result.data[i*4+1] = src[i*4+1];
    result.data[i*4+2] = src[i*4+2];
    result.data[i*4+3] = Math.round(alpha * 255);
  }

  const tmpCanvas = document.createElement('canvas');
  tmpCanvas.width = w; tmpCanvas.height = h;
  const tmpCtx = tmpCanvas.getContext('2d');
  tmpCtx.putImageData(result, 0, 0);

  // Apply background
  ctx.clearRect(0, 0, w, h);
  if (currentBg !== 'transparent' || currentBgGradient || currentBgImage) {
    if (currentBgImage) {
      ctx.drawImage(currentBgImage, 0, 0, w, h);
    } else if (currentBgGradient) {
      const grad = parseGradient(currentBgGradient, w, h);
      ctx.fillStyle = grad;
      ctx.fillRect(0, 0, w, h);
    } else {
      ctx.fillStyle = currentBg;
      ctx.fillRect(0, 0, w, h);
    }
  }
  ctx.drawImage(tmpCanvas, 0, 0);

  processedCanvas = tmpCanvas;
}

function parseGradient(gradStr, w, h) {
  try {
    const match = gradStr.match(/linear-gradient\((.*?),(.*?),(.*?)\)/s);
    if (!match) return '#222';
    const anglePart = match[1].trim();
    const angle = anglePart === '135deg' ? 135 : parseInt(anglePart) || 135;
    const rad = (angle - 90) * Math.PI / 180;
    const x1 = w/2 + Math.cos(rad + Math.PI) * w, y1 = h/2 + Math.sin(rad + Math.PI) * h;
    const x2 = w/2 + Math.cos(rad) * w, y2 = h/2 + Math.sin(rad) * h;
    const g = ctx.createLinearGradient(x1, y1, x2, y2);
    g.addColorStop(0, match[2].trim());
    g.addColorStop(1, match[3].trim());
    return g;
  } catch { return '#222'; }
}

function setupBrushEvents() {
  resultCanvas.addEventListener('mousedown', startBrush);
  resultCanvas.addEventListener('mousemove', doBrush);
  resultCanvas.addEventListener('mouseup', () => { isDrawing = false; if (brushMode) saveToHistory(); });
  resultCanvas.addEventListener('touchstart', e => { e.preventDefault(); startBrush(e.touches[0]); }, {passive:false});
  resultCanvas.addEventListener('touchmove', e => { e.preventDefault(); doBrush(e.touches[0]); }, {passive:false});
  resultCanvas.addEventListener('touchend', () => { isDrawing = false; if (brushMode) saveToHistory(); });
}

function getCanvasPos(e) {
  const rect = resultCanvas.getBoundingClientRect();
  const scaleX = resultCanvas.width / rect.width;
  const scaleY = resultCanvas.height / rect.height;
  return { x: (e.clientX - rect.left) * scaleX, y: (e.clientY - rect.top) * scaleY };
}

function startBrush(e) { if (!brushMode) return; isDrawing = true; paintBrush(e); }
function doBrush(e) { if (!isDrawing || !brushMode) return; paintBrush(e); }

function paintBrush(e) {
  const pos = getCanvasPos(e);
  const r = brushSize;
  const w = resultCanvas.width, h = resultCanvas.height;
  const imgData = ctx.getImageData(0, 0, w, h);
  const data = imgData.data;

  for (let dy = -r; dy <= r; dy++) {
    for (let dx = -r; dx <= r; dx++) {
      const dist = Math.sqrt(dx*dx + dy*dy);
      if (dist > r) continue;
      const falloff = 1 - (dist / r);
      const px = Math.round(pos.x + dx), py = Math.round(pos.y + dy);
      if (px < 0 || px >= w || py < 0 || py >= h) continue;
      const idx = (py * w + px) * 4;
      if (brushMode === 'erase') {
        data[idx+3] = Math.max(0, data[idx+3] - Math.round(falloff * 180));
      } else {
        data[idx+3] = Math.min(255, data[idx+3] + Math.round(falloff * 180));
      }
    }
  }
  ctx.putImageData(imgData, 0, 0);
}

function reRender() {
  if (!processedCanvas) return;
  const w = resultCanvas.width, h = resultCanvas.height;
  ctx.clearRect(0, 0, w, h);
  if (currentBgImage) {
    ctx.drawImage(currentBgImage, 0, 0, w, h);
  } else if (currentBgGradient) {
    const grad = parseGradient(currentBgGradient, w, h);
    ctx.fillStyle = grad;
    ctx.fillRect(0, 0, w, h);
  } else if (currentBg !== 'transparent') {
    ctx.fillStyle = currentBg;
    ctx.fillRect(0, 0, w, h);
  }
  ctx.drawImage(processedCanvas, 0, 0);
}

function flipImage(dir) {
  if (dir === 'h') flipH = !flipH; else flipV = !flipV;
  const w = resultCanvas.width, h = resultCanvas.height;
  const tmp = ctx.getImageData(0, 0, w, h);
  const data = tmp.data;
  const flipped = new Uint8ClampedArray(data.length);
  for (let y = 0; y < h; y++) {
    for (let x = 0; x < w; x++) {
      const sx = dir === 'h' ? w-1-x : x;
      const sy = dir === 'v' ? h-1-y : y;
      const si = (sy*w+sx)*4, di = (y*w+x)*4;
      flipped[di]=data[si]; flipped[di+1]=data[si+1]; flipped[di+2]=data[si+2]; flipped[di+3]=data[si+3];
    }
  }
  const imgD = new ImageData(flipped, w, h);
  ctx.putImageData(imgD, 0, 0);
  saveToHistory();
  showNotification(`Image flipped ${dir === 'h' ? 'horizontally' : 'vertically'}`, 'info');
}

function rotateImage() {
  const w = resultCanvas.width, h = resultCanvas.height;
  const tmp = document.createElement('canvas');
  tmp.width = h; tmp.height = w;
  const tCtx = tmp.getContext('2d');
  tCtx.translate(h/2, w/2);
  tCtx.rotate(Math.PI/2);
  tCtx.drawImage(resultCanvas, -w/2, -h/2);
  resultCanvas.width = h; resultCanvas.height = w;
  ctx.drawImage(tmp, 0, 0);
  saveToHistory();
  showNotification('Rotated 90°', 'info');
}

// ---- HISTORY ----
function saveToHistory() {
  const data = resultCanvas.toDataURL();
  history = history.slice(0, historyIndex + 1);
  history.push(data);
  if (history.length > 20) history.shift();
  historyIndex = history.length - 1;
}

function undoLast() {
  if (historyIndex <= 0) { showNotification('Nothing to undo', 'info'); return; }
  historyIndex--;
  restoreFromHistory(history[historyIndex]);
}

function redoLast() {
  if (historyIndex >= history.length - 1) { showNotification('Nothing to redo', 'info'); return; }
  historyIndex++;
  restoreFromHistory(history[historyIndex]);
}

function restoreFromHistory(dataUrl) {
  const img = new Image();
  img.onload = () => { ctx.clearRect(0, 0, resultCanvas.width, resultCanvas.height); ctx.drawImage(img, 0, 0); };
  img.src = dataUrl;
}

  document.querySelectorAll('.history-thumb').forEach(t => t.classList.remove('active'));
  thumb.classList.add('active');
}
