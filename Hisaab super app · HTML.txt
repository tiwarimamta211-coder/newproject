function renderDashboard() {
  const inc = txns.filter(t=>t.type==='income').reduce((a,t)=>a+t.amount,0);
  const exp = txns.filter(t=>t.type==='expense').reduce((a,t)=>a+t.amount,0);
  const bal = inc - exp;
  document.getElementById('d-income').textContent = fmt(inc);
  document.getElementById('d-expense').textContent = fmt(exp);
  const balEl = document.getElementById('d-balance');
  balEl.textContent = fmt(bal);
  balEl.className = 'stat-val '+(bal>=0?'green':'red');
  document.getElementById('d-gst').textContent = fmt(gstRisk);
  document.getElementById('chip-balance').textContent = 'Balance: '+fmt(bal);
  document.getElementById('gst-chip').textContent = 'GST Risk: '+fmt(gstRisk);

  // Chart
  const months = ['Oct','Nov','Dec','Jan','Feb','Mar'];
  const now = new Date();
  const data = months.map((_,i) => {
    const d = new Date(now.getFullYear(), now.getMonth()-5+i, 1);
    const key = `${d.getFullYear()}-${String(d.getMonth()+1).padStart(2,'0')}`;
    const mi = txns.filter(t=>t.type==='income'&&t.date.startsWith(key)).reduce((a,t)=>a+t.amount,0);
    const me = txns.filter(t=>t.type==='expense'&&t.date.startsWith(key)).reduce((a,t)=>a+t.amount,0);
    return {m:months[i],mi,me};
  });
  const max = Math.max(...data.flatMap(d=>[d.mi,d.me]),1);
  document.getElementById('dash-chart').innerHTML = data.map(d=>`
    <div class="bar-col">
      <div style="display:flex;gap:3px;align-items:flex-end;height:70px;">
        <div class="bar-fill inc" style="height:${(d.mi/max*65)}px;flex:1;"></div>
        <div class="bar-fill exp" style="height:${(d.me/max*65)}px;flex:1;"></div>
      </div>
      <div class="bar-lbl">${d.m}</div>
    </div>`).join('');

  // Recent txns
  const icons = {Sales:'🛒',Service:'⚙️',Salary:'👤',Rent:'🏠',Utilities:'⚡',Purchase:'📦',Marketing:'📢',Other:'📁'};
  const recent = txns.slice(0,6);
  document.getElementById('d-txns').innerHTML = recent.length ? recent.map(t=>`
    <div class="txn-row">
      <div class="txn-ico ${t.type}">${icons[t.cat]||'📁'}</div>
      <div style="flex:1;">
        <div class="txn-name">${t.desc}</div>
        <div class="txn-meta">${t.date} · <span class="tag">${t.cat}</span></div>
      </div>
      <div class="txn-amt ${t.type}">${t.type==='income'?'+':'-'}${fmt(t.amount)}</div>
    </div>`).join('') : '<div style="color:var(--muted);text-align:center;padding:20px;font-size:13px;">कोई transaction नहीं</div>';
}

// ═══════════════════════════════════════════════════════
function renderReports() {
  ['income','expense'].forEach(type => {
    const cats = {};
    txns.filter(t=>t.type===type).forEach(t => { cats[t.cat]=(cats[t.cat]||0)+t.amount; });
    const total = Object.values(cats).reduce((a,v)=>a+v,0)||1;
    const id = type==='income'?'rpt-income':'rpt-expense';
    const col = type==='income'?'var(--accent)':'var(--red)';
    document.getElementById(id).innerHTML = Object.entries(cats).sort((a,b)=>b[1]-a[1]).map(([cat,val])=>`
      <div style="padding:8px 0;border-bottom:1px solid var(--border);">
        <div style="display:flex;justify-content:space-between;font-size:12px;margin-bottom:4px;">
          <span style="color:var(--muted2);">${cat}</span>
          <span style="font-weight:700;font-family:'JetBrains Mono',monospace;">${fmt(val)}</span>
        </div>
        <div style="height:3px;background:var(--border);border-radius:2px;">
          <div style="height:100%;width:${(val/total*100).toFixed(0)}%;background:${col};border-radius:2px;"></div>
        </div>
      </div>`).join('') || '<div style="color:var(--muted);font-size:12px;padding:10px 0;">No data</div>';
  });

  // Monthly table
  const months = {};
  txns.forEach(t => {
    const mo = t.date.substring(0,7);
    if (!months[mo]) months[mo]={inc:0,exp:0};
    months[mo][t.type==='income'?'inc':'exp'] += t.amount;
  });
  document.getElementById('monthly-body').innerHTML = Object.entries(months).sort((a,b)=>b[0].localeCompare(a[0])).map(([mo,d])=>`
    <tr>
      <td>${mo}</td>
      <td style="color:var(--accent);">+${fmt(d.inc)}</td>
      <td style="color:var(--red);">-${fmt(d.exp)}</td>
      <td style="color:${d.inc-d.exp>=0?'var(--green)':'var(--red)'};">${fmt(d.inc-d.exp)}</td>
    </tr>`).join('') || '<tr><td colspan="4" style="color:var(--muted);text-align:center;">No data</td></tr>';
}

function genXML() {
  const type=document.getElementById('ve-type').value;
  const party=document.getElementById('ve-party').value||'Sample Party';
  const amt=document.getElementById('ve-amt').value||'10000';
  const nar=document.getElementById('ve-nar').value||'Entry';
  const date=document.getElementById('ve-date').value||today;
  const gst=document.getElementById('ve-gst').value;
  const gstA=(amt*gst/100).toFixed(2);
  const td=date.replace(/-/g,'').substring(0,8);
  genXMLStr=`<ENVELOPE>
  <HEADER><TALLYREQUEST>Import Data</TALLYREQUEST></HEADER>
  <BODY><IMPORTDATA>
    <REQUESTDESC><REPORTNAME>Vouchers</REPORTNAME></REQUESTDESC>
    <REQUESTDATA><TALLYMESSAGE xmlns:UDF="TallyUDF">
      <VOUCHER VCHTYPE="${type}" ACTION="Create">
        <DATE>${td}</DATE>
        <VOUCHERTYPENAME>${type}</VOUCHERTYPENAME>
        <NARRATION>${nar}</NARRATION>
        <PARTYLEDGERNAME>${party}</PARTYLEDGERNAME>
        <ALLLEDGERENTRIES.LIST>
          <LEDGERNAME>${party}</LEDGERNAME>
          <ISDEEMEDPOSITIVE>Yes</ISDEEMEDPOSITIVE>
          <AMOUNT>-${amt}</AMOUNT>
        </ALLLEDGERENTRIES.LIST>
        <ALLLEDGERENTRIES.LIST>
          <LEDGERNAME>${type} Account</LEDGERNAME>
          <ISDEEMEDPOSITIVE>No</ISDEEMEDPOSITIVE>
          <AMOUNT>${amt}</AMOUNT>
        </ALLLEDGERENTRIES.LIST>${gst>0?`
        <ALLLEDGERENTRIES.LIST>
          <LEDGERNAME>Output GST ${gst}%</LEDGERNAME>
          <ISDEEMEDPOSITIVE>No</ISDEEMEDPOSITIVE>
          <AMOUNT>${gstA}</AMOUNT>
        </ALLLEDGERENTRIES.LIST>`:''}
      </VOUCHER>
    </TALLYMESSAGE></REQUESTDATA>
  </IMPORTDATA></BODY>
</ENVELOPE>`;
  const col=genXMLStr.replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;')
    .replace(/(&lt;\/?[\w.:]+)(\s|&gt;)/g,'<span class="xt">$1</span>$2')
    .replace(/"([^"]*)"/g,'"<span class="xv">$1</span>"');
  document.getElementById('xml-out').innerHTML=col;
  document.getElementById('xml-card').style.display='block';
  log('info',`XML generated: ${type} — ${fmt(amt)}`);
  toast('✅ XML Generated!');
}

function calcGST() {
  const days=parseInt(document.getElementById('g-days').value)||0;
  const rtype=document.getElementById('g-rtype').value;
  const unpaid=parseFloat(document.getElementById('g-unpaid').value)||0;
  const itc=parseFloat(document.getElementById('g-itc').value)||0;
  const eway=parseFloat(document.getElementById('g-eway').value)||0;

  const perDay=rtype==='nil'?20:50;
  const lateFee=Math.min(days*perDay,10000);
  const interest=(unpaid*0.18*days)/365;
  const itcPen=itc>0?Math.max(itc*0.10,10000):0;
  const ewayPen=eway>0?Math.max(eway*0.01,10000):0;
  const total=lateFee+interest+itcPen+ewayPen;

  gstRisk=total;
  document.getElementById('gb-late').textContent=fmt(lateFee);
  document.getElementById('gb-int').textContent=fmt(interest);
  document.getElementById('gb-itc').textContent=fmt(itcPen);
  document.getElementById('gb-eway').textContent=fmt(ewayPen);
  document.getElementById('gb-total').textContent=fmt(total);
  document.getElementById('g-total').textContent=fmt(total);

  const tagEl=document.getElementById('g-tag');
  const subEl=document.getElementById('g-sub');
  const totEl=document.getElementById('g-total');
  const chipEl=document.getElementById('gst-chip');
  chipEl.textContent='GST Risk: '+fmt(total);

  if(total>50000){
    tagEl.textContent='🚨 CRITICAL RISK: High Financial Liability';
    tagEl.style.color='var(--red)'; totEl.style.color='var(--red)';
    subEl.textContent='Immediate compliance action required!';
    chipEl.className='chip red';
    document.getElementById('gst-badge').style.display='inline-block';
  } else if(total>5000){
    tagEl.textContent='⚠️ MODERATE RISK: Compliance Required';
    tagEl.style.color='var(--yellow)'; totEl.style.color='var(--yellow)';
    subEl.textContent='कृपया जल्द GST filing करें।';
    chipEl.className='chip yellow';
    document.getElementById('gst-badge').style.display='inline-block';
  } else if(total>0){
    tagEl.textContent='✅ LOW RISK: Standard Late Fees';
    tagEl.style.color='var(--green)'; totEl.style.color='var(--green)';
    subEl.textContent='Normal compliance range.';
    chipEl.className='chip green';
    document.getElementById('gst-badge').style.display='none';
  } else {
    tagEl.textContent='✅ No Risk Detected'; tagEl.style.color='var(--green)';
    totEl.style.color='var(--accent)'; subEl.textContent='सभी values भरें to calculate';
    chipEl.className='chip'; document.getElementById('gst-badge').style.display='none';
  }
  document.getElementById('d-gst').textContent=fmt(total);
}
