<script>
  var credits   = 850;
  var imageB64  = null;
  var imageMime = 'image/jpeg';

  // Sample captions for demo mode
  var sampleCaptions = {
    descriptive: "A visually striking image with rich detail and vibrant composition. The subject is well-lit with natural lighting, showcasing clear textures and depth. The overall mood is inviting and professionally captured, suitable for editorial or commercial use.",
    social:      "This view is absolutely 🔥 Can't stop staring! Which one is your favorite part? Drop a comment below 👇✨ #Amazing #Viral #MustSee",
    alt:         "Image showing the main subject clearly positioned in the foreground with a blurred background, captured in natural daylight conditions.",
    short:       "Pure perfection. No filter needed.",
    seo:         "High-quality image featuring detailed subject matter with excellent composition and lighting. Perfect for web use, blog posts, and digital marketing campaigns targeting visual content consumers.",
    hindi:       "यह तस्वीर बेहद खूबसूरत है! रोशनी और रंगों का संयोजन इसे और भी आकर्षक बनाता है। एक पल जो हमेशा याद रहेगा। 📸✨"
  };

  function previewImage(event) {
    var file = event.target.files[0];
    if (!file) return;
    imageMime = file.type || 'image/jpeg';
    var reader = new FileReader();
    reader.onload = function() {
      var preview = document.getElementById('imagePreview');
      preview.src = reader.result;
      preview.style.display = 'block';
      document.getElementById('dropZone').style.padding = '16px';
      // Extract base64
      imageB64 = reader.result.split(',')[1];
    };
    reader.readAsDataURL(file);
  }

  // Drag & drop support
  var dz = document.getElementById('dropZone');
  dz.addEventListener('dragover',  function(e){ e.preventDefault(); dz.style.borderColor = '#4f46e5'; });
  dz.addEventListener('dragleave', function()  { dz.style.borderColor = '#c7d2fe'; });
  dz.addEventListener('drop', function(e) {
    e.preventDefault();
    dz.style.borderColor = '#c7d2fe';
    var file = e.dataTransfer.files[0];
    if (file && file.type.startsWith('image/')) {
      document.getElementById('fileInput').files = e.dataTransfer.files;
      previewImage({ target: { files: e.dataTransfer.files } });
    }
  });

  function getStylePrompt(style, lang) {
    var prompts = {
      descriptive: 'Write a detailed, professional descriptive caption for this image in ' + lang + '. Describe the subject, lighting, mood, and composition in 2-3 sentences.',
      social:      'Write a viral social media caption for this image in ' + lang + '. Make it engaging, use emojis, and add a call-to-action. Also suggest 5 relevant hashtags.',
      alt:         'Write a concise web accessibility alt-text description for this image in ' + lang + '. Focus on what a visually impaired person needs to know. Keep it under 125 characters.',
      short:       'Write a short, punchy, memorable caption for this image in ' + lang + '. Maximum 10 words. Make it impactful.',
      seo:         'Write an SEO-optimized image caption and description in ' + lang + '. Include relevant keywords naturally. Write 2-3 sentences suitable for a website or blog.',
      hindi:       'इस तस्वीर के लिए हिंदी में एक सुंदर और भावनापूर्ण कैप्शन लिखें। 2-3 वाक्य लिखें और 3-4 हैशटैग भी सुझाएं।'
    };
    return prompts[style] || prompts.descriptive;
  }

  function generateCaption() {
    if (!imageB64) { alert('Please upload an image first!'); return; }
    if (credits <= 0) { alert('No credits remaining!'); return; }

    var apiKey = document.getElementById('apiKey').value.trim();
    var style  = document.getElementById('captionStyle').value;
    var lang   = document.getElementById('captionLang').value;
    var btn    = document.getElementById('genBtn');
    var out    = document.getElementById('outputCard');
    var res    = document.getElementById('captionResult');
    var tags   = document.getElementById('hashtagResult');

    // Show output card
    out.style.display = 'block';
    res.innerHTML = '<span class="spin">⏳</span> Analyzing image...';
    tags.innerText = '';
    btn.disabled   = true;
    btn.innerHTML  = '<span class="spin">⏳</span> Analyzing...';

    if (!apiKey || !apiKey.startsWith('sk-ant-')) {
      // Demo mode
      setTimeout(function() {
        res.innerText  = sampleCaptions[style] || sampleCaptions.descriptive;
        tags.innerText = style === 'social' ? '#Photography #Viral #Amazing #Trending #MustSee' : '';
        finishGeneration(false);
      }, 1800);
      return;
    }

    // Real API call
    var prompt = getStylePrompt(style, lang);
    fetch('https://api.anthropic.com/v1/messages', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        model: 'claude-sonnet-4-20250514',
        max_tokens: 400,
        messages: [{
          role: 'user',
          content: [
            { type: 'image', source: { type: 'base64', media_type: imageMime, data: imageB64 } },
            { type: 'text',  text: prompt }
          ]
        }]
      })
    })
    .then(function(r){ return r.json(); })
    .then(function(data) {
      if (data.content && data.content[0]) {
        var text = data.content[0].text || '';
        // Separate hashtags if present
        var hashMatch = text.match(/#[\w\u0900-\u097F]+(\s+#[\w\u0900-\u097F]+)*/g);
        if (hashMatch) {
          res.innerText  = text.replace(hashMatch.join(' '), '').trim();
          tags.innerText = hashMatch.join(' ');
        } else {
          res.innerText  = text;
          tags.innerText = '';
        }
        document.getElementById('apiNotice').className = 'notice green';
        document.getElementById('apiNotice').innerText = '✅ Real AI Caption generated via Claude Vision API!';
        finishGeneration(true);
      } else {
        res.innerText = data.error ? ('API Error: ' + data.error.message) : 'Unexpected response. Try again.';
        finishGeneration(false);
      }
    })
    .catch(function(err) {
      res.innerText = '❌ Network error. Check your API key or internet connection.';
      document.getElementById('apiNotice').className = 'notice red';
      document.getElementById('apiNotice').innerText = '❌ API call failed. Check your key at console.anthropic.com';
      finishGeneration(false);
    });
  }

  function finishGeneration(success) {
    var btn = document.getElementById('genBtn');
    btn.disabled  = false;
    btn.innerHTML = '✨ Analyze &amp; Generate Caption (1 Credit)';
    if (success) {
      credits--;
      document.getElementById('creditCount').innerText = credits;
    }
  }

  function copyCaption() {
    var text = document.getElementById('captionResult').innerText +
               '\n' + document.getElementById('hashtagResult').innerText;
    if (navigator.clipboard) {
      navigator.clipboard.writeText(text.trim()).then(function() { alert('Caption copied! ✅'); });
    } else {
      var ta = document.createElement('textarea');
      ta.value = text.trim();
      document.body.appendChild(ta);
      ta.select();
      document.execCommand('copy');
      document.body.removeChild(ta);
      alert('Caption copied! ✅');
    }
  }
</script>
