// Counters
function animateCounter(el, target, suffix='') {
  let start = 0;
  const duration = 1800;
  const step = (timestamp) => {
    if (!start) start = timestamp;
    const progress = Math.min((timestamp - start) / duration, 1);
    const eased = 1 - Math.pow(1 - progress, 3);
    el.textContent = Math.floor(eased * target) + suffix;
    if (progress < 1) requestAnimationFrame(step);
  };
  requestAnimationFrame(step);
}

// Generate Presentation
async function generatePresentation() {
  const topic = document.getElementById('topicInput').value.trim();
  if (!topic) {
    document.getElementById('topicInput').style.borderColor = 'var(--accent)';
    setTimeout(() => document.getElementById('topicInput').style.borderColor = '', 1500);
    return;
  }
  const context = document.getElementById('contextInput').value.trim();
  const tone = document.getElementById('toneSelect').value;
  const count = document.getElementById('slideCount').value;

  const btn = document.getElementById('generateBtn');
  btn.classList.add('loading');
  btn.innerHTML = `<div class="spinner"></div><span>Generating…</span>`;

  document.getElementById('previewPlaceholder').style.display = 'none';
  const output = document.getElementById('slidesOutput');
  output.innerHTML = '';
  output.classList.remove('visible');

  const prompt = `You are an expert presentation designer. Generate a ${count}-slide presentation structure for the following topic.

Topic: ${topic}
${context ? `Context: ${context}` : ''}
Tone: ${tone}

Respond ONLY with a JSON array. No preamble, no markdown. Format:
[
  {
    "slideNumber": 1,
    "slideType": "Title Slide",
    "title": "...",
    "subtitle": "...",
    "layout": "Full-bleed title",
    "speakerNote": "..."
  }
]

Each slide must have: slideNumber, slideType (e.g. "Title Slide", "Agenda", "Problem Statement", "Key Insight", "Data Slide", "Case Study", "Solution", "Conclusion", "CTA"), title, subtitle (1 sentence), layout (layout hint), speakerNote (1-2 sentences for the speaker).

Return exactly ${count} slides. JSON only.`;

  try {
    const res = await fetch('https://api.anthropic.com/v1/messages', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        model: 'claude-sonnet-4-20250514',
        max_tokens: 1000,
        messages: [{ role: 'user', content: prompt }]
      })
    });
    const data = await res.json();
    const raw = data.content?.map(b => b.text || '').join('') || '';
    const clean = raw.replace(/```json|```/g, '').trim();
    const slides = JSON.parse(clean);

    slides.forEach((slide, i) => {
      const card = document.createElement('div');
      card.className = 'slide-card';
      card.style.animationDelay = (i * 0.08) + 's';
      card.innerHTML = `
        <div class="slide-card-num">${String(slide.slideNumber).padStart(2,'0')}</div>
        <div class="slide-card-content">
          <div class="slide-card-title">${slide.title || 'Untitled'}</div>
          <div class="slide-card-body">${slide.subtitle || ''}<br><em style="opacity:0.6;font-size:10px;">🗣 ${slide.speakerNote || ''}</em></div>
          <span class="slide-badge">${slide.slideType || 'Slide'}</span>
          <span class="slide-badge" style="margin-left:4px;">Layout: ${slide.layout || 'Standard'}</span>
        </div>`;
      output.appendChild(card);
    });
    output.classList.add('visible');
  } catch (err) {
    output.innerHTML = `<div class="error-msg">⚠ Could not generate slides. Please check your connection and try again.<br><small style="opacity:0.6;">${err.message}</small></div>`;
    output.classList.add('visible');
  }

  btn.classList.remove('loading');
  btn.innerHTML = `<span>✦</span><span>Generate Again</span>`;
}

document.getElementById('topicInput').addEventListener('keydown', e => {
  if (e.key === 'Enter') generatePresentation();
});
