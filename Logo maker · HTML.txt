/* ══════════════════════════════════════════
   MAIN DRAW
══════════════════════════════════════════ */
function v(id) { return document.getElementById(id); }

function updateCanvas() {
  const W = canvas.width, H = canvas.height;
  ctx.clearRect(0, 0, W, H);

  // Background
  ctx.fillStyle = state.bgColor;
  ctx.fillRect(0, 0, W, H);

  // Grid
  if (state.showGrid) {
    ctx.strokeStyle = 'rgba(0,0,0,0.06)';
    ctx.lineWidth = 1;
    for (let x=0; x<W; x+=20) { ctx.beginPath(); ctx.moveTo(x,0); ctx.lineTo(x,H); ctx.stroke(); }
    for (let y=0; y<H; y+=20) { ctx.beginPath(); ctx.moveTo(0,y); ctx.lineTo(W,y); ctx.stroke(); }
  }

  const cx = W/2 + state.offsetX;
  const cy = H/2 + state.offsetY;
  const opacity = parseInt(v('globalOpacity').value) / 100;
  ctx.globalAlpha = opacity;

  const layout = v('layoutSelect').value;
  const iconSz = parseInt(v('iconSize').value);
  const nameText = v('companyName').value;
  const tagText = v('tagline').value;
  const nameSz = parseInt(v('nameSize').value);
  const tagSz = parseInt(v('taglineSize').value);
  const ls = parseInt(v('letterSpacing').value);
  const lg = parseInt(v('lineGap').value);

  // compute positions
  let iconX = cx, iconY = cy;
  let textX = cx, textY = cy;

  if (layout === 'icon-top') {
    const totalH = iconSz + lg + nameSz + (tagText ? tagSz + 4 : 0);
    iconX = cx; iconY = cy - totalH/2 + iconSz/2;
    textX = cx; textY = iconY + iconSz/2 + lg;
  } else if (layout === 'icon-left') {
    const totalW = iconSz + lg + estimateTextW(nameText, nameSz);
    iconX = cx - totalW/2 + iconSz/2; iconY = cy;
    textX = iconX + iconSz/2 + lg; textY = cy;
  } else if (layout === 'icon-right') {
    const totalW = estimateTextW(nameText, nameSz) + lg + iconSz;
    textX = cx - totalW/2; textY = cy;
    iconX = cx + totalW/2 - iconSz/2; iconY = cy;
  } else if (layout === 'icon-only') {
    iconX = cx; iconY = cy; textX = -9999; textY = -9999;
  } else { // text-only
    iconX = -9999; iconY = -9999; textX = cx; textY = cy;
  }

  // Draw shape behind icon
  if (state.selectedShape !== 'none') drawShape(cx, cy);

  // Draw icon circle bg
  if (v('iconCircle').checked && layout !== 'text-only') {
    ctx.save();
    ctx.fillStyle = v('iconCircleColor').value;
    ctx.beginPath();
    ctx.arc(iconX, iconY, iconSz * 0.7, 0, Math.PI*2);
    ctx.fill();
    ctx.restore();
  }

  // Draw icon
  if (layout !== 'text-only' && iconX > -999) {
    ctx.save();
    ctx.translate(iconX, iconY);
    const rot = parseInt(v('iconRotation').value) * Math.PI / 180;
    ctx.rotate(rot);
    ctx.font = `${iconSz}px serif`;
    ctx.fillStyle = v('iconColor').value;
    ctx.textAlign = 'center';
    ctx.textBaseline = 'middle';
    ctx.fillText(state.selectedIcon, 0, 0);
    ctx.restore();
  }

  // Draw text
  if (layout !== 'icon-only' && textX > -999) {
    ctx.textAlign = layout === 'icon-left' || layout === 'icon-right' ? 'left' : 'center';

    // Name
    if (v('textShadow').checked) {
      ctx.shadowColor = 'rgba(0,0,0,0.2)';
      ctx.shadowBlur = 6;
      ctx.shadowOffsetY = 3;
    }
    const bold = v('boldName').checked ? 'bold ' : '';
    ctx.font = `${bold}${nameSz}px '${state.font}', sans-serif`;
    ctx.fillStyle = v('nameColor').value;
    if (ls !== 0) {
      drawTextWithSpacing(ctx, nameText, textX, textY + nameSz/3, ls);
    } else {
      ctx.fillText(nameText, textX, textY + nameSz/3);
    }
    ctx.shadowColor = 'transparent'; ctx.shadowBlur = 0; ctx.shadowOffsetY = 0;

    // Tagline
    if (tagText) {
      const italic = v('italicTagline').checked ? 'italic ' : '';
      ctx.font = `${italic}${tagSz}px '${state.font}', sans-serif`;
      ctx.fillStyle = v('taglineColor').value;
      ctx.fillText(tagText, textX, textY + nameSz/3 + nameSz/2 + lg + tagSz/2);
    }
  }

  ctx.globalAlpha = 1;
  pushHistory();
}

function estimateTextW(text, size) { return text.length * size * 0.55; }

function drawTextWithSpacing(ctx, text, x, y, spacing) {
  const chars = [...text];
  const totalW = chars.reduce((sum, c) => {
    ctx.save(); const w = ctx.measureText(c).width; ctx.restore(); return sum + w + spacing;
  }, 0) - spacing;
  let curX = ctx.textAlign === 'center' ? x - totalW/2 : x;
  const savedAlign = ctx.textAlign;
  ctx.textAlign = 'left';
  chars.forEach(c => {
    ctx.fillText(c, curX, y);
    curX += ctx.measureText(c).width + spacing;
  });
  ctx.textAlign = savedAlign;
}

function drawShape(cx, cy) {
  const s = state.selectedShape;
  const sz = parseInt(v('shapeSize').value);
  const half = sz / 2;
  ctx.save();

  // fill
  if (state.gradientFill !== null) {
    const [c1, c2] = GRADIENTS[state.gradientFill];
    const g = ctx.createLinearGradient(cx-half, cy-half, cx+half, cy+half);
    g.addColorStop(0, c1); g.addColorStop(1, c2);
    ctx.fillStyle = g;
  } else {
    ctx.fillStyle = v('shapeFill').value;
  }

  if (v('shapeShadow').checked) {
    ctx.shadowColor = 'rgba(0,0,0,0.25)';
    ctx.shadowBlur = 20;
    ctx.shadowOffsetY = 8;
  }

  ctx.beginPath();
  if (s === 'rect') {
    ctx.rect(cx-half, cy-half, sz, sz);
  } else if (s === 'rounded') {
    roundedRect(ctx, cx-half, cy-half, sz, sz, sz*0.15);
  } else if (s === 'circle') {
    ctx.arc(cx, cy, half, 0, Math.PI*2);
  } else if (s === 'ellipse') {
    ctx.ellipse(cx, cy, half, half*0.65, 0, 0, Math.PI*2);
  } else if (s === 'diamond') {
    ctx.moveTo(cx, cy-half); ctx.lineTo(cx+half, cy);
    ctx.lineTo(cx, cy+half); ctx.lineTo(cx-half, cy); ctx.closePath();
  } else if (s === 'hexagon') {
    hexPath(ctx, cx, cy, half);
  } else if (s === 'star') {
    starPath(ctx, cx, cy, 5, half, half*0.42);
  }
  ctx.fill();

  if (v('shapeBorder').checked) {
    ctx.shadowColor = 'transparent'; ctx.shadowBlur = 0;
    ctx.strokeStyle = v('shapeBorderColor').value;
    ctx.lineWidth = parseInt(v('shapeBorderW').value);
    ctx.stroke();
  }
  ctx.restore();
}

function roundedRect(ctx, x, y, w, h, r) {
  ctx.moveTo(x+r,y);
  ctx.lineTo(x+w-r,y); ctx.quadraticCurveTo(x+w,y,x+w,y+r);
  ctx.lineTo(x+w,y+h-r); ctx.quadraticCurveTo(x+w,y+h,x+w-r,y+h);
  ctx.lineTo(x+r,y+h); ctx.quadraticCurveTo(x,y+h,x,y+h-r);
  ctx.lineTo(x,y+r); ctx.quadraticCurveTo(x,y,x+r,y); ctx.closePath();
}
function hexPath(ctx, cx, cy, r) {
  for (let i=0; i<6; i++) {
    const a = Math.PI/180*(60*i-30);
    i===0 ? ctx.moveTo(cx+r*Math.cos(a),cy+r*Math.sin(a))
           : ctx.lineTo(cx+r*Math.cos(a),cy+r*Math.sin(a));
  }
  ctx.closePath();
}
function starPath(ctx, cx, cy, pts, outerR, innerR) {
  for (let i=0; i<pts*2; i++) {
    const r = i%2===0?outerR:innerR;
    const a = Math.PI/pts*i - Math.PI/2;
    i===0 ? ctx.moveTo(cx+r*Math.cos(a),cy+r*Math.sin(a))
           : ctx.lineTo(cx+r*Math.cos(a),cy+r*Math.sin(a));
  }
  ctx.closePath();
}

function alignLogo(dir) {
  const W = state.canvasW, H = state.canvasH;
  if (dir==='left') state.offsetX = -W/4;
  else if (dir==='right') state.offsetX = W/4;
  else if (dir==='centerH') state.offsetX = 0;
  else if (dir==='top') state.offsetY = -H/4;
  else if (dir==='bottom') state.offsetY = H/4;
  else if (dir==='centerV') state.offsetY = 0;
  updateCanvas();
}

/* ══════════════════════════════════════════
   DRAG TO MOVE LOGO
══════════════════════════════════════════ */
let dragging = false, dragStart = {x:0, y:0}, dragOffset = {x:0, y:0};
canvas.addEventListener('mousedown', e => {
  dragging = true;
  dragStart = { x: e.clientX, y: e.clientY };
  dragOffset = { x: state.offsetX, y: state.offsetY };
  canvas.style.cursor = 'grabbing';
});
window.addEventListener('mousemove', e => {
  if (!dragging) return;
  const dx = (e.clientX - dragStart.x) / state.zoom;
  const dy = (e.clientY - dragStart.y) / state.zoom;
  state.offsetX = dragOffset.x + dx;
  state.offsetY = dragOffset.y + dy;
  updateCanvas();
});
window.addEventListener('mouseup', () => { dragging = false; canvas.style.cursor = 'crosshair'; });

/* ══════════════════════════════════════════
   HISTORY (UNDO/REDO)
══════════════════════════════════════════ */
function pushHistory() {
  const snapshot = canvas.toDataURL();
  if (state.history[state.historyIndex] === snapshot) return;
  state.history = state.history.slice(0, state.historyIndex+1);
  state.history.push(snapshot);
  if (state.history.length > 30) state.history.shift();
  state.historyIndex = state.history.length - 1;
  renderHistoryList();
}

function undoAction() {
  if (state.historyIndex <= 0) return;
  state.historyIndex--;
  restoreSnapshot(state.historyIndex);
  renderHistoryList();
}
function redoAction() {
  if (state.historyIndex >= state.history.length-1) return;
  state.historyIndex++;
  restoreSnapshot(state.historyIndex);
  renderHistoryList();
}
function restoreSnapshot(idx) {
  const img = new Image();
  img.onload = () => { ctx.clearRect(0,0,canvas.width,canvas.height); ctx.drawImage(img,0,0); };
  img.src = state.history[idx];
}
function renderHistoryList() {
  const list = document.getElementById('historyList');
  list.innerHTML = state.history.slice(-8).map((_, i) => {
    const realIdx = Math.max(0, state.history.length-8) + i;
    return `<div class="history-item${realIdx===state.historyIndex?' current':''}" onclick="jumpHistory(${realIdx})">Step ${realIdx+1}</div>`;
  }).join('');
}
function jumpHistory(idx) {
  state.historyIndex = idx;
  restoreSnapshot(idx);
  renderHistoryList();
}

  state.selectedIcon = RAND_ICONS[Math.floor(Math.random()*RAND_ICONS.length)];

function randomHex() {

function buildTemplateGrid() {
  const grid = document.getElementById('templateGrid');
  grid.innerHTML = TEMPLATES.map((t, i) =>
    `<div class="template-item" onclick="TEMPLATES[${i}].fn()">
      <canvas id="tmpl${i}" width="120" height="80"></canvas>
      <div class="template-name">${t.name}</div>
    </div>`
  ).join('');
  // Render mini previews
  TEMPLATES.forEach((t, i) => {
    const c = document.getElementById('tmpl'+i);
    const c2 = c.getContext('2d');
    c2.fillStyle = t.fn ? '#eee' : '#fff';
    // Quick mini render
    renderMiniTemplate(c, c2, t);
  });
}

function renderMiniTemplate(c, c2, t) {
  // tiny preview - just colored bg + icon
  const bg = t.fn.toString().match(/bg:'([^']+)'/)?.[1] || '#fff';
  const icon = t.fn.toString().match(/icon:'([^']+)'/)?.[1] || '⬡';
  c2.fillStyle = bg;
  c2.fillRect(0,0,120,80);
  c2.font = '28px serif';
  c2.textAlign = 'center';
  c2.textBaseline = 'middle';
  c2.fillText(icon, 60, 35);
  c2.font = 'bold 9px sans-serif';
  c2.fillStyle = bg === '#fff' || bg === '#f8f9fa' ? '#222' : '#fff';
  c2.fillText('LogoCraft', 60, 63);
}

function openDownloadModal() { document.getElementById('downloadModal').classList.add('open'); }
function closeDownloadModal() { document.getElementById('downloadModal').classList.remove('open'); }

function downloadAs(fmt) {
  const scale = parseInt(v('exportScale').value);
  const offscreen = document.createElement('canvas');
  offscreen.width = canvas.width * scale;
  offscreen.height = canvas.height * scale;
  const octx = offscreen.getContext('2d');
  octx.scale(scale, scale);
  octx.drawImage(canvas, 0, 0);

  let mime, ext;
  if (fmt === 'png') { mime = 'image/png'; ext = 'png'; }
  else if (fmt === 'jpeg') { mime = 'image/jpeg'; ext = 'jpg'; }
  else if (fmt === 'webp') { mime = 'image/webp'; ext = 'webp'; }
  else if (fmt === 'svg') {
    downloadSVG(); return;
  }

  const a = document.createElement('a');
  a.download = `logo-${Date.now()}.${ext}`;
  a.href = offscreen.toDataURL(mime, 0.95);
  a.click();
  closeDownloadModal();
}

function downloadSVG() {
  const name = v('companyName').value;
  const tag = v('tagline').value;
  const svg = `<svg xmlns="http://www.w3.org/2000/svg" width="${state.canvasW}" height="${state.canvasH}">
  <rect width="100%" height="100%" fill="${state.bgColor}"/>
  <text x="50%" y="45%" font-size="${v('nameSize').value}" font-family="${state.font}" fill="${v('nameColor').value}" text-anchor="middle" dominant-baseline="middle" font-weight="${v('boldName').checked?'bold':'normal'}">${name}</text>
  <text x="50%" y="62%" font-size="${v('taglineSize').value}" font-family="${state.font}" fill="${v('taglineColor').value}" text-anchor="middle" dominant-baseline="middle">${tag}</text>
</svg>`;
  const blob = new Blob([svg], {type:'image/svg+xml'});
  const url = URL.createObjectURL(blob);
  const a = document.createElement('a');
  a.download = `logo-${Date.now()}.svg`;
  a.href = url;
  a.click();
  URL.revokeObjectURL(url);
  closeDownloadModal();
}
