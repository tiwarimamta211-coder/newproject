<script>
  var conversationHistory = [];
  var tokens = 3500000;
  var isGenerating = false;

  // Demo responses
  var demoResponses = {
    'gst': 'GST (Goods and Services Tax) is a comprehensive indirect tax levied on the supply of goods and services in India. It replaced multiple cascading taxes and was implemented on July 1, 2017. The main GST slabs are 0%, 5%, 12%, 18%, and 28%.\n\n**Key components:**\n- CGST: Central GST (collected by Central Govt)\n- SGST: State GST (collected by State Govt)\n- IGST: Integrated GST (for inter-state transactions)\n\nWould you like to know more about a specific aspect of GST?',
    'itc': 'Input Tax Credit (ITC) allows businesses to reduce their tax liability by the amount of GST paid on inputs. For example, if you pay ₹18,000 GST on purchases and collect ₹25,000 GST on sales, you only pay ₹7,000 to the government.\n\n**Conditions to claim ITC:**\n1. Must have a valid tax invoice\n2. Goods/services must be received\n3. Supplier must have filed their returns\n4. You must file GSTR-3B\n\nITC cannot be claimed on personal expenses, food & beverages, and certain exempt items.',
    'default': 'That\'s a great question! In demo mode, I provide sample responses. For real AI-powered answers from Claude, please add your Anthropic API Key in the field above.\n\nYou can get a free API key at **console.anthropic.com** — it takes just 2 minutes to sign up!\n\nIn the meantime, feel free to explore the quick prompts above for demo responses on GST and Finance topics. 😊'
  };

  function getDemoResponse(text) {
    var lower = text.toLowerCase();
    if (lower.includes('itc') || lower.includes('input tax')) return demoResponses.itc;
    if (lower.includes('gst') || lower.includes('gstin') || lower.includes('gstr')) return demoResponses.gst;
    return demoResponses.default;
  }

  function setPrompt(text) {
    document.getElementById('msgInput').value = text;
    autoResize(document.getElementById('msgInput'));
    document.getElementById('msgInput').focus();
  }

  function autoResize(el) {
    el.style.height = 'auto';
    el.style.height = Math.min(el.scrollHeight, 120) + 'px';
  }

  function handleKey(e) {
    if (e.key === 'Enter' && !e.shiftKey) { e.preventDefault(); sendMessage(); }
  }

  function addMessage(role, text, meta) {
    var win = document.getElementById('chatWindow');
    var div = document.createElement('div');
    div.className = 'msg ' + role;
    if (meta) {
      var metaEl = document.createElement('div');
      metaEl.className = 'msg-meta';
      metaEl.innerText = meta;
      div.appendChild(metaEl);
    }
    var textEl = document.createElement('div');
    textEl.innerText = text;
    div.appendChild(textEl);
    win.appendChild(div);
    win.scrollTop = win.scrollHeight;
    return textEl;
  }

  function addTyping() {
    var win = document.getElementById('chatWindow');
    var div = document.createElement('div');
    div.className = 'msg ai typing';
    div.id = 'typingIndicator';
    div.innerHTML = '<div class="typing-dot"></div><div class="typing-dot"></div><div class="typing-dot"></div>';
    win.appendChild(div);
    win.scrollTop = win.scrollHeight;
  }

  function removeTyping() {
    var t = document.getElementById('typingIndicator');
    if (t) t.remove();
  }

  function streamText(textEl, fullText, callback) {
    var i = 0;
    var win = document.getElementById('chatWindow');
    var interval = setInterval(function() {
      textEl.innerText += fullText[i];
      i++;
      win.scrollTop = win.scrollHeight;
      if (i >= fullText.length) {
        clearInterval(interval);
        if (callback) callback();
      }
    }, 18);
  }

  function sendMessage() {
    if (isGenerating) return;
    var input  = document.getElementById('msgInput');
    var apiKey = document.getElementById('apiKey').value.trim();
    var model  = document.getElementById('modelSelect').value;
    var system = document.getElementById('systemPrompt').value.trim();
    var text   = input.value.trim();
    if (!text) return;

    // Add user message
    addMessage('user', text, '👤 You');
    conversationHistory.push({ role: 'user', content: text });

    // Deduct tokens (estimate)
    tokens -= Math.max(50, text.length * 3);
    document.getElementById('tokenCount').innerText = Math.max(0, tokens).toLocaleString('en-IN');

    input.value = '';
    input.style.height = 'auto';
    isGenerating = true;
    document.getElementById('sendBtn').disabled = true;
    addTyping();

    if (!apiKey || !apiKey.startsWith('sk-ant-')) {
      // Demo mode
      setTimeout(function() {
        removeTyping();
        var aiMsgDiv = document.createElement('div');
        aiMsgDiv.className = 'msg ai';
        var metaEl = document.createElement('div');
        metaEl.className = 'msg-meta';
        metaEl.innerText = '🧠 NeuralStream AI (Demo)';
        aiMsgDiv.appendChild(metaEl);
        var textEl = document.createElement('div');
        aiMsgDiv.appendChild(textEl);
        document.getElementById('chatWindow').appendChild(aiMsgDiv);
        var response = getDemoResponse(text);
        streamText(textEl, response, function() {
          conversationHistory.push({ role: 'assistant', content: response });
          finishGeneration();
        });
      }, 900);
      return;
    }

    // Real Claude API
    var messages = conversationHistory.slice(-20); // last 20 messages for context
    var body = { model: model, max_tokens: 1000, messages: messages };
    if (system) body.system = system;

    fetch('https://api.anthropic.com/v1/messages', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(body)
    })
    .then(function(r){ return r.json(); })
    .then(function(data) {
      removeTyping();
      if (data.content && data.content[0]) {
        var reply = data.content[0].text || '';
        var aiMsgDiv = document.createElement('div');
        aiMsgDiv.className = 'msg ai';
        var metaEl = document.createElement('div');
        metaEl.className = 'msg-meta';
        metaEl.innerText = '🧠 NeuralStream AI';
        aiMsgDiv.appendChild(metaEl);
        var textEl = document.createElement('div');
        aiMsgDiv.appendChild(textEl);
        document.getElementById('chatWindow').appendChild(aiMsgDiv);
        // Update token count from API response
        if (data.usage) {
          tokens -= (data.usage.input_tokens + data.usage.output_tokens);
          document.getElementById('tokenCount').innerText = Math.max(0,tokens).toLocaleString('en-IN');
        }
        document.getElementById('apiStatus').className = 'api-status live';
        document.getElementById('apiStatus').innerText = '✅ Live AI';
        streamText(textEl, reply, function() {
          conversationHistory.push({ role: 'assistant', content: reply });
          finishGeneration();
        });
      } else {
        addMessage('ai', '❌ ' + (data.error ? data.error.message : 'Unexpected error. Try again.'), '⚠️ Error');
        finishGeneration();
      }
    })
    .catch(function() {
      removeTyping();
      addMessage('ai', '❌ Network error. Check your API key and internet connection.', '⚠️ Error');
      finishGeneration();
    });
  }

  function finishGeneration() {
    isGenerating = false;
    document.getElementById('sendBtn').disabled = false;
    document.getElementById('msgInput').focus();
  }

  function clearChat() {
    if (!confirm('Clear all chat history?')) return;
    conversationHistory = [];
    var win = document.getElementById('chatWindow');
    win.innerHTML = '';
    var div = document.createElement('div');
    div.className = 'msg ai';
    div.innerHTML = '<div class="msg-meta">🧠 NeuralStream AI</div>Chat cleared! How can I help you? 😊';
    win.appendChild(div);
  }
</script>
